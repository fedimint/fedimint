1use devimint::cmd;
2use devimint::federation::Client;
3use devimint::util::{FedimintCli, FedimintdCmd};
4use devimint::version_constants::{VERSION_0_3_0_ALPHA, VERSION_0_4_0};
5use fedimint_logging::LOG_TEST;
6use futures::try_join;
7use tracing::{debug, info};
8
9#[tokio::main]
10async fn main() -> anyhow::Result<()> {
11    devimint::run_devfed_test(|dev_fed, _process_mgr| async move {
12        let fedimint_cli_version = FedimintCli::version_or_default().await;
13        let fedimintd_version = FedimintdCmd::version_or_default().await;
14        // TODO(support:v0.4): recovery was introduced in v0.4.0
15        // see: https://github.com/fedimint/fedimint/pull/5546
16        if fedimint_cli_version < *VERSION_0_4_0 || fedimintd_version < *VERSION_0_4_0 {
17            info!(target: LOG_TEST, "Skipping whole test on old fedimint-cli/fedimintd that is missing some irrelevant bolts");
18            return Ok(());
19        }
20
21        let (fed, _bitcoind) = try_join!(dev_fed.fed(), dev_fed.bitcoind())?;
22
23        let peg_in_amount_sats = 100_000;
24
25            // Start this client early, as we need to test waiting for session to close
26        let reference_client = fed
27            .new_joined_client("wallet-client-recovery-origin")
28            .await?;
29        info!(target: LOG_TEST, "Join and claim");
30        fed.pegin_client(peg_in_amount_sats, &reference_client).await?;
31
32
33        let secret = cmd!(reference_client, "print-secret").out_json().await?["secret"]
34            .as_str()
35            .map(ToOwned::to_owned)
36            .unwrap();
37
38        let pre_notes = cmd!(reference_client, "info").out_json().await?;
39
40        let pre_balance = pre_notes["total_amount_msat"].as_u64().unwrap();
41
42        debug!(target: LOG_TEST, %pre_notes, pre_balance, "State before backup");
43
44        // we need to have some funds
45        // TODO: right now we rely on previous tests to leave some balance
46        assert!(0 < pre_balance);
47
48        // without existing backup
49        // TODO: Change this test and make them exercise more scenarios.
50        // Currently (and probably indefinitely) we can support only one
51        // restoration per client state (datadir), as it only makes sense to do
52        // once (at the very beginning) and we used a fixed operation id for it.
53        // Testing restore in different setups would require multiple clients,
54        // which is a larger refactor.
55        {
56            let post_balance = if fedimint_cli_version >= *VERSION_0_3_0_ALPHA {
57                let client = Client::create("restore-without-backup").await?;
58                let _ = cmd!(
59                    client,
60                    "restore",
61                    "--mnemonic",
62                    &secret,
63                    "--invite-code",
64                    fed.invite_code()?
65                )
66                .out_json()
67                .await?;
68
69                // `wait-complete` was introduced in v0.3.0 (90f3082)
70                let _ = cmd!(client, "dev", "wait-complete").out_json().await?;
71                let post_notes = cmd!(client, "info").out_json().await?;
72                let post_balance = post_notes["total_amount_msat"].as_u64().unwrap();
73                debug!(target: LOG_TEST, %post_notes, post_balance, "State after backup");
74
75                post_balance
76            } else {
77                let client = reference_client
78                    .new_forked("restore-without-backup")
79                    .await?;
80                let _ = cmd!(client, "wipe", "--force",).out_json().await?;
81
82                assert_eq!(
83                    0,
84                    cmd!(client, "info").out_json().await?["total_amount_msat"]
85                        .as_u64()
86                        .unwrap()
87                );
88
89                let post_balance = cmd!(client, "restore", &secret,)
90                    .out_json()
91                    .await?
92                    .as_u64()
93                    .unwrap();
94                let post_notes = cmd!(client, "info").out_json().await?;
95                debug!(target: LOG_TEST, %post_notes, post_balance, "State after backup");
96
97                post_balance
98            };
99            assert_eq!(pre_balance, post_balance);
100        }
101
102        // with a backup
103        {
104            if fedimint_cli_version >= *VERSION_0_3_0_ALPHA {
105                let _ = cmd!(reference_client, "backup",).out_json().await?;
106                let client = Client::create("restore-with-backup").await?;
107
108                {
109                    let _ = cmd!(
110                        client,
111                        "restore",
112                        "--mnemonic",
113                        &secret,
114                        "--invite-code",
115                        fed.invite_code()?
116                    )
117                    .out_json()
118                    .await?;
119
120                    let _ = cmd!(client, "dev", "wait-complete").out_json().await?;
121                    let post_notes = cmd!(client, "info").out_json().await?;
122                    let post_balance = post_notes["total_amount_msat"].as_u64().unwrap();
123                    debug!(target: LOG_TEST, %post_notes, post_balance, "State after backup");
124
125                    assert_eq!(pre_balance, post_balance);
126                }
127
128                // Now make a backup using the just restored client, and confirm restoring again
129                // still works (no corruption was introduced)
130                let _ = cmd!(client, "backup",).out_json().await?;
131
132                const EXTRA_PEGIN_SATS: u64 = 1000;
133                fed.pegin_client(EXTRA_PEGIN_SATS, &client).await?;
134
135                {
136                    let client = Client::create("restore-with-backup-again").await?;
137                    let _ = cmd!(
138                        client,
139                        "restore",
140                        "--mnemonic",
141                        &secret,
142                        "--invite-code",
143                        fed.invite_code()?
144                    )
145                    .out_json()
146                    .await?;
147
148                    let _ = cmd!(client, "dev", "wait-complete").out_json().await?;
149                    let post_notes = cmd!(client, "info").out_json().await?;
150                    let post_balance = post_notes["total_amount_msat"].as_u64().unwrap();
151                    debug!(target: LOG_TEST, %post_notes, post_balance, "State after (subsequent) backup");
152
153                    assert_eq!(pre_balance + EXTRA_PEGIN_SATS * 1000, post_balance);
154                }
155            } else {
156                let client = reference_client.new_forked("restore-with-backup").await?;
157                let _ = cmd!(client, "backup",).out_json().await?;
158                let _ = cmd!(client, "wipe", "--force",).out_json().await?;
159                assert_eq!(
160                    0,
161                    cmd!(client, "info").out_json().await?["total_amount_msat"]
162                        .as_u64()
163                        .unwrap()
164                );
165                let _ = cmd!(client, "restore", &secret,).out_json().await?;
166                let post_notes = cmd!(client, "info").out_json().await?;
167                let post_balance = post_notes["total_amount_msat"].as_u64().unwrap();
168                debug!(target: LOG_TEST, %post_notes, post_balance, "State after backup");
169
170                assert_eq!(pre_balance, post_balance);
171            }
172        }
173
174        Ok(())
175    })
176    .await
177}
