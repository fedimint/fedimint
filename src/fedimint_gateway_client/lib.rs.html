57impl GatewayIrohConnector {
58    pub async fn new(
59        iroh_pk: iroh::PublicKey,
60        password: Option<String>,
61        iroh_dns: Option<SafeUrl>,
62    ) -> anyhow::Result<Self> {
63        let mut builder = Endpoint::builder();
64
65        let iroh_dns_servers: Vec<_> = iroh_dns.map_or_else(
66            || {
67                FM_IROH_DNS_FEDIMINT_PROD
68                    .into_iter()
69                    .map(|url| Url::parse(url).expect("Hardcoded, can't fail"))
70                    .collect()
71            },
72            |url| vec![url.to_unsafe()],
73        );
74
75        for iroh_dns in iroh_dns_servers {
76            builder = builder.add_discovery(|_| Some(PkarrResolver::new(iroh_dns)));
77        }
78
79        // As a client, we don't need to register on any relays
80        let mut builder = builder.relay_mode(iroh::RelayMode::Disabled);
81
82        // See <https://github.com/fedimint/fedimint/issues/7811>
83        if is_env_var_set(FM_IROH_ENABLE_DHT_ENV) {
84            #[cfg(not(target_family = "wasm"))]
85            {
86                builder = builder.discovery_dht();
87            }
88        } else {
89            info!(
90                target: LOG_NET_IROH,
91                "Iroh DHT is disabled"
92            );
93        }
94
95        // instead of `.discovery_n0`, which brings publisher we don't want
96        {
97            #[cfg(target_family = "wasm")]
98            {
99                builder = builder.add_discovery(move |_| Some(PkarrResolver::n0_dns()));
100            }
101
102            #[cfg(not(target_family = "wasm"))]
103            {
104                builder = builder
105                    .add_discovery(move |_| Some(iroh::discovery::dns::DnsDiscovery::n0_dns()));
106            }
107        }
108
109        let endpoint = builder.bind().await?;
110
111        Ok(Self {
112            node_id: iroh_pk,
113            endpoint,
114            password,
115            connection_overrides: BTreeMap::new(),
116        })
117    }
118
119    pub fn with_connection_override(mut self, node: NodeId, addr: NodeAddr) -> Self {
120        self.connection_overrides.insert(node, addr);
121        self
122    }
123
124    async fn connect(&self) -> anyhow::Result<Connection> {
125        let connection = match self.connection_overrides.get(&self.node_id) {
126            Some(node_addr) => {
127                trace!(target: LOG_NET_IROH, node_id = %self.node_id, "Using a connectivity override for connection");
128                self.endpoint
129                    .connect(node_addr.clone(), FEDIMINT_GATEWAY_ALPN)
130                    .await?
131            }
132            None => {
133                self.endpoint
134                    .connect(self.node_id, FEDIMINT_GATEWAY_ALPN)
135                    .await?
136            }
137        };
138
139        // TODO: Spawn connection monitoring?
140        Ok(connection)
141    }
142
143    pub async fn request(
144        &self,
145        route: &str,
146        payload: Option<serde_json::Value>,
147    ) -> anyhow::Result<IrohGatewayResponse> {
148        let iroh_request = IrohGatewayRequest {
149            route: route.to_string(),
150            params: payload,
151            password: self.password.clone(),
152        };
153        let json = serde_json::to_vec(&iroh_request).expect("serialization cant fail");
154        let connection = self.connect().await?;
155        let (mut sink, mut stream) = connection.open_bi().await?;
156        sink.write_all(&json).await?;
157        sink.finish()?;
158        let response = stream.read_to_end(1_000_000).await?;
159        let iroh_response = serde_json::from_slice::<IrohGatewayResponse>(&response)?;
160        Ok(iroh_response)
161    }
162}

164impl GatewayRpcClient {
165    pub async fn new(
166        api: SafeUrl,
167        password: Option<String>,
168        iroh_dns: Option<SafeUrl>,
169        connection_override: Option<SafeUrl>,
170    ) -> anyhow::Result<Self> {
171        let mut base_url = api.clone();
172        let iroh_connector = if api.scheme() == "iroh" {
173            let host = api.host_str().context("Url is missing host")?;
174            let iroh_pk = iroh::PublicKey::from_str(host).context(format!(
175                "Could not parse Iroh Public key: Invalid public key: {host}"
176            ))?;
177            let mut iroh_connector =
178                GatewayIrohConnector::new(iroh_pk, password.clone(), iroh_dns).await?;
179
180            if let Some(connection_override) = connection_override {
181                let node_addr = NodeAddr {
182                    node_id: iroh_pk,
183                    relay_url: None,
184                    direct_addresses: BTreeSet::from([SocketAddr::V4(SocketAddrV4::new(
185                        connection_override
186                            .host_str()
187                            .ok_or(anyhow::anyhow!("No connection override host"))?
188                            .parse::<Ipv4Addr>()?,
189                        connection_override.port().ok_or(anyhow::anyhow!(
190                            "No iroh port supplied for connection override"
191                        ))?,
192                    ))]),
193                };
194
195                iroh_connector = iroh_connector.with_connection_override(iroh_pk, node_addr);
196            }
197            Some(iroh_connector)
198        } else {
199            base_url = base_url.join(V1_API_ENDPOINT)?;
200            None
201        };
202
203        Ok(Self {
204            base_url,
205            iroh_connector,
206            client: reqwest::Client::new(),
207            password,
208        })
209    }

366    async fn call<P: Serialize, T: DeserializeOwned>(
367        &self,
368        method: Method,
369        route: &str,
370        payload: Option<P>,
371    ) -> Result<T, GatewayRpcError> {
372        match &self.iroh_connector {
373            Some(iroh_connector) => {
374                let payload =
375                    payload.map(|p| serde_json::to_value(p).expect("Could not serialize"));
376                let response = iroh_connector
377                    .request(route, payload)
378                    .await
379                    .map_err(|e| GatewayRpcError::IrohError(e.to_string()))?;
380                let status_code = StatusCode::from_u16(response.status)
381                    .map_err(|e| GatewayRpcError::IrohError(e.to_string()))?;
382                match status_code {
383                    StatusCode::OK => {
384                        let response = serde_json::from_value::<T>(response.body)
385                            .map_err(|e| GatewayRpcError::IrohError(e.to_string()))?;
386                        Ok(response)
387                    }
388                    status => Err(GatewayRpcError::BadStatus(status)),
389                }
390            }
391            None => {
392                let url = self.base_url.join(route).expect("Invalid base url");
393                let mut builder = self.client.request(method, url.clone().to_unsafe());
394                if let Some(password) = self.password.clone() {
395                    builder = builder.bearer_auth(password);
396                }
397                if let Some(payload) = payload {
398                    builder = builder
399                        .json(&payload)
400                        .header(reqwest::header::CONTENT_TYPE, "application/json");
401                }
402
403                let response = builder.send().await?;
404
405                match response.status() {
406                    StatusCode::OK => Ok(response.json::<T>().await?),
407                    status => Err(GatewayRpcError::BadStatus(status)),
408                }
409            }
410        }
411    }

413    async fn call_get<T: DeserializeOwned>(&self, route: &str) -> Result<T, GatewayRpcError> {
414        self.call(Method::GET, route, None::<()>).await
415    }
416
417    async fn call_post<P: Serialize, T: DeserializeOwned>(
418        &self,
419        route: &str,
420        payload: P,
421    ) -> Result<T, GatewayRpcError> {
422        self.call(Method::POST, route, Some(payload)).await
423    }

426pub type GatewayRpcResult<T> = Result<T, GatewayRpcError>;

428#[derive(Error, Debug)]
429pub enum GatewayRpcError {
430    #[error("Bad status returned {0}")]
431    BadStatus(StatusCode),
432    #[error(transparent)]
433    RequestError(#[from] reqwest::Error),
434    #[error("Iroh error: {0}")]
435    IrohError(String),
