28pub struct GatewayRpcClient {
29    /// Base URL to gateway web server
30    /// This should include an applicable API version, e.g. http://localhost:8080/v1
31    base_url: SafeUrl,
32    /// A request client
33    client: reqwest::Client,
34    /// Optional gateway password
35    password: Option<String>,
36}

296    async fn call<P: Serialize, T: DeserializeOwned>(
297        &self,
298        method: Method,
299        url: SafeUrl,
300        payload: Option<P>,
301    ) -> Result<T, GatewayRpcError> {
302        let mut builder = self.client.request(method, url.clone().to_unsafe());
303        if let Some(password) = self.password.clone() {
304            builder = builder.bearer_auth(password);
305        }
306        if let Some(payload) = payload {
307            builder = builder
308                .json(&payload)
309                .header(reqwest::header::CONTENT_TYPE, "application/json");
310        }
311
312        let response = builder.send().await?;
313
314        match response.status() {
315            StatusCode::OK => Ok(response.json::<T>().await?),
316            status => Err(GatewayRpcError::BadStatus(status)),
317        }
318    }
319
320    async fn call_get<T: DeserializeOwned>(&self, url: SafeUrl) -> Result<T, GatewayRpcError> {
321        self.call(Method::GET, url, None::<()>).await
322    }
323
324    async fn call_post<P: Serialize, T: DeserializeOwned>(
325        &self,
326        url: SafeUrl,
327        payload: P,
328    ) -> Result<T, GatewayRpcError> {
329        self.call(Method::POST, url, Some(payload)).await
330    }
331}

333pub type GatewayRpcResult<T> = Result<T, GatewayRpcError>;
334
335#[derive(Error, Debug)]
336pub enum GatewayRpcError {
337    #[error("Bad status returned {0}")]
338    BadStatus(StatusCode),
339    #[error(transparent)]
340    RequestError(#[from] reqwest::Error),
