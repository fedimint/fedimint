160/// `fedimintd` builder
161///
162/// Fedimint supports third party modules. Right now (and for forseable feature)
163/// modules needs to be combined with the rest of the code at compilation time.
164///
165/// To make this easier, [`Fedimintd`] builder is exposed, allowing
166/// building `fedimintd` with custom set of modules.
167///
168///
169/// Example:
170///
171/// ```
172/// use fedimint_ln_server::LightningInit;
173/// use fedimint_mint_server::MintInit;
174/// use fedimint_wallet_server::WalletInit;
175/// use fedimintd::Fedimintd;
176///
177/// // Note: not called `main` to avoid rustdoc executing it
178/// // #[tokio::main]
179/// async fn main_() -> anyhow::Result<()> {
180///     Fedimintd::new(env!("FEDIMINT_BUILD_CODE_VERSION"), Some("vendor-xyz-1"))?
181///         // use `.with_default_modules()` to avoid having
182///         // to import these manually
183///         .with_module_kind(WalletInit)
184///         .with_module_kind(MintInit)
185///         .with_module_kind(LightningInit)
186///         .run()
187///         .await
188/// }
189/// ```
190pub struct Fedimintd {
191    server_gens: ServerModuleInitRegistry,
192    server_gen_params: ServerModuleConfigGenParamsRegistry,
193    code_version_hash: String,
194    code_version_str: String,
195    opts: ServerOpts,
196}
197
198impl Fedimintd {
199    /// Builds a new `fedimintd`
200    ///
201    /// # Arguments
202    ///
203    /// * `code_version_hash` - The git hash of the code that the `fedimintd`
204    ///   binary is being built from. This is used mostly for information
205    ///   purposes (`fedimintd version-hash`). See `fedimint-build` crate for
206    ///   easy way to obtain it.
207    ///
208    /// * `code_version_vendor_suffix` - An optional suffix that will be
209    ///   appended to the internal fedimint release version, to distinguish
210    ///   binaries built by different vendors, usually with a different set of
211    ///   modules. Currently DKG will enforce that the combined `code_version`
212    ///   is the same between all peers.
213    pub fn new(
214        code_version_hash: &str,
215        code_version_vendor_suffix: Option<&str>,
216    ) -> anyhow::Result<Fedimintd> {
217        assert_eq!(
218            env!("FEDIMINT_BUILD_CODE_VERSION").len(),
219            code_version_hash.len(),
220            "version_hash must have an expected length"
221        );
222
223        handle_version_hash_command(code_version_hash);
224
225        let fedimint_version = env!("CARGO_PKG_VERSION");
226
227        APP_START_TS
228            .with_label_values(&[fedimint_version, code_version_hash])
229            .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
230
231        let opts = ServerOpts::parse();
232
233        let mut tracing_builder = TracingSetup::default();
234
235        tracing_builder
236            .tokio_console_bind(opts.bind_tokio_console)
237            .with_jaeger(opts.with_jaeger);
238
239        tracing_builder.init().unwrap();
240
241        info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
242
243        Ok(Self {
244            opts,
245            server_gens: ServerModuleInitRegistry::new(),
246            server_gen_params: ServerModuleConfigGenParamsRegistry::default(),
247            code_version_hash: code_version_hash.to_owned(),
248            code_version_str: code_version_vendor_suffix.map_or_else(
249                || fedimint_version.to_string(),
250                |suffix| format!("{fedimint_version}+{suffix}"),
251            ),
252        })
253    }

255    /// Attach a server module kind to the Fedimintd instance
256    ///
257    /// This makes `fedimintd` support additional module types (aka. kinds)
258    pub fn with_module_kind<T>(mut self, r#gen: T) -> Self
259    where
260        T: ServerModuleInit + 'static + Send + Sync,
261    {
262        self.server_gens.attach(r#gen);
263        self
264    }

272    /// Attach additional module instance with parameters
273    ///
274    /// Note: The `kind` needs to be added with [`Self::with_module_kind`] if
275    /// it's not the default one.
276    pub fn with_module_instance<P>(mut self, kind: ModuleKind, params: P) -> Self

278        P: ModuleInitParams,

280        self.server_gen_params

285    /// Attach default server modules to Fedimintd instance
286    pub fn with_default_modules(self) -> anyhow::Result<Self> {
287        let network = self.opts.bitcoin_network;
288
289        let bitcoin_rpc_config = match (
290            self.opts.bitcoind_url.as_ref(),
291            self.opts.esplora_url.as_ref(),
292        ) {
293            (Some(url), None) => BitcoinRpcConfig {
294                kind: "bitcoind".to_string(),
295                url: url.clone(),
296            },
297            (None, Some(url)) => BitcoinRpcConfig {
298                kind: "esplora".to_string(),
299                url: url.clone(),
300            },
301            _ => unreachable!("ArgGroup already enforced XOR relation"),
302        };
303
304        let s = self
305            .with_module_kind(LightningInit)
306            .with_module_instance(
307                LightningInit::kind(),
308                LightningGenParams {
309                    local: LightningGenParamsLocal {
310                        bitcoin_rpc: bitcoin_rpc_config.clone(),
311                    },
312                    consensus: LightningGenParamsConsensus { network },
313                },
314            )
315            .with_module_kind(MintInit)
316            .with_module_instance(
317                MintInit::kind(),
318                MintGenParams {
319                    local: EmptyGenParams::default(),
320                    consensus: MintGenParamsConsensus::new(
321                        2,
322                        // TODO: wait for clients to support the relative fees and set them to
323                        // non-zero in 0.6
324                        fedimint_mint_common::config::FeeConsensus::zero(),
325                    ),
326                },
327            )
328            .with_module_kind(WalletInit)
329            .with_module_instance(
330                WalletInit::kind(),
331                WalletGenParams {
332                    local: WalletGenParamsLocal {
333                        bitcoin_rpc: bitcoin_rpc_config.clone(),
334                    },
335                    consensus: WalletGenParamsConsensus {
336                        network,
337                        finality_delay: 10,
338                        client_default_bitcoin_rpc: default_esplora_server(network),
339                        fee_consensus:
340                            fedimint_wallet_server::common::config::FeeConsensus::default(),
341                    },
342                },
343            );
344
345        let enable_lnv2 = std::env::var_os(FM_ENABLE_MODULE_LNV2_ENV).is_none()
346            || is_env_var_set(FM_ENABLE_MODULE_LNV2_ENV);
347
348        let s = if enable_lnv2 {
349            s.with_module_kind(fedimint_lnv2_server::LightningInit)
350                .with_module_instance(
351                    fedimint_lnv2_server::LightningInit::kind(),
352                    fedimint_lnv2_common::config::LightningGenParams {
353                        local: fedimint_lnv2_common::config::LightningGenParamsLocal {
354                            bitcoin_rpc: bitcoin_rpc_config.clone(),
355                        },
356                        consensus: fedimint_lnv2_common::config::LightningGenParamsConsensus {
357                            // TODO: actually make the relative fee configurable
358                            fee_consensus: fedimint_lnv2_common::config::FeeConsensus::new(100)?,
359                            network,
360                        },
361                    },
362                )
363        } else {
364            s
365        };
366
367        let s = if is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
368            s
369        } else {
370            s.with_module_kind(MetaInit)
371                .with_module_instance(MetaInit::kind(), MetaGenParams::default())
372        };
373
374        let s = if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
375            s.with_module_kind(UnknownInit)
376                .with_module_instance(UnknownInit::kind(), UnknownGenParams::default())
377        } else {
378            s
379        };
380
381        Ok(s)
382    }

384    /// Block thread and run a Fedimintd server
385    pub async fn run(self) -> ! {
386        let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
387
388        let root_task_group = TaskGroup::new();
389
390        if let Some(bind_metrics) = self.opts.bind_metrics.as_ref() {
391            root_task_group.spawn_cancellable(
392                "metrics-server",
393                fedimint_metrics::run_api_server(*bind_metrics, root_task_group.clone()),
394            );
395        }
396
397        let settings = ConfigGenSettings {
398            p2p_bind: self.opts.bind_p2p,
399            api_bind: self.opts.bind_api,
400            ui_bind: self.opts.bind_ui,
401            p2p_url: self.opts.p2p_url,
402            api_url: self.opts.api_url,
403            enable_iroh: self.opts.enable_iroh,
404            modules: self.server_gen_params.clone(),
405            registry: self.server_gens.clone(),
406        };
407
408        let db = Database::new(
409            RocksDb::open(self.opts.data_dir.join(DB_FILE))
410                .await
411                .unwrap(),
412            ModuleRegistry::default(),
413        );
414
415        let dyn_server_bitcoin_rpc = match (
416            self.opts.bitcoind_url.as_ref(),
417            self.opts.esplora_url.as_ref(),
418        ) {
419            (Some(url), None) => BitcoindClient::new(url).unwrap().into_dyn(),
420            (None, Some(url)) => EsploraClient::new(url).unwrap().into_dyn(),
421            _ => unreachable!("ArgGroup already enforced XOR relation"),
422        };
423
424        root_task_group.install_kill_handler();
425
426        fedimint_server::run(
427            self.opts.data_dir,
428            self.opts.force_api_secrets,
429            settings,
430            db,
431            self.code_version_str,
432            self.server_gens,
433            root_task_group.clone(),
434            dyn_server_bitcoin_rpc,
435            Box::new(fedimint_server_ui::setup::router),
436            Box::new(fedimint_server_ui::dashboard::router),
437            self.opts.db_checkpoint_retention,
438        )
439        .await
440        .inspect_err(|e| crit!(target: LOG_SERVER, ?e, "Main task returned error"))
441        .ok();
442
443        info!(target: LOG_CORE, "Awaiting shutdown of root task group");
444
445        root_task_group
446            .join_all(Some(SHUTDOWN_TIMEOUT))
447            .await
448            .inspect_err(|e| error!(target: LOG_CORE, ?e, "Error while shutting down task group"))
449            .ok();
450
451        debug!(target: LOG_CORE, "Shutdown complete");
452
453        fedimint_logging::shutdown();
454
455        drop(timing_total_runtime);
456
457        // Should we ever shut down without an error code?
458        std::process::exit(-1);
459    }
