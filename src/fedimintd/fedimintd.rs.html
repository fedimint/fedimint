138/// Parse a key-value map from a string.
139///
140/// The string should be a comma-separated list of key-value pairs, where each
141/// pair is separated by an equals sign. For example, `key1=value1,key2=value2`.
142/// The keys and values are trimmed of whitespace, so
143/// `key1 = value1, key2 = value2` would be parsed the same as the previous
144/// example.
145fn parse_map(s: &str) -> anyhow::Result<BTreeMap<String, String>> {
146    let mut map = BTreeMap::new();
147
148    if s.is_empty() {
149        return Ok(map);
150    }
151
152    for pair in s.split(',') {
153        let parts: Vec<&str> = pair.split('=').collect();
154
155        if parts.len() != 2 {
156            return Err(format_err!(
157                "Invalid key-value pair in map: '{}'. Expected format: 'key=value'.",
158                pair
159            ));
160        }
161
162        let key = parts[0].trim();
163        let value = parts[1].trim();
164
165        if let Some(previous_value) = map.insert(key.to_string(), value.to_string()) {
166            return Err(format_err!(
167                "Duplicate key in map: '{key}' (found values '{previous_value}' and '{value}')",
168            ));
169        }
170    }
171
172    Ok(map)
173}

175/// `fedimintd` builder
176///
177/// Fedimint supports third party modules. Right now (and for forseable feature)
178/// modules needs to be combined with the rest of the code at compilation time.
179///
180/// To make this easier, [`Fedimintd`] builder is exposed, allowing
181/// building `fedimintd` with custom set of modules.
182///
183///
184/// Example:
185///
186/// ```
187/// use fedimint_ln_server::LightningInit;
188/// use fedimint_mint_server::MintInit;
189/// use fedimint_wallet_server::WalletInit;
190/// use fedimintd::Fedimintd;
191///
192/// // Note: not called `main` to avoid rustdoc executing it
193/// // #[tokio::main]
194/// async fn main_() -> anyhow::Result<()> {
195///     Fedimintd::new(env!("FEDIMINT_BUILD_CODE_VERSION"), Some("vendor-xyz-1"))?
196///         // use `.with_default_modules()` to avoid having
197///         // to import these manually
198///         .with_module_kind(WalletInit)
199///         .with_module_kind(MintInit)
200///         .with_module_kind(LightningInit)
201///         .run()
202///         .await
203/// }
204/// ```
205pub struct Fedimintd {
206    server_gens: ServerModuleInitRegistry,
207    server_gen_params: ServerModuleConfigGenParamsRegistry,
208    code_version_hash: String,
209    code_version_str: String,
210    opts: ServerOpts,
211    bitcoind_rpc: BitcoinRpcConfig,
212}

214impl Fedimintd {
215    /// Builds a new `fedimintd`
216    ///
217    /// # Arguments
218    ///
219    /// * `code_version_hash` - The git hash of the code that the `fedimintd`
220    ///   binary is being built from. This is used mostly for information
221    ///   purposes (`fedimintd version-hash`). See `fedimint-build` crate for
222    ///   easy way to obtain it.
223    ///
224    /// * `code_version_vendor_suffix` - An optional suffix that will be
225    ///   appended to the internal fedimint release version, to distinguish
226    ///   binaries built by different vendors, usually with a different set of
227    ///   modules. Currently DKG will enforce that the combined `code_version`
228    ///   is the same between all peers.
229    pub fn new(
230        code_version_hash: &str,
231        code_version_vendor_suffix: Option<&str>,
232    ) -> anyhow::Result<Fedimintd> {
233        assert_eq!(
234            env!("FEDIMINT_BUILD_CODE_VERSION").len(),
235            code_version_hash.len(),
236            "version_hash must have an expected length"
237        );
238
239        handle_version_hash_command(code_version_hash);
240
241        let fedimint_version = env!("CARGO_PKG_VERSION");
242
243        APP_START_TS
244            .with_label_values(&[fedimint_version, code_version_hash])
245            .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
246
247        let opts: ServerOpts = ServerOpts::parse();
248
249        TracingSetup::default()
250            .tokio_console_bind(opts.tokio_console_bind)
251            .with_jaeger(opts.with_telemetry)
252            .init()
253            .unwrap();
254
255        info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
256
257        let bitcoind_rpc = BitcoinRpcConfig::get_defaults_from_env_vars()?;
258
259        Ok(Self {
260            opts,
261            bitcoind_rpc,
262            server_gens: ServerModuleInitRegistry::new(),
263            server_gen_params: ServerModuleConfigGenParamsRegistry::default(),
264            code_version_hash: code_version_hash.to_owned(),
265            code_version_str: code_version_vendor_suffix.map_or_else(
266                || fedimint_version.to_string(),
267                |suffix| format!("{fedimint_version}+{suffix}"),
268            ),
269        })
270    }

327                        // non-zero in 0.6

330                },
331            )
332            .with_module_kind(WalletInit)
333            .with_module_instance(
334                WalletInit::kind(),
335                WalletGenParams {
336                    local: WalletGenParamsLocal {
337                        bitcoin_rpc: bitcoind_rpc.clone(),

358                        },
359                        consensus: fedimint_lnv2_common::config::LightningGenParamsConsensus {
360                            // TODO: actually make the relative fee configurable
361                            fee_consensus: fedimint_lnv2_common::config::FeeConsensus::new(1_000)?,
362                            network,
363                        },
364                    },
365                )
366        } else {
367            s
368        };
369
370        let s = if is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
371            s
372        } else {
373            s.with_module_kind(MetaInit)
374                .with_module_instance(MetaInit::kind(), MetaGenParams::default())
375        };
376
377        let s = if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
378            s.with_module_kind(UnknownInit)
379                .with_module_instance(UnknownInit::kind(), UnknownGenParams::default())
380        } else {
381            s
382        };
383
384        Ok(s)
385    }

387    /// Block thread and run a Fedimintd server
388    pub async fn run(self) -> ! {
389        // handle optional subcommand
390        if let Some(subcommand) = &self.opts.subcommand {
391            match subcommand {
392                ServerSubcommand::Dev(DevSubcommand::ListApiVersions) => {
393                    let api_versions = self.get_server_api_versions();
394                    let api_versions = serde_json::to_string_pretty(&api_versions)
395                        .expect("API versions struct is serializable");
396                    println!("{api_versions}");
397                    std::process::exit(0);
398                }
399                ServerSubcommand::Dev(DevSubcommand::ListDbVersions) => {
400                    let db_versions = self.get_server_db_versions();
401                    let db_versions = serde_json::to_string_pretty(&db_versions)
402                        .expect("API versions struct is serializable");
403                    println!("{db_versions}");
404                    std::process::exit(0);
405                }
406            }
407        }
408
409        let root_task_group = TaskGroup::new();
410        root_task_group.install_kill_handler();
411
412        let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
413
414        let task_group = root_task_group.clone();
415        root_task_group.spawn_cancellable("main", async move {
416            match run(
417                self.opts,
418                &task_group,
419                self.server_gens,
420                self.server_gen_params,
421                self.code_version_str,
422            )
423            .await
424            {
425                Ok(()) => {}
426                Err(error) => {
427                    error!(?error, "Main task returned error, shutting down");
428                    task_group.shutdown();
429                }
430            }
431        });
432
433        let shutdown_future = root_task_group
434            .make_handle()
435            .make_shutdown_rx()
436            .then(|()| async {
437                info!(target: LOG_CORE, "Shutdown called");
438            });
439
440        shutdown_future.await;
441        debug!(target: LOG_CORE, "Terminating main task");
442
443        if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
444            error!(target: LOG_CORE, ?err, "Error while shutting down task group");
445        }
446
447        debug!(target: LOG_CORE, "Shutdown complete");
448
449        fedimint_logging::shutdown();
450
451        drop(timing_total_runtime);
452
453        // Should we ever shut down without an error code?
454        std::process::exit(-1);
455    }

457    fn get_server_api_versions(&self) -> ServerApiVersionsSummary {
458        ServerApiVersionsSummary {
459            core: ServerConfig::supported_api_versions().api,
460            modules: self
461                .server_gens
462                .kinds()
463                .into_iter()
464                .map(|module_kind| {
465                    self.server_gens
466                        .get(&module_kind)
467                        .expect("module is present")
468                })
469                .map(|module_init| {
470                    (
471                        module_init.module_kind(),
472                        module_init.supported_api_versions().api,
473                    )
474                })
475                .collect(),
476        }
477    }

501async fn run(
502    opts: ServerOpts,
503    task_group: &TaskGroup,
504    module_inits: ServerModuleInitRegistry,
505    module_inits_params: ServerModuleConfigGenParamsRegistry,
506    code_version_str: String,
507) -> anyhow::Result<()> {
508    if let Some(socket_addr) = opts.bind_metrics_api.as_ref() {
509        task_group.spawn_cancellable("metrics-server", {
510            let task_group = task_group.clone();
511            let socket_addr = *socket_addr;
512            async move { fedimint_metrics::run_api_server(socket_addr, task_group).await }
513        });
514    }
515
516    let data_dir = opts.data_dir.context("data-dir option is not present")?;
517
518    // TODO: Fedimintd should use the config gen API
519    // on each run we want to pass the currently passed password, so we need to
520    // overwrite
521    if let Some(password) = opts.password {
522        write_overwrite(data_dir.join(PLAINTEXT_PASSWORD), password)?;
523    };
524    let use_iroh = is_env_var_set(FM_FORCE_IROH_ENV);
525
526    // TODO: meh, move, refactor
527    let settings = ConfigGenSettings {
528        p2p_bind: opts.bind_p2p,
529        api_bind: opts.bind_api,
530        p2p_url: opts.p2p_url,
531        api_url: opts.api_url,
532        meta: opts.extra_dkg_meta.clone(),
533        modules: module_inits_params.clone(),
534        registry: module_inits.clone(),
535        networking: if use_iroh {
536            NetworkingStack::Iroh
537        } else {
538            NetworkingStack::default()
539        },
540    };
541
542    let db = Database::new(
543        fedimint_rocksdb::RocksDb::open(data_dir.join(DB_FILE))?,
544        ModuleRegistry::default(),
545    );
546
547    Box::pin(fedimint_server::run(
548        data_dir,
549        opts.force_api_secrets,
550        settings,
551        db,
552        code_version_str,
553        &module_inits,
554        task_group.clone(),
555    ))
556    .await
