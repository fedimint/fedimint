154/// `fedimintd` builder
155///
156/// Fedimint supports third party modules. Right now (and for forseable feature)
157/// modules needs to be combined with the rest of the code at compilation time.
158///
159/// To make this easier, [`Fedimintd`] builder is exposed, allowing
160/// building `fedimintd` with custom set of modules.
161///
162///
163/// Example:
164///
165/// ```
166/// use fedimint_ln_server::LightningInit;
167/// use fedimint_mint_server::MintInit;
168/// use fedimint_wallet_server::WalletInit;
169/// use fedimintd::Fedimintd;
170///
171/// // Note: not called `main` to avoid rustdoc executing it
172/// // #[tokio::main]
173/// async fn main_() -> anyhow::Result<()> {
174///     Fedimintd::new(env!("FEDIMINT_BUILD_CODE_VERSION"), Some("vendor-xyz-1"))?
175///         // use `.with_default_modules()` to avoid having
176///         // to import these manually
177///         .with_module_kind(WalletInit)
178///         .with_module_kind(MintInit)
179///         .with_module_kind(LightningInit)
180///         .run()
181///         .await
182/// }
183/// ```
184pub struct Fedimintd {
185    server_gens: ServerModuleInitRegistry,
186    server_gen_params: ServerModuleConfigGenParamsRegistry,
187    code_version_hash: String,
188    code_version_str: String,
189    opts: ServerOpts,
190}
191
192impl Fedimintd {
193    /// Builds a new `fedimintd`
194    ///
195    /// # Arguments
196    ///
197    /// * `code_version_hash` - The git hash of the code that the `fedimintd`
198    ///   binary is being built from. This is used mostly for information
199    ///   purposes (`fedimintd version-hash`). See `fedimint-build` crate for
200    ///   easy way to obtain it.
201    ///
202    /// * `code_version_vendor_suffix` - An optional suffix that will be
203    ///   appended to the internal fedimint release version, to distinguish
204    ///   binaries built by different vendors, usually with a different set of
205    ///   modules. Currently DKG will enforce that the combined `code_version`
206    ///   is the same between all peers.
207    pub fn new(
208        code_version_hash: &str,
209        code_version_vendor_suffix: Option<&str>,
210    ) -> anyhow::Result<Fedimintd> {
211        assert_eq!(
212            env!("FEDIMINT_BUILD_CODE_VERSION").len(),
213            code_version_hash.len(),
214            "version_hash must have an expected length"
215        );
216
217        handle_version_hash_command(code_version_hash);
218
219        let fedimint_version = env!("CARGO_PKG_VERSION");
220
221        APP_START_TS
222            .with_label_values(&[fedimint_version, code_version_hash])
223            .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
224
225        let opts = ServerOpts::parse();
226
227        let mut tracing_builder = TracingSetup::default();
228
229        tracing_builder
230            .tokio_console_bind(opts.bind_tokio_console)
231            .with_jaeger(opts.with_jaeger);
232
233        tracing_builder.init().unwrap();
234
235        info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
236
237        Ok(Self {
238            opts,
239            server_gens: ServerModuleInitRegistry::new(),
240            server_gen_params: ServerModuleConfigGenParamsRegistry::default(),
241            code_version_hash: code_version_hash.to_owned(),
242            code_version_str: code_version_vendor_suffix.map_or_else(
243                || fedimint_version.to_string(),
244                |suffix| format!("{fedimint_version}+{suffix}"),
245            ),
246        })
247    }

279    /// Attach default server modules to Fedimintd instance
280    pub fn with_default_modules(self) -> anyhow::Result<Self> {
281        let network = self.opts.bitcoin_network;
282
283        let bitcoin_rpc_config = match (
284            self.opts.bitcoind_url.as_ref(),
285            self.opts.esplora_url.as_ref(),
286        ) {
287            (Some(url), None) => BitcoinRpcConfig {
288                kind: "bitcoind".to_string(),
289                url: url.clone(),
290            },
291            (None, Some(url)) => BitcoinRpcConfig {
292                kind: "esplora".to_string(),
293                url: url.clone(),
294            },
295            _ => unreachable!("ArgGroup already enforced XOR relation"),
296        };
297
298        let s = self
299            .with_module_kind(LightningInit)
300            .with_module_instance(
301                LightningInit::kind(),
302                LightningGenParams {
303                    local: LightningGenParamsLocal {
304                        bitcoin_rpc: bitcoin_rpc_config.clone(),
305                    },
306                    consensus: LightningGenParamsConsensus { network },
307                },
308            )
309            .with_module_kind(MintInit)
310            .with_module_instance(
311                MintInit::kind(),
312                MintGenParams {
313                    local: EmptyGenParams::default(),
314                    consensus: MintGenParamsConsensus::new(
315                        2,
316                        // TODO: wait for clients to support the relative fees and set them to
317                        // non-zero in 0.6
318                        fedimint_mint_common::config::FeeConsensus::zero(),
319                    ),
320                },
321            )
322            .with_module_kind(WalletInit)
323            .with_module_instance(
324                WalletInit::kind(),
325                WalletGenParams {
326                    local: WalletGenParamsLocal {
327                        bitcoin_rpc: bitcoin_rpc_config.clone(),
328                    },
329                    consensus: WalletGenParamsConsensus {
330                        network,
331                        finality_delay: 10,
332                        client_default_bitcoin_rpc: default_esplora_server(network),
333                        fee_consensus:
334                            fedimint_wallet_server::common::config::FeeConsensus::default(),
335                    },
336                },
337            );
338
339        let enable_lnv2 = std::env::var_os(FM_ENABLE_MODULE_LNV2_ENV).is_none()
340            || is_env_var_set(FM_ENABLE_MODULE_LNV2_ENV);
341
342        let s = if enable_lnv2 {
343            s.with_module_kind(fedimint_lnv2_server::LightningInit)
344                .with_module_instance(
345                    fedimint_lnv2_server::LightningInit::kind(),
346                    fedimint_lnv2_common::config::LightningGenParams {
347                        local: fedimint_lnv2_common::config::LightningGenParamsLocal {
348                            bitcoin_rpc: bitcoin_rpc_config.clone(),
349                        },
350                        consensus: fedimint_lnv2_common::config::LightningGenParamsConsensus {
351                            // TODO: actually make the relative fee configurable
352                            fee_consensus: fedimint_lnv2_common::config::FeeConsensus::new(100)?,
353                            network,
354                        },
355                    },
356                )
357        } else {
358            s
359        };
360
361        let s = if is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
362            s
363        } else {
364            s.with_module_kind(MetaInit)
365                .with_module_instance(MetaInit::kind(), MetaGenParams::default())
366        };
367
368        let s = if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
369            s.with_module_kind(UnknownInit)
370                .with_module_instance(UnknownInit::kind(), UnknownGenParams::default())
371        } else {
372            s
373        };
374
375        Ok(s)
376    }

378    /// Block thread and run a Fedimintd server
379    pub async fn run(self) -> ! {
380        let root_task_group = TaskGroup::new();
381        root_task_group.install_kill_handler();
382
383        let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
384
385        let task_group = root_task_group.clone();
386        root_task_group.spawn_cancellable("main", async move {
387            match run(
388                self.opts,
389                &task_group,
390                self.server_gens,
391                self.server_gen_params,
392                self.code_version_str,
393            )
394            .await
395            {
396                Ok(()) => {}
397                Err(error) => {
398                    crit!(target: LOG_SERVER, err = %error.fmt_compact_anyhow(), "Main task returned error, shutting down");
399                    task_group.shutdown();
400                }
401            }
402        });
403
404        let shutdown_future = root_task_group
405            .make_handle()
406            .make_shutdown_rx()
407            .then(|()| async {
408                info!(target: LOG_CORE, "Shutdown called");
409            });
410
411        shutdown_future.await;
412        debug!(target: LOG_CORE, "Terminating main task");
413
414        if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
415            error!(target: LOG_CORE, ?err, "Error while shutting down task group");
416        }
417
418        debug!(target: LOG_CORE, "Shutdown complete");
419
420        fedimint_logging::shutdown();
421
422        drop(timing_total_runtime);
423
424        // Should we ever shut down without an error code?
425        std::process::exit(-1);
426    }
427}

429async fn run(
430    opts: ServerOpts,
431    task_group: &TaskGroup,
432    module_inits: ServerModuleInitRegistry,
433    module_inits_params: ServerModuleConfigGenParamsRegistry,
434    code_version_str: String,
435) -> anyhow::Result<()> {
436    if let Some(bind_metrics) = opts.bind_metrics.as_ref() {
437        task_group.spawn_cancellable(
438            "metrics-server",
439            fedimint_metrics::run_api_server(*bind_metrics, task_group.clone()),
440        );
441    }
442
443    // TODO: meh, move, refactor
444    let settings = ConfigGenSettings {
445        p2p_bind: opts.bind_p2p,
446        api_bind: opts.bind_api,
447        ui_bind: opts.bind_ui,
448        p2p_url: opts.p2p_url,
449        api_url: opts.api_url,
450        modules: module_inits_params.clone(),
451        registry: module_inits.clone(),
452        networking: if is_env_var_set(FM_FORCE_IROH_ENV) {
453            NetworkingStack::Iroh
454        } else {
455            NetworkingStack::default()
456        },
457    };
458
459    let db = Database::new(
460        fedimint_rocksdb::RocksDb::open(opts.data_dir.join(DB_FILE)).await?,
461        ModuleRegistry::default(),
462    );
463
464    let dyn_server_bitcoin_rpc = match (opts.bitcoind_url.as_ref(), opts.esplora_url.as_ref()) {
465        (Some(url), None) => BitcoindClient::new(url)?.into_dyn(),
466        (None, Some(url)) => EsploraClient::new(url)?.into_dyn(),
467        _ => unreachable!("ArgGroup already enforced XOR relation"),
468    };
469
470    Box::pin(fedimint_server::run(
471        opts.data_dir,
472        opts.force_api_secrets,
473        settings,
474        db,
475        code_version_str,
476        &module_inits,
477        task_group.clone(),
478        dyn_server_bitcoin_rpc,
479        Box::new(fedimint_server_ui::setup::router),
480        Box::new(fedimint_server_ui::dashboard::router),
481    ))
482    .await
