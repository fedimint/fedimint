273impl ServerOpts {
274    pub async fn get_bitcoind_url_and_password(&self) -> anyhow::Result<(SafeUrl, String)> {
275        let url = self
276            .bitcoind_url
277            .clone()
278            .ok_or_else(|| anyhow::anyhow!("No bitcoind url set"))?;
279        if let Some(password_file) = self.bitcoind_url_password_file.as_ref() {
280            let password = tokio::fs::read_to_string(password_file)
281                .await
282                .context("Failed to read the password")?
283                .trim()
284                .to_owned();
285            Ok((url, password))
286        } else {
287            let password = self
288                .bitcoind_password
289                .clone()
290                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_PASSWORD is not");
291            Ok((url, password))
292        }
293    }
294}

296/// Block the thread and run a Fedimintd server
297///
298/// # Arguments
299///
300/// * `module_init_registry` - The registry of available modules.
301///
302/// * `code_version_hash` - The git hash of the code that the `fedimintd` binary
303///   is being built from. This is used mostly for information purposes
304///   (`fedimintd version-hash`). See `fedimint-build` crate for easy way to
305///   obtain it.
306///
307/// * `code_version_vendor_suffix` - An optional suffix that will be appended to
308///   the internal fedimint release version, to distinguish binaries built by
309///   different vendors, usually with a different set of modules. The suffix is
310///   informational in setup/DKG: compatibility and consensus config generation
311///   use the normalized `x.y.z` release version.
312#[allow(clippy::too_many_lines)]
313pub async fn run(
314    module_init_registry: ServerModuleInitRegistry,
315    code_version_hash: &str,
316    code_version_vendor_suffix: Option<&str>,
317) -> anyhow::Result<Infallible> {
318    assert_eq!(
319        env!("FEDIMINT_BUILD_CODE_VERSION").len(),
320        code_version_hash.len(),
321        "version_hash must have an expected length"
322    );
323
324    handle_version_hash_command(code_version_hash);
325
326    let fedimint_version = env!("CARGO_PKG_VERSION");
327
328    APP_START_TS
329        .with_label_values(&[fedimint_version, code_version_hash])
330        .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
331
332    let server_opts = {
333        // Collect env vars from all registered modules and append them to the
334        // long-help text so operators can discover them via `fedimintd --help`.
335        let mut module_env_help = String::from("\nModule environment variables:\n");
336        for (_kind, module_init) in module_init_registry.iter() {
337            for doc in module_init.get_documented_env_vars() {
338                let _ = writeln!(module_env_help, "  {:40}  {}", doc.name, doc.description);
339            }
340        }
341        let matches = ServerOpts::command()
342            .after_long_help(module_env_help)
343            .get_matches();
344        ServerOpts::from_arg_matches(&matches)
345            .expect("clap arg matches must be valid after parsing")
346    };
347
348    let mut tracing_builder = TracingSetup::default();
349
350    tracing_builder
351        .tokio_console_bind(server_opts.bind_tokio_console)
352        .with_jaeger(server_opts.with_jaeger);
353
354    tracing_builder.init().unwrap();
355
356    info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
357
358    #[cfg(all(
359        not(feature = "jemalloc"),
360        not(any(target_env = "msvc", target_os = "ios", target_os = "android"))
361    ))]
362    warn!(
363        target: LOG_SERVER,
364        "fedimintd was built without the `jemalloc` feature. rocksdb is prone to memory \
365         fragmentation with the default allocator; consider rebuilding with `--features jemalloc`."
366    );
367
368    debug!(
369        target: LOG_SERVER,
370        core_consensus = %CORE_CONSENSUS_VERSION,
371        "Supported core consensus version",
372    );
373
374    let code_version_str = code_version_vendor_suffix.map_or_else(
375        || fedimint_version.to_string(),
376        |suffix| format!("{fedimint_version}+{suffix}"),
377    );
378
379    let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
380
381    let root_task_group = TaskGroup::new();
382
383    if let Some(bind_metrics) = server_opts.bind_metrics.as_ref() {
384        info!(
385            target: LOG_SERVER,
386            url = %format!("http://{}/metrics", bind_metrics),
387            "Initializing metrics server",
388        );
389        fedimint_metrics::spawn_api_server(*bind_metrics, root_task_group.clone()).await?;
390    }
391
392    let enable_iroh = server_opts.enable_iroh.unwrap_or(!is_running_in_test_env());
393    let iroh_next_api_settings = if server_opts.enable_iroh_next {
394        Some(IrohNextApiSettings::new(server_opts.bind_api_next))
395    } else {
396        None
397    };
398
399    let settings = ConfigGenSettings {
400        p2p_bind: server_opts.bind_p2p,
401        api_bind: server_opts.bind_api,
402        ui_bind: server_opts.bind_ui,
403        p2p_url: server_opts.p2p_url.clone(),
404        api_url: server_opts.api_url.clone(),
405        enable_iroh,
406        iroh_dns: server_opts.iroh_dns.clone(),
407        iroh_relays: server_opts.iroh_relays.clone(),
408        network: server_opts.bitcoin_network,
409        available_modules: module_init_registry.kinds(),
410        default_modules: module_init_registry.default_modules(),
411    };
412
413    let db = Database::new(
414        RocksDb::build(server_opts.data_dir.join(DB_FILE))
415            .open()
416            .await
417            .unwrap(),
418        ModuleRegistry::default(),
419    );
420
421    let dyn_server_bitcoin_rpc = match (
422        server_opts.bitcoind_url.as_ref(),
423        server_opts.esplora_url.as_ref(),
424    ) {
425        (Some(_), None) => {
426            let bitcoind_username = server_opts
427                .bitcoind_username
428                .clone()
429                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
430            let (bitcoind_url, bitcoind_password) = server_opts
431                .get_bitcoind_url_and_password()
432                .await
433                .expect("Failed to get bitcoind url");
434            BitcoindClient::new(bitcoind_username, bitcoind_password, &bitcoind_url)
435                .unwrap()
436                .into_dyn()
437        }
438        (None, Some(url)) => EsploraClient::new(url).unwrap().into_dyn(),
439        (Some(_), Some(esplora_url)) => {
440            let bitcoind_username = server_opts
441                .bitcoind_username
442                .clone()
443                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
444            let (bitcoind_url, bitcoind_password) = server_opts
445                .get_bitcoind_url_and_password()
446                .await
447                .expect("Failed to get bitcoind url");
448            BitcoindClientWithFallback::new(
449                bitcoind_username,
450                bitcoind_password,
451                &bitcoind_url,
452                esplora_url,
453            )
454            .unwrap()
455            .into_dyn()
456        }
457        _ => unreachable!("ArgGroup already enforced XOR relation"),
458    };
459    let dyn_server_bitcoin_rpc =
460        ServerBitcoinRpcTracked::new(dyn_server_bitcoin_rpc, "server").into_dyn();
461
462    root_task_group.install_kill_handler();
463
464    install_crypto_provider().await;
465
466    // The UI password falls back to the legacy password.private file on disk
467    // if its env var is not set, since the UI defaults to a loopback bind.
468    // The API password never falls back: the public admin API is always
469    // network-reachable, so it must be enabled explicitly via its env var or
470    // else admin RPCs return 401. Absence of a password for a given plane opts
471    // that plane into passwordless mode (UI) or disabled admin RPCs (API).
472    let password_file = std::fs::read_to_string(server_opts.data_dir.join(PLAINTEXT_PASSWORD))
473        .ok()
474        .map(|s| s.trim().to_owned());
475
476    let auth_ui = server_opts.password_ui.or(password_file).map(ApiAuth::new);
477    let auth_api = server_opts.password_api.map(ApiAuth::new);
478
479    let task_group = root_task_group.clone();
480    let code_version_hash = code_version_hash.to_string();
481    root_task_group.spawn_cancellable("main", async move {
482        fedimint_server::run_with_iroh_p2p_relays_and_next_api(
483            server_opts.data_dir,
484            auth_ui,
485            auth_api,
486            server_opts.force_api_secrets,
487            settings,
488            db,
489            code_version_str,
490            code_version_hash,
491            module_init_registry,
492            task_group,
493            dyn_server_bitcoin_rpc,
494            Box::new(fedimint_server_ui::setup::router),
495            Box::new(fedimint_server_ui::dashboard::router),
496            server_opts.db_checkpoint_retention,
497            Duration::from_secs(server_opts.session_timeout_secs),
498            fedimint_server::ConnectionLimits::new(
499                server_opts.iroh_api_max_connections,
500                server_opts.iroh_api_max_requests_per_connection,
501            ),
502            server_opts.iroh_p2p_relays,
503            iroh_next_api_settings,
504        )
505        .await
506        .unwrap_or_else(|err| panic!("Main task returned error: {}", err.fmt_compact_anyhow()));
507    });
508
509    let shutdown_future = root_task_group
510        .make_handle()
511        .make_shutdown_rx()
512        .then(|()| async {
513            info!(target: LOG_CORE, "Shutdown called");
514        });
515
516    shutdown_future.await;
517
518    debug!(target: LOG_CORE, "Terminating main task");
519
520    if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
521        error!(target: LOG_CORE, err = %err.fmt_compact_anyhow(), "Error while shutting down task group");
522    }
523
524    debug!(target: LOG_CORE, "Shutdown complete");
525
526    fedimint_logging::shutdown();
527
528    drop(timing_total_runtime);
529
530    std::process::exit(-1);
531}

533pub fn default_modules() -> ServerModuleInitRegistry {
534    let mut server_gens = ServerModuleInitRegistry::new();
535
536    server_gens.attach(MintInit);
537    server_gens.attach(fedimint_mintv2_server::MintInit);
538
539    server_gens.attach(WalletInit);
540    server_gens.attach(fedimint_walletv2_server::WalletInit);
541
542    server_gens.attach(LightningInit);
543    server_gens.attach(fedimint_lnv2_server::LightningInit);
544
545    if !is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
546        server_gens.attach(MetaInit);
547    }
548
549    if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
550        server_gens.attach(UnknownInit);
551    }
552
553    server_gens
554}

578    fn parse_server_opts_with_enable_iroh_env(value: &str) -> ServerOpts {
579        let previous = std::env::var_os(FM_ENABLE_IROH_ENV);
580        // This test does not spawn threads while mutating the process
581        // environment.
582        unsafe {
583            std::env::set_var(FM_ENABLE_IROH_ENV, value);
584        }
585
586        let opts = parse_server_opts();
587
588        // This test does not spawn threads while mutating the process
589        // environment.
590        unsafe {
591            if let Some(previous) = previous {
592                std::env::set_var(FM_ENABLE_IROH_ENV, previous);
593            } else {
594                std::env::remove_var(FM_ENABLE_IROH_ENV);
595            }
596        }
597
598        opts
599    }
