167/// Block the thread and run a Fedimintd server
168///
169/// # Arguments
170///
171/// * `modules_fn` - A function to initialize the modules.
172///
173/// * `code_version_hash` - The git hash of the code that the `fedimintd` binary
174///   is being built from. This is used mostly for information purposes
175///   (`fedimintd version-hash`). See `fedimint-build` crate for easy way to
176///   obtain it.
177///
178/// * `code_version_vendor_suffix` - An optional suffix that will be appended to
179///   the internal fedimint release version, to distinguish binaries built by
180///   different vendors, usually with a different set of modules. Currently DKG
181///   will enforce that the combined `code_version` is the same between all
182///   peers.
183pub async fn run(
184    modules_fn: fn(
185        Network,
186    ) -> (
187        ServerModuleInitRegistry,
188        ServerModuleConfigGenParamsRegistry,
189    ),
190    code_version_hash: &str,
191    code_version_vendor_suffix: Option<&str>,
192) -> ! {
193    assert_eq!(
194        env!("FEDIMINT_BUILD_CODE_VERSION").len(),
195        code_version_hash.len(),
196        "version_hash must have an expected length"
197    );
198
199    handle_version_hash_command(code_version_hash);
200
201    let fedimint_version = env!("CARGO_PKG_VERSION");
202
203    APP_START_TS
204        .with_label_values(&[fedimint_version, code_version_hash])
205        .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
206
207    let server_opts = ServerOpts::parse();
208
209    let mut tracing_builder = TracingSetup::default();
210
211    tracing_builder
212        .tokio_console_bind(server_opts.bind_tokio_console)
213        .with_jaeger(server_opts.with_jaeger);
214
215    tracing_builder.init().unwrap();
216
217    info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
218
219    let code_version_str = code_version_vendor_suffix.map_or_else(
220        || fedimint_version.to_string(),
221        |suffix| format!("{fedimint_version}+{suffix}"),
222    );
223
224    let (server_gens, server_gen_params) = modules_fn(server_opts.bitcoin_network);
225
226    let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
227
228    let root_task_group = TaskGroup::new();
229
230    if let Some(bind_metrics) = server_opts.bind_metrics.as_ref() {
231        root_task_group.spawn_cancellable(
232            "metrics-server",
233            fedimint_metrics::run_api_server(*bind_metrics, root_task_group.clone()),
234        );
235    }
236
237    let settings = ConfigGenSettings {
238        p2p_bind: server_opts.bind_p2p,
239        api_bind: server_opts.bind_api,
240        ui_bind: server_opts.bind_ui,
241        p2p_url: server_opts.p2p_url,
242        api_url: server_opts.api_url,
243        enable_iroh: server_opts.enable_iroh,
244        modules: server_gen_params.clone(),
245        registry: server_gens.clone(),
246    };
247
248    let db = Database::new(
249        RocksDb::open(server_opts.data_dir.join(DB_FILE))
250            .await
251            .unwrap(),
252        ModuleRegistry::default(),
253    );
254
255    let dyn_server_bitcoin_rpc = match (
256        server_opts.bitcoind_url.as_ref(),
257        server_opts.esplora_url.as_ref(),
258    ) {
259        (Some(url), None) => BitcoindClient::new(url).unwrap().into_dyn(),
260        (None, Some(url)) => EsploraClient::new(url).unwrap().into_dyn(),
261        _ => unreachable!("ArgGroup already enforced XOR relation"),
262    };
263
264    root_task_group.install_kill_handler();
265
266    let task_group = root_task_group.clone();
267    root_task_group.spawn_cancellable("main", async move {
268        fedimint_server::run(
269            server_opts.data_dir,
270            server_opts.force_api_secrets,
271            settings,
272            db,
273            code_version_str,
274            server_gens,
275            task_group,
276            dyn_server_bitcoin_rpc,
277            Box::new(fedimint_server_ui::setup::router),
278            Box::new(fedimint_server_ui::dashboard::router),
279            server_opts.db_checkpoint_retention,
280        )
281        .await
282        .unwrap_or_else(|err| panic!("Main task returned error: {}", err.fmt_compact_anyhow()));
283    });
284
285    let shutdown_future = root_task_group
286        .make_handle()
287        .make_shutdown_rx()
288        .then(|()| async {
289            info!(target: LOG_CORE, "Shutdown called");
290        });
291
292    shutdown_future.await;
293    debug!(target: LOG_CORE, "Terminating main task");
294
295    if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
296        error!(target: LOG_CORE, ?err, "Error while shutting down task group");
297    }
298
299    debug!(target: LOG_CORE, "Shutdown complete");
300
301    fedimint_logging::shutdown();
302
303    drop(timing_total_runtime);
304
305    // Should we ever shut down without an error code?
306    std::process::exit(-1);
307}

309pub fn default_modules(
310    network: Network,
311) -> (
312    ServerModuleInitRegistry,
313    ServerModuleConfigGenParamsRegistry,
314) {
315    let mut server_gens = ServerModuleInitRegistry::new();
316    let mut server_gen_params = ServerModuleConfigGenParamsRegistry::default();
317
318    let bitcoin_rpc_config = BitcoinRpcConfig {
319        kind: "bitcoind".to_string(),
320        url: "http://unused_dummy.xyz".parse().unwrap(),
321    };
322
323    server_gens.attach(LightningInit);
324    server_gen_params.attach_config_gen_params(
325        LightningInit::kind(),
326        LightningGenParams {
327            local: LightningGenParamsLocal {
328                bitcoin_rpc: bitcoin_rpc_config.clone(),
329            },
330            consensus: LightningGenParamsConsensus { network },
331        },
332    );
333
334    server_gens.attach(MintInit);
335    server_gen_params.attach_config_gen_params(
336        MintInit::kind(),
337        MintGenParams {
338            local: EmptyGenParams::default(),
339            consensus: MintGenParamsConsensus::new(
340                2,
341                // TODO: wait for clients to support the relative fees and set them to
342                // non-zero in 0.6
343                fedimint_mint_common::config::FeeConsensus::zero(),
344            ),
345        },
346    );
347
348    server_gens.attach(WalletInit);
349    server_gen_params.attach_config_gen_params(
350        WalletInit::kind(),
351        WalletGenParams {
352            local: WalletGenParamsLocal {
353                bitcoin_rpc: bitcoin_rpc_config.clone(),
354            },
355            consensus: WalletGenParamsConsensus {
356                network,
357                finality_delay: default_finality_delay(network),
358                client_default_bitcoin_rpc: default_esplora_server(network),
359                fee_consensus: fedimint_wallet_server::common::config::FeeConsensus::default(),
360            },
361        },
362    );
363
364    let enable_lnv2 = std::env::var_os(FM_ENABLE_MODULE_LNV2_ENV).is_none()
365        || is_env_var_set(FM_ENABLE_MODULE_LNV2_ENV);
366
367    if enable_lnv2 {
368        server_gens.attach(fedimint_lnv2_server::LightningInit);
369        server_gen_params.attach_config_gen_params(
370            fedimint_lnv2_server::LightningInit::kind(),
371            fedimint_lnv2_common::config::LightningGenParams {
372                local: fedimint_lnv2_common::config::LightningGenParamsLocal {
373                    bitcoin_rpc: bitcoin_rpc_config.clone(),
374                },
375                consensus: fedimint_lnv2_common::config::LightningGenParamsConsensus {
376                    // TODO: actually make the relative fee configurable
377                    fee_consensus: fedimint_lnv2_common::config::FeeConsensus::new(100).unwrap(),
378                    network,
379                },
380            },
381        );
382    }
383
384    if !is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
385        server_gens.attach(MetaInit);
386        server_gen_params.attach_config_gen_params(MetaInit::kind(), MetaGenParams::default());
387    };
388
389    if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
390        server_gens.attach(UnknownInit);
391        server_gen_params
392            .attach_config_gen_params(UnknownInit::kind(), UnknownGenParams::default());
393    }
394
395    (server_gens, server_gen_params)
396}

398pub fn default_esplora_server(network: Network) -> BitcoinRpcConfig {
399    BitcoinRpcConfig {
400        kind: "esplora".to_string(),
401        url: match network {
402            Network::Bitcoin => SafeUrl::parse("https://mempool.space/api/"),
403            Network::Testnet => SafeUrl::parse("https://mempool.space/testnet/api/"),
404            Network::Testnet4 => SafeUrl::parse("https://mempool.space/testnet4/api/"),
405            Network::Signet => SafeUrl::parse("https://mutinynet.com/api/"),
406            Network::Regtest => SafeUrl::parse(&format!(
407                "http://127.0.0.1:{}/",
408                std::env::var(FM_PORT_ESPLORA_ENV).unwrap_or(String::from("50002"))
409            )),
410            _ => panic!("Failed to parse default esplora server"),
411        }
412        .expect("Failed to parse default esplora server"),
413    }
414}

416/// For real Bitcoin we want to have a responsible default, while for demos
417/// nobody wants to wait multiple minutes
418fn default_finality_delay(network: Network) -> u32 {
419    match network {
420        Network::Bitcoin | Network::Regtest => 10,
421        Network::Testnet | Network::Signet | Network::Testnet4 => 2,
422        _ => panic!("Unsupported network"),
423    }
