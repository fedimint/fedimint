209impl ServerOpts {
210    pub async fn get_bitcoind_url_and_password(&self) -> anyhow::Result<(SafeUrl, String)> {
211        let url = self
212            .bitcoind_url
213            .clone()
214            .ok_or_else(|| anyhow::anyhow!("No bitcoind url set"))?;
215        if let Some(password_file) = self.bitcoind_url_password_file.as_ref() {
216            let password = tokio::fs::read_to_string(password_file)
217                .await
218                .context("Failed to read the password")?
219                .trim()
220                .to_owned();
221            Ok((url, password))
222        } else {
223            let password = self
224                .bitcoind_password
225                .clone()
226                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_PASSWORD is not");
227            Ok((url, password))
228        }
229    }
230}

232/// Block the thread and run a Fedimintd server
233///
234/// # Arguments
235///
236/// * `module_init_registry` - The registry of available modules.
237///
238/// * `code_version_hash` - The git hash of the code that the `fedimintd` binary
239///   is being built from. This is used mostly for information purposes
240///   (`fedimintd version-hash`). See `fedimint-build` crate for easy way to
241///   obtain it.
242///
243/// * `code_version_vendor_suffix` - An optional suffix that will be appended to
244///   the internal fedimint release version, to distinguish binaries built by
245///   different vendors, usually with a different set of modules. Currently DKG
246///   will enforce that the combined `code_version` is the same between all
247///   peers.
248#[allow(clippy::too_many_lines)]
249pub async fn run(
250    module_init_registry: ServerModuleInitRegistry,
251    code_version_hash: &str,
252    code_version_vendor_suffix: Option<&str>,
253) -> ! {
254    assert_eq!(
255        env!("FEDIMINT_BUILD_CODE_VERSION").len(),
256        code_version_hash.len(),
257        "version_hash must have an expected length"
258    );
259
260    handle_version_hash_command(code_version_hash);
261
262    let fedimint_version = env!("CARGO_PKG_VERSION");
263
264    APP_START_TS
265        .with_label_values(&[fedimint_version, code_version_hash])
266        .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
267
268    let server_opts = ServerOpts::parse();
269
270    let mut tracing_builder = TracingSetup::default();
271
272    tracing_builder
273        .tokio_console_bind(server_opts.bind_tokio_console)
274        .with_jaeger(server_opts.with_jaeger);
275
276    tracing_builder.init().unwrap();
277
278    info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
279
280    let code_version_str = code_version_vendor_suffix.map_or_else(
281        || fedimint_version.to_string(),
282        |suffix| format!("{fedimint_version}+{suffix}"),
283    );
284
285    let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
286
287    let root_task_group = TaskGroup::new();
288
289    if let Some(bind_metrics) = server_opts.bind_metrics.as_ref() {
290        root_task_group.spawn_cancellable(
291            "metrics-server",
292            fedimint_metrics::run_api_server(*bind_metrics, root_task_group.clone()),
293        );
294    }
295
296    let settings = ConfigGenSettings {
297        p2p_bind: server_opts.bind_p2p,
298        api_bind: server_opts.bind_api,
299        ui_bind: server_opts.bind_ui,
300        p2p_url: server_opts.p2p_url.clone(),
301        api_url: server_opts.api_url.clone(),
302        enable_iroh: server_opts.enable_iroh,
303        iroh_dns: server_opts.iroh_dns.clone(),
304        iroh_relays: server_opts.iroh_relays.clone(),
305        network: server_opts.bitcoin_network,
306        available_modules: module_init_registry.kinds(),
307    };
308
309    let db = Database::new(
310        RocksDb::open(server_opts.data_dir.join(DB_FILE))
311            .await
312            .unwrap(),
313        ModuleRegistry::default(),
314    );
315
316    let dyn_server_bitcoin_rpc = match (
317        server_opts.bitcoind_url.as_ref(),
318        server_opts.esplora_url.as_ref(),
319    ) {
320        (Some(_), None) => {
321            let bitcoind_username = server_opts
322                .bitcoind_username
323                .clone()
324                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
325            let (bitcoind_url, bitcoind_password) = server_opts
326                .get_bitcoind_url_and_password()
327                .await
328                .expect("Failed to get bitcoind url");
329            BitcoindClient::new(bitcoind_username, bitcoind_password, &bitcoind_url)
330                .unwrap()
331                .into_dyn()
332        }
333        (None, Some(url)) => EsploraClient::new(url).unwrap().into_dyn(),
334        (Some(_), Some(esplora_url)) => {
335            let bitcoind_username = server_opts
336                .bitcoind_username
337                .clone()
338                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
339            let (bitcoind_url, bitcoind_password) = server_opts
340                .get_bitcoind_url_and_password()
341                .await
342                .expect("Failed to get bitcoind url");
343            BitcoindClientWithFallback::new(
344                bitcoind_username,
345                bitcoind_password,
346                &bitcoind_url,
347                esplora_url,
348            )
349            .unwrap()
350            .into_dyn()
351        }
352        _ => unreachable!("ArgGroup already enforced XOR relation"),
353    };
354
355    root_task_group.install_kill_handler();
356
357    install_crypto_provider().await;
358
359    let task_group = root_task_group.clone();
360    root_task_group.spawn_cancellable("main", async move {
361        fedimint_server::run(
362            server_opts.data_dir,
363            server_opts.force_api_secrets,
364            settings,
365            db,
366            code_version_str,
367            module_init_registry,
368            task_group,
369            dyn_server_bitcoin_rpc,
370            Box::new(fedimint_server_ui::setup::router),
371            Box::new(fedimint_server_ui::dashboard::router),
372            server_opts.db_checkpoint_retention,
373            fedimint_server::ConnectionLimits::new(
374                server_opts.iroh_api_max_connections,
375                server_opts.iroh_api_max_requests_per_connection,
376            ),
377        )
378        .await
379        .unwrap_or_else(|err| panic!("Main task returned error: {}", err.fmt_compact_anyhow()));
380    });
381
382    let shutdown_future = root_task_group
383        .make_handle()
384        .make_shutdown_rx()
385        .then(|()| async {
386            info!(target: LOG_CORE, "Shutdown called");
387        });
388
389    shutdown_future.await;
390
391    debug!(target: LOG_CORE, "Terminating main task");
392
393    if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
394        error!(target: LOG_CORE, err = %err.fmt_compact_anyhow(), "Error while shutting down task group");
395    }
396
397    debug!(target: LOG_CORE, "Shutdown complete");
398
399    fedimint_logging::shutdown();
400
401    drop(timing_total_runtime);
402
403    // Should we ever shut down without an error code?
404    std::process::exit(-1);
405}

407pub fn default_modules() -> ServerModuleInitRegistry {
408    let mut server_gens = ServerModuleInitRegistry::new();
409
410    server_gens.attach(MintInit);
411    server_gens.attach(WalletInit);
412
413    if is_env_var_set_opt(FM_ENABLE_MODULE_LNV1_ENV).unwrap_or(true) {
414        server_gens.attach(LightningInit);
415    }
416
417    if is_env_var_set_opt(FM_ENABLE_MODULE_LNV2_ENV).unwrap_or(true) {
418        server_gens.attach(fedimint_lnv2_server::LightningInit);
419    }
420
421    if !is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
422        server_gens.attach(MetaInit);
423    }
424
425    if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
426        server_gens.attach(UnknownInit);
427    }
428
429    server_gens
