215impl ServerOpts {
216    pub async fn get_bitcoind_url_and_password(&self) -> anyhow::Result<(SafeUrl, String)> {
217        let url = self
218            .bitcoind_url
219            .clone()
220            .ok_or_else(|| anyhow::anyhow!("No bitcoind url set"))?;
221        if let Some(password_file) = self.bitcoind_url_password_file.as_ref() {
222            let password = tokio::fs::read_to_string(password_file)
223                .await
224                .context("Failed to read the password")?
225                .trim()
226                .to_owned();
227            Ok((url, password))
228        } else {
229            let password = self
230                .bitcoind_password
231                .clone()
232                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_PASSWORD is not");
233            Ok((url, password))
234        }
235    }
236}

238/// Block the thread and run a Fedimintd server
239///
240/// # Arguments
241///
242/// * `modules_fn` - A function to initialize the modules.
243///
244/// * `code_version_hash` - The git hash of the code that the `fedimintd` binary
245///   is being built from. This is used mostly for information purposes
246///   (`fedimintd version-hash`). See `fedimint-build` crate for easy way to
247///   obtain it.
248///
249/// * `code_version_vendor_suffix` - An optional suffix that will be appended to
250///   the internal fedimint release version, to distinguish binaries built by
251///   different vendors, usually with a different set of modules. Currently DKG
252///   will enforce that the combined `code_version` is the same between all
253///   peers.
254#[allow(clippy::too_many_lines)]
255pub async fn run(
256    modules_fn: fn(
257        Network,
258    ) -> (
259        ServerModuleInitRegistry,
260        ServerModuleConfigGenParamsRegistry,
261    ),
262    code_version_hash: &str,
263    code_version_vendor_suffix: Option<&str>,
264) -> ! {
265    assert_eq!(
266        env!("FEDIMINT_BUILD_CODE_VERSION").len(),
267        code_version_hash.len(),
268        "version_hash must have an expected length"
269    );
270
271    handle_version_hash_command(code_version_hash);
272
273    let fedimint_version = env!("CARGO_PKG_VERSION");
274
275    APP_START_TS
276        .with_label_values(&[fedimint_version, code_version_hash])
277        .set(fedimint_core::time::duration_since_epoch().as_secs() as i64);
278
279    let server_opts = ServerOpts::parse();
280
281    let mut tracing_builder = TracingSetup::default();
282
283    tracing_builder
284        .tokio_console_bind(server_opts.bind_tokio_console)
285        .with_jaeger(server_opts.with_jaeger);
286
287    tracing_builder.init().unwrap();
288
289    info!("Starting fedimintd (version: {fedimint_version} version_hash: {code_version_hash})");
290
291    let code_version_str = code_version_vendor_suffix.map_or_else(
292        || fedimint_version.to_string(),
293        |suffix| format!("{fedimint_version}+{suffix}"),
294    );
295
296    let (server_gens, server_gen_params) = modules_fn(server_opts.bitcoin_network);
297
298    let timing_total_runtime = timing::TimeReporter::new("total-runtime").info();
299
300    let root_task_group = TaskGroup::new();
301
302    if let Some(bind_metrics) = server_opts.bind_metrics.as_ref() {
303        root_task_group.spawn_cancellable(
304            "metrics-server",
305            fedimint_metrics::run_api_server(*bind_metrics, root_task_group.clone()),
306        );
307    }
308
309    let settings = ConfigGenSettings {
310        p2p_bind: server_opts.bind_p2p,
311        api_bind: server_opts.bind_api,
312        ui_bind: server_opts.bind_ui,
313        p2p_url: server_opts.p2p_url.clone(),
314        api_url: server_opts.api_url.clone(),
315        enable_iroh: server_opts.enable_iroh,
316        iroh_dns: server_opts.iroh_dns.clone(),
317        iroh_relays: server_opts.iroh_relays.clone(),
318        modules: server_gen_params.clone(),
319        registry: server_gens.clone(),
320    };
321
322    let db = Database::new(
323        RocksDb::open(server_opts.data_dir.join(DB_FILE))
324            .await
325            .unwrap(),
326        ModuleRegistry::default(),
327    );
328
329    let dyn_server_bitcoin_rpc = match (
330        server_opts.bitcoind_url.as_ref(),
331        server_opts.esplora_url.as_ref(),
332    ) {
333        (Some(_), None) => {
334            let bitcoind_username = server_opts
335                .bitcoind_username
336                .clone()
337                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
338            let (bitcoind_url, bitcoind_password) = server_opts
339                .get_bitcoind_url_and_password()
340                .await
341                .expect("Failed to get bitcoind url");
342            BitcoindClient::new(bitcoind_username, bitcoind_password, &bitcoind_url)
343                .unwrap()
344                .into_dyn()
345        }
346        (None, Some(url)) => EsploraClient::new(url).unwrap().into_dyn(),
347        (Some(_), Some(esplora_url)) => {
348            let bitcoind_username = server_opts
349                .bitcoind_username
350                .clone()
351                .expect("FM_BITCOIND_URL is set but FM_BITCOIND_USERNAME is not");
352            let (bitcoind_url, bitcoind_password) = server_opts
353                .get_bitcoind_url_and_password()
354                .await
355                .expect("Failed to get bitcoind url");
356            BitcoindClientWithFallback::new(
357                bitcoind_username,
358                bitcoind_password,
359                &bitcoind_url,
360                esplora_url,
361            )
362            .unwrap()
363            .into_dyn()
364        }
365        _ => unreachable!("ArgGroup already enforced XOR relation"),
366    };
367
368    root_task_group.install_kill_handler();
369
370    install_crypto_provider().await;
371
372    let task_group = root_task_group.clone();
373    root_task_group.spawn_cancellable("main", async move {
374        fedimint_server::run(
375            server_opts.data_dir,
376            server_opts.force_api_secrets,
377            settings,
378            db,
379            code_version_str,
380            server_gens,
381            task_group,
382            dyn_server_bitcoin_rpc,
383            Box::new(fedimint_server_ui::setup::router),
384            Box::new(fedimint_server_ui::dashboard::router),
385            server_opts.db_checkpoint_retention,
386            fedimint_server::ConnectionLimits::new(
387                server_opts.iroh_api_max_connections,
388                server_opts.iroh_api_max_requests_per_connection,
389            ),
390        )
391        .await
392        .unwrap_or_else(|err| panic!("Main task returned error: {}", err.fmt_compact_anyhow()));
393    });
394
395    let shutdown_future = root_task_group
396        .make_handle()
397        .make_shutdown_rx()
398        .then(|()| async {
399            info!(target: LOG_CORE, "Shutdown called");
400        });
401
402    shutdown_future.await;
403    debug!(target: LOG_CORE, "Terminating main task");
404
405    if let Err(err) = root_task_group.join_all(Some(SHUTDOWN_TIMEOUT)).await {
406        error!(target: LOG_CORE, err = %err.fmt_compact_anyhow(), "Error while shutting down task group");
407    }
408
409    debug!(target: LOG_CORE, "Shutdown complete");
410
411    fedimint_logging::shutdown();
412
413    drop(timing_total_runtime);
414
415    // Should we ever shut down without an error code?
416    std::process::exit(-1);
417}

419pub fn default_modules(
420    network: Network,
421) -> (
422    ServerModuleInitRegistry,
423    ServerModuleConfigGenParamsRegistry,
424) {
425    let mut server_gens = ServerModuleInitRegistry::new();
426    let mut server_gen_params = ServerModuleConfigGenParamsRegistry::default();
427
428    server_gens.attach(LightningInit);
429    server_gen_params
430        .attach_config_gen_params(LightningInit::kind(), LightningGenParams { network });
431
432    server_gens.attach(MintInit);
433    server_gen_params.attach_config_gen_params(MintInit::kind(), MintGenParams::new(2, None));
434
435    server_gens.attach(WalletInit);
436    server_gen_params.attach_config_gen_params(
437        WalletInit::kind(),
438        WalletGenParams {
439            network,
440            finality_delay: default_finality_delay(network),
441            client_default_bitcoin_rpc: default_esplora_server(
442                network,
443                std::env::var(FM_PORT_ESPLORA_ENV).ok(),
444            ),
445            fee_consensus: fedimint_wallet_server::common::config::FeeConsensus::default(),
446        },
447    );
448
449    let enable_lnv2 = std::env::var_os(FM_ENABLE_MODULE_LNV2_ENV).is_none()
450        || is_env_var_set(FM_ENABLE_MODULE_LNV2_ENV);
451
452    if enable_lnv2 {
453        server_gens.attach(fedimint_lnv2_server::LightningInit);
454        server_gen_params.attach_config_gen_params(
455            fedimint_lnv2_server::LightningInit::kind(),
456            fedimint_lnv2_common::config::LightningGenParams {
457                fee_consensus: None,
458                network,
459            },
460        );
461    }
462
463    if !is_env_var_set(FM_DISABLE_META_MODULE_ENV) {
464        server_gens.attach(MetaInit);
465        server_gen_params.attach_config_gen_params(MetaInit::kind(), MetaGenParams);
466    }
467
468    if is_env_var_set(FM_USE_UNKNOWN_MODULE_ENV) {
469        server_gens.attach(UnknownInit);
470        server_gen_params.attach_config_gen_params(UnknownInit::kind(), UnknownGenParams);
471    }
472
473    (server_gens, server_gen_params)
474}

476/// For real Bitcoin we want to have a responsible default, while for demos
477/// nobody wants to wait multiple minutes
478fn default_finality_delay(network: Network) -> u32 {
479    match network {
480        Network::Bitcoin | Network::Regtest => 10,
481        Network::Testnet | Network::Signet | Network::Testnet4 => 2,
482        _ => panic!("Unsupported network"),
483    }
