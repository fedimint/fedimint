500    pub fn validate_config(
501        &self,
502        identity: &PeerId,
503        module_config_gens: &ServerModuleInitRegistry,
504    ) -> anyhow::Result<()> {
505        let endpoints = self.consensus.api_endpoints().clone();
506        let consensus = self.consensus.clone();
507        let private = self.private.clone();
508
509        let my_public_key = private.broadcast_secret_key.public_key(&Secp256k1::new());
510
511        if Some(&my_public_key) != consensus.broadcast_public_keys.get(identity) {
512            bail!("Broadcast secret key doesn't match corresponding public key");
513        }
514        if endpoints.keys().max().copied().map(PeerId::to_usize) != Some(endpoints.len() - 1) {
515            bail!("Peer ids are not indexed from 0");
516        }
517        if endpoints.keys().min().copied() != Some(PeerId::from(0)) {
518            bail!("Peer ids are not indexed from 0");
519        }
520
521        for (module_id, module_kind) in &self
522            .consensus
523            .modules
524            .iter()
525            .map(|(id, config)| Ok((*id, config.kind.clone())))
526            .collect::<anyhow::Result<BTreeSet<_>>>()?
527        {
528            module_config_gens
529                .get(module_kind)
530                .ok_or_else(|| format_err!("module config gen not found {module_kind}"))?
531                .validate_config(identity, self.get_module_config(*module_id)?)?;
532        }
533
534        Ok(())
535    }

537    pub fn trusted_dealer_gen(
538        modules: ServerModuleConfigGenParamsRegistry,
539        params: &HashMap<PeerId, ConfigGenParams>,
540        registry: &ServerModuleInitRegistry,
541        code_version_str: &str,
542    ) -> BTreeMap<PeerId, Self> {
543        let peer0 = &params[&PeerId::from(0)];
544
545        let mut broadcast_pks = BTreeMap::new();
546        let mut broadcast_sks = BTreeMap::new();
547        for peer_id in peer0.peer_ids() {
548            let (broadcast_sk, broadcast_pk) = secp256k1::generate_keypair(&mut OsRng);
549            broadcast_pks.insert(peer_id, broadcast_pk);
550            broadcast_sks.insert(peer_id, broadcast_sk);
551        }
552
553        let module_configs: BTreeMap<_, _> = modules
554            .iter_modules()
555            .map(|(module_id, kind, module_params)| {
556                (
557                    module_id,
558                    registry
559                        .get(kind)
560                        .expect("Module not registered")
561                        .trusted_dealer_gen(&peer0.peer_ids(), module_params),
562                )
563            })
564            .collect();
565
566        let server_config: BTreeMap<_, _> = peer0
567            .peer_ids()
568            .iter()
569            .map(|&id| {
570                let config = ServerConfig::from(
571                    params[&id].clone(),
572                    id,
573                    broadcast_pks.clone(),
574                    *broadcast_sks.get(&id).expect("We created this entry"),
575                    module_configs
576                        .iter()
577                        .map(|(module_id, cfgs)| (*module_id, cfgs[&id].clone()))
578                        .collect(),
579                    code_version_str.to_string(),
580                );
581                (id, config)
582            })
583            .collect();
584
585        server_config
586    }

588    /// Runs the distributed key gen algorithm
589    pub async fn distributed_gen(
590        modules: ServerModuleConfigGenParamsRegistry,
591        params: &ConfigGenParams,
592        registry: ServerModuleInitRegistry,
593        code_version_str: String,
594        connections: DynP2PConnections<P2PMessage>,
595        p2p_status_receivers: BTreeMap<PeerId, watch::Receiver<P2PConnectionStatus>>,
596    ) -> anyhow::Result<Self> {
597        let _timing /* logs on drop */ = timing::TimeReporter::new("distributed-gen").info();
598
599        // in case we are running by ourselves, avoid DKG
600        if params.peer_ids().len() == 1 {
601            let server = Self::trusted_dealer_gen(
602                modules,
603                &HashMap::from([(params.identity, params.clone())]),
604                &registry,
605                &code_version_str,
606            );
607            return Ok(server[&params.identity].clone());
608        }
609
610        while p2p_status_receivers
611            .values()
612            .any(|r| *r.borrow() == P2PConnectionStatus::Disconnected)
613        {
614            info!(
615                target: LOG_NET_PEER_DKG,
616                "Waiting for all p2p connections to open..."
617            );
618
619            sleep(Duration::from_secs(1)).await;
620        }
621
622        info!(
623            target: LOG_NET_PEER_DKG,
624            "Comparing peer connection info checksum..."
625        );
626
627        let checksum = params.peers.consensus_hash_sha256();
628
629        connections
630            .send(Recipient::Everyone, P2PMessage::Checksum(checksum))
631            .await;
632
633        for peer in params
634            .peer_ids()
635            .into_iter()
636            .filter(|p| *p != params.identity)
637        {
638            ensure!(
639                connections
640                    .receive_from_peer(peer)
641                    .await
642                    .context("Unexpected shutdown of p2p connections")?
643                    == P2PMessage::Checksum(checksum),
644                "Peer {peer} has not send the correct checksum message"
645            );
646        }
647
648        info!(
649            target: LOG_NET_PEER_DKG,
650            "Running distributed key generation..."
651        );
652
653        let handle = PeerHandle::new(
654            params.peer_ids().to_num_peers(),
655            params.identity,
656            &connections,
657        );
658
659        let (broadcast_sk, broadcast_pk) = secp256k1::generate_keypair(&mut OsRng);
660
661        let broadcast_public_keys = handle.exchange_encodable(broadcast_pk).await?;
662
663        let mut module_cfgs = BTreeMap::new();
664
665        for (module_id, kind, module_params) in modules.iter_modules() {
666            info!(
667                target: LOG_NET_PEER_DKG,
668                "Running distributed key generation for module of kind {kind}..."
669            );
670
671            let cfg = registry
672                .get(kind)
673                .context("Module of kind {kind} not found")?
674                .distributed_gen(&handle, module_params)
675                .await?;
676
677            module_cfgs.insert(module_id, cfg);
678        }
679
680        let cfg = ServerConfig::from(
681            params.clone(),
682            params.identity,
683            broadcast_public_keys,
684            broadcast_sk,
685            module_cfgs,
686            code_version_str,
687        );
688
689        info!(
690            target: LOG_NET_PEER_DKG,
691            "Comparing consensus config checksum..."
692        );
693
694        let checksum = cfg.consensus.consensus_hash_sha256();
695
696        connections
697            .send(Recipient::Everyone, P2PMessage::Checksum(checksum))
698            .await;
699
700        for peer in params
701            .peer_ids()
702            .into_iter()
703            .filter(|p| *p != params.identity)
704        {
705            ensure!(
706                connections
707                    .receive_from_peer(peer)
708                    .await
709                    .context("Unexpected shutdown of p2p connections")?
710                    == P2PMessage::Checksum(checksum),
711                "Peer {peer} has not send the correct checksum message"
712            );
713        }
714
715        info!(
716            target: LOG_NET_PEER_DKG,
717            "Distributed key generation has completed successfully!"
718        );
719
720        Ok(cfg)
721    }
722}

739                .iter()

746impl ConfigGenParams {
747    pub fn peer_ids(&self) -> Vec<PeerId> {
748        self.peers.keys().copied().collect()
749    }
750
751    pub fn tls_config(&self) -> TlsConfig {
752        TlsConfig {
753            private_key: self.tls_key.clone().unwrap(),
754            certificates: self
755                .tls_certs()
756                .iter()
757                .map(|(peer, cert)| (*peer, rustls::Certificate(Vec::from_hex(cert).unwrap())))
758                .collect(),
759            peer_names: self
760                .p2p_urls()
761                .into_iter()
762                .map(|(id, peer)| (id, peer.name))
763                .collect(),
764        }
765    }
766
767    pub fn tls_certs(&self) -> BTreeMap<PeerId, String> {
768        self.peers
769            .iter()
770            .filter_map(|(id, peer)| {
771                match peer.endpoints.clone() {
772                    PeerEndpoints::Tcp { cert, .. } => Some(cert.encode_hex()),
773                    PeerEndpoints::Iroh { .. } => None,
774                }
775                .map(|peer| (*id, peer))
776            })
777            .collect()
778    }
779
780    pub fn p2p_urls(&self) -> BTreeMap<PeerId, PeerUrl> {
781        self.peers
782            .iter()
783            .filter_map(|(id, peer)| {
784                match peer.endpoints.clone() {
785                    PeerEndpoints::Tcp { p2p_url, .. } => Some(PeerUrl {
786                        name: peer.name.clone(),
787                        url: p2p_url.clone(),
788                    }),
789                    PeerEndpoints::Iroh { .. } => None,
790                }
791                .map(|peer| (*id, peer))
792            })
793            .collect()
794    }
795
796    pub fn api_urls(&self) -> BTreeMap<PeerId, PeerUrl> {
797        self.peers
798            .iter()
799            .filter_map(|(id, peer)| {
800                match peer.endpoints.clone() {
801                    PeerEndpoints::Tcp { api_url, .. } => Some(PeerUrl {
802                        name: peer.name.clone(),
803                        url: api_url.clone(),
804                    }),
805                    PeerEndpoints::Iroh { .. } => None,
806                }
807                .map(|peer| (*id, peer))
808            })
809            .collect()
810    }
811
812    pub fn iroh_endpoints(&self) -> BTreeMap<PeerId, PeerIrohEndpoints> {
813        self.peers
814            .iter()
815            .filter_map(|(id, peer)| {
816                match peer.endpoints.clone() {
817                    PeerEndpoints::Tcp { .. } => None,
818                    PeerEndpoints::Iroh { api_pk, p2p_pk } => Some(PeerIrohEndpoints {
819                        name: peer.name.clone(),
820                        api_pk,
821                        p2p_pk,
822                    }),
823                }
824                .map(|peer| (*id, peer))
825            })
826            .collect()
827    }
