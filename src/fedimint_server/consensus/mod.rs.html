59pub async fn run(
60    connections: DynP2PConnections<P2PMessage>,
61    p2p_status_receivers: P2PStatusReceivers,
62    p2p_connection_type_receivers: P2PConnectionTypeReceivers,
63    api_bind: SocketAddr,
64    iroh_dns: Option<SafeUrl>,
65    iroh_relays: Vec<SafeUrl>,
66    cfg: ServerConfig,
67    db: Database,
68    module_init_registry: ServerModuleInitRegistry,
69    task_group: &TaskGroup,
70    force_api_secrets: ApiSecrets,
71    data_dir: PathBuf,
72    code_version_str: String,
73    dyn_server_bitcoin_rpc: DynServerBitcoinRpc,
74    ui_bind: SocketAddr,
75    dashboard_ui_router: DashboardUiRouter,
76    db_checkpoint_retention: u64,
77    iroh_api_limits: ConnectionLimits,
78) -> anyhow::Result<()> {
79    cfg.validate_config(&cfg.local.identity, &module_init_registry)?;
80
81    let mut global_dbtx = db.begin_transaction().await;
82    apply_migrations_server_dbtx(
83        &mut global_dbtx.to_ref_nc(),
84        Arc::new(ServerDbMigrationContext),
85        "fedimint-server".to_string(),
86        get_global_database_migrations(),
87    )
88    .await?;
89
90    update_server_info_version_dbtx(&mut global_dbtx.to_ref_nc(), &code_version_str).await;
91
92    if is_running_in_test_env() {
93        verify_server_db_integrity_dbtx(&mut global_dbtx.to_ref_nc()).await;
94    }
95    global_dbtx.commit_tx_result().await?;
96
97    let mut modules = BTreeMap::new();
98
99    // TODO: make it work with all transports and federation secrets
100    let global_api = DynGlobalApi::from_endpoints(
101        cfg.consensus
102            .api_endpoints()
103            .iter()
104            .map(|(&peer_id, url)| (peer_id, url.url.clone())),
105        &None,
106    )
107    .await?;
108
109    let server_bitcoin_rpc_monitor = ServerBitcoinRpcMonitor::new(
110        dyn_server_bitcoin_rpc,
111        if is_running_in_test_env() {
112            Duration::from_millis(100)
113        } else {
114            Duration::from_secs(60)
115        },
116        task_group,
117    );
118
119    for (module_id, module_cfg) in &cfg.consensus.modules {
120        match module_init_registry.get(&module_cfg.kind) {
121            Some(module_init) => {
122                info!(target: LOG_CORE, "Initialise module {module_id}...");
123
124                let mut dbtx = db.begin_transaction().await;
125                apply_migrations_dbtx(
126                    &mut dbtx.to_ref_nc(),
127                    Arc::new(ServerDbMigrationContext) as Arc<_>,
128                    module_init.module_kind().to_string(),
129                    module_init.get_database_migrations(),
130                    Some(*module_id),
131                    None,
132                )
133                .await?;
134
135                if let Some(used_db_prefixes) = module_init.used_db_prefixes()
136                    && is_running_in_test_env()
137                {
138                    verify_module_db_integrity_dbtx(
139                        &mut dbtx.to_ref_nc(),
140                        *module_id,
141                        module_init.module_kind(),
142                        &used_db_prefixes,
143                    )
144                    .await;
145                }
146                dbtx.commit_tx_result().await?;
147
148                let module = module_init
149                    .init(
150                        NumPeers::from(cfg.consensus.api_endpoints().len()),
151                        cfg.get_module_config(*module_id)?,
152                        db.with_prefix_module_id(*module_id).0,
153                        task_group,
154                        cfg.local.identity,
155                        global_api.with_module(*module_id),
156                        server_bitcoin_rpc_monitor.clone(),
157                    )
158                    .await?;
159
160                modules.insert(*module_id, (module_cfg.kind.clone(), module));
161            }
162            None => bail!("Detected configuration for unsupported module id: {module_id}"),
163        }
164    }
165
166    let module_registry = ModuleRegistry::from(modules);
167
168    let client_cfg = cfg.consensus.to_client_config(&module_init_registry)?;
169
170    let (submission_sender, submission_receiver) = async_channel::bounded(TRANSACTION_BUFFER);
171    let (shutdown_sender, shutdown_receiver) = watch::channel(None);
172    let (ord_latency_sender, ord_latency_receiver) = watch::channel(None);
173
174    let mut ci_status_senders = BTreeMap::new();
175    let mut ci_status_receivers = BTreeMap::new();
176
177    for peer in cfg.consensus.broadcast_public_keys.keys().copied() {
178        let (ci_sender, ci_receiver) = watch::channel(None);
179
180        ci_status_senders.insert(peer, ci_sender);
181        ci_status_receivers.insert(peer, ci_receiver);
182    }
183
184    let consensus_api = ConsensusApi {
185        cfg: cfg.clone(),
186        cfg_dir: data_dir.clone(),
187        db: db.clone(),
188        modules: module_registry.clone(),
189        client_cfg: client_cfg.clone(),
190        submission_sender: submission_sender.clone(),
191        shutdown_sender,
192        shutdown_receiver: shutdown_receiver.clone(),
193        supported_api_versions: ServerConfig::supported_api_versions_summary(
194            &cfg.consensus.modules,
195            &module_init_registry,
196        ),
197        p2p_status_receivers,
198        p2p_connection_type_receivers,
199        ci_status_receivers,
200        ord_latency_receiver,
201        bitcoin_rpc_connection: server_bitcoin_rpc_monitor,
202        force_api_secret: force_api_secrets.get_active(),
203        code_version_str,
204        task_group: task_group.clone(),
205    };
206
207    info!(target: LOG_CONSENSUS, "Starting Consensus Api...");
208
209    let api_handler = start_consensus_api(
210        &cfg.local,
211        consensus_api.clone(),
212        force_api_secrets.clone(),
213        api_bind,
214    )
215    .await;
216
217    if let Some(iroh_api_sk) = cfg.private.iroh_api_sk.clone()
218        && let Err(e) = Box::pin(start_iroh_api(
219            iroh_api_sk,
220            api_bind,
221            iroh_dns,
222            iroh_relays,
223            consensus_api.clone(),
224            task_group,
225            iroh_api_limits,
226        ))
227        .await
228    {
229        // clean up ws api before propagating error
230        api_handler.stop().expect("Just started");
231        api_handler.stopped().await;
232        return Err(e);
233    }
234
235    info!(target: LOG_CONSENSUS, "Starting Submission of Module CI proposals...");
236
237    for (module_id, kind, module) in module_registry.iter_modules() {
238        submit_module_ci_proposals(
239            task_group,
240            db.clone(),
241            module_id,
242            kind.clone(),
243            module.clone(),
244            submission_sender.clone(),
245        );
246    }
247
248    info!(target: LOG_CONSENSUS, "Starting Consensus Engine...");
249
250    let api_urls = get_api_urls(&db, &cfg.consensus).await;
251
252    let ui_service = dashboard_ui_router(consensus_api.clone().into_dyn()).into_make_service();
253
254    let ui_listener = TcpListener::bind(ui_bind)
255        .await
256        .expect("Failed to bind dashboard UI");
257
258    task_group.spawn("dashboard-ui", move |handle| async move {
259        axum::serve(ui_listener, ui_service)
260            .with_graceful_shutdown(handle.make_shutdown_rx())
261            .await
262            .expect("Failed to serve dashboard UI");
263    });
264
265    info!(target: LOG_CONSENSUS, "Dashboard UI running at http://{ui_bind} 🚀");
266
267    // FIXME: (@leonardo) How should this be handled ?
268    // Using the `Connector::default()` for now!
269    ConsensusEngine {
270        db,
271        federation_api: DynGlobalApi::from_endpoints(api_urls, &force_api_secrets.get_active())
272            .await?,
273        cfg: cfg.clone(),
274        connections,
275        ord_latency_sender,
276        ci_status_senders,
277        submission_receiver,
278        shutdown_receiver,
279        modules: module_registry,
280        task_group: task_group.clone(),
281        data_dir,
282        db_checkpoint_retention,
283    }
284    .run()
285    .await?;
286
287    api_handler
288        .stop()
289        .expect("Consensus api should still be running");
290
291    api_handler.stopped().await;
292
293    Ok(())
294}

296async fn start_consensus_api(
297    cfg: &ServerConfigLocal,
298    api: ConsensusApi,
299    force_api_secrets: ApiSecrets,
300    api_bind: SocketAddr,
301) -> ServerHandle {
302    let mut rpc_module = RpcModule::new(api.clone());
303
304    net::api::attach_endpoints(&mut rpc_module, api::server_endpoints(), None);
305
306    for (id, _, module) in api.modules.iter_modules() {
307        net::api::attach_endpoints(&mut rpc_module, module.api_endpoints(), Some(id));
308    }
309
310    net::api::spawn(
311        "consensus",
312        api_bind,
313        rpc_module,
314        cfg.max_connections,
315        force_api_secrets,
316    )
317    .await
318}

320const CONSENSUS_PROPOSAL_TIMEOUT: Duration = Duration::from_secs(30);
321
322fn submit_module_ci_proposals(
323    task_group: &TaskGroup,
324    db: Database,
325    module_id: ModuleInstanceId,
326    kind: ModuleKind,
327    module: DynServerModule,
328    submission_sender: Sender<ConsensusItem>,
329) {
330    let mut interval = tokio::time::interval(if is_running_in_test_env() {
331        Duration::from_millis(100)
332    } else {
333        Duration::from_secs(1)
334    });
335
336    task_group.spawn(
337        format!("citem_proposals_{module_id}"),
338        move |task_handle| async move {
339            while !task_handle.is_shutting_down() {
340                let module_consensus_items = tokio::time::timeout(
341                    CONSENSUS_PROPOSAL_TIMEOUT,
342                    module.consensus_proposal(
343                        &mut db
344                            .begin_transaction_nc()
345                            .await
346                            .to_ref_with_prefix_module_id(module_id)
347                            .0
348                            .into_nc(),
349                        module_id,
350                    ),
351                )
352                .await;
353
354                match module_consensus_items {
355                    Ok(items) => {
356                        for item in items {
357                            if submission_sender
358                                .send(ConsensusItem::Module(item))
359                                .await
360                                .is_err()
361                            {
362                                warn!(
363                                    target: LOG_CONSENSUS,
364                                    module_id,
365                                    "Unable to submit module consensus item proposal via channel"
366                                );
367                            }
368                        }
369                    }
370                    Err(..) => {
371                        warn!(
372                            target: LOG_CONSENSUS,
373                            module_id,
374                            %kind,
375                            "Module failed to propose consensus items on time"
376                        );
377                    }
378                }
379
380                interval.tick().await;
381            }
382        },
383    );
384}

386async fn start_iroh_api(
387    secret_key: iroh::SecretKey,
388    api_bind: SocketAddr,
389    iroh_dns: Option<SafeUrl>,
390    iroh_relays: Vec<SafeUrl>,
391    consensus_api: ConsensusApi,
392    task_group: &TaskGroup,
393    iroh_api_limits: ConnectionLimits,
394) -> anyhow::Result<()> {
395    let endpoint = build_iroh_endpoint(
396        secret_key,
397        api_bind,
398        iroh_dns,
399        iroh_relays,
400        FEDIMINT_API_ALPN,
401    )
402    .await?;
403    task_group.spawn_cancellable(
404        "iroh-api",
405        run_iroh_api(consensus_api, endpoint, task_group.clone(), iroh_api_limits),
406    );
407
408    Ok(())
409}

411async fn run_iroh_api(
412    consensus_api: ConsensusApi,
413    endpoint: Endpoint,
414    task_group: TaskGroup,
415    iroh_api_limits: ConnectionLimits,
416) {
417    let core_api = server_endpoints()
418        .into_iter()
419        .map(|endpoint| (endpoint.path.to_string(), endpoint))
420        .collect::<BTreeMap<String, ApiEndpoint<ConsensusApi>>>();
421
422    let module_api = consensus_api
423        .modules
424        .iter_modules()
425        .map(|(id, _, module)| {
426            let api_endpoints = module
427                .api_endpoints()
428                .into_iter()
429                .map(|endpoint| (endpoint.path.to_string(), endpoint))
430                .collect::<BTreeMap<String, ApiEndpoint<DynServerModule>>>();
431
432            (id, api_endpoints)
433        })
434        .collect::<BTreeMap<ModuleInstanceId, BTreeMap<String, ApiEndpoint<DynServerModule>>>>();
435
436    let consensus_api = Arc::new(consensus_api);
437    let core_api = Arc::new(core_api);
438    let module_api = Arc::new(module_api);
439    let parallel_connections_limit = Arc::new(Semaphore::new(iroh_api_limits.max_connections));
440
441    loop {
442        match endpoint.accept().await {
443            Some(incoming) => {
444                if parallel_connections_limit.available_permits() == 0 {
445                    warn!(
446                        target: LOG_NET_API,
447                        limit = iroh_api_limits.max_connections,
448                        "Iroh API connection limit reached, blocking new connections"
449                    );
450                }
451                let permit = parallel_connections_limit
452                    .clone()
453                    .acquire_owned()
454                    .await
455                    .expect("semaphore should not be closed");
456                task_group.spawn_cancellable_silent(
457                    "handle-iroh-connection",
458                    handle_incoming(
459                        consensus_api.clone(),
460                        core_api.clone(),
461                        module_api.clone(),
462                        task_group.clone(),
463                        incoming,
464                        permit,
465                        iroh_api_limits.max_requests_per_connection,
466                    )
467                    .then(|result| async {
468                        if let Err(err) = result {
469                            warn!(target: LOG_NET_API, err = %err.fmt_compact_anyhow(), "Failed to handle iroh connection");
470                        }
471                    }),
472                );
473            }
474            None => return,
475        }
476    }
477}

479async fn handle_incoming(
480    consensus_api: Arc<ConsensusApi>,
481    core_api: Arc<BTreeMap<String, ApiEndpoint<ConsensusApi>>>,
482    module_api: Arc<BTreeMap<ModuleInstanceId, BTreeMap<String, ApiEndpoint<DynServerModule>>>>,
483    task_group: TaskGroup,
484    incoming: Incoming,
485    _connection_permit: tokio::sync::OwnedSemaphorePermit,
486    iroh_api_max_requests_per_connection: usize,
487) -> anyhow::Result<()> {
488    let connection = incoming.accept()?.await?;
489    let parallel_requests_limit = Arc::new(Semaphore::new(iroh_api_max_requests_per_connection));
490
491    loop {
492        let (send_stream, recv_stream) = connection.accept_bi().await?;
493
494        if parallel_requests_limit.available_permits() == 0 {
495            warn!(
496                target: LOG_NET_API,
497                limit = iroh_api_max_requests_per_connection,
498                "Iroh API request limit reached for connection, blocking new requests"
499            );
500        }
501        let permit = parallel_requests_limit
502            .clone()
503            .acquire_owned()
504            .await
505            .expect("semaphore should not be closed");
506        task_group.spawn_cancellable_silent(
507            "handle-iroh-request",
508            handle_request(
509                consensus_api.clone(),
510                core_api.clone(),
511                module_api.clone(),
512                send_stream,
513                recv_stream,
514                permit,
515            )
516            .then(|result| async {
517                if let Err(err) = result {
518                    warn!(target: LOG_NET_API, err = %err.fmt_compact_anyhow(), "Failed to handle iroh request");
519                }
520            }),
521        );
522    }
523}

525async fn handle_request(
526    consensus_api: Arc<ConsensusApi>,
527    core_api: Arc<BTreeMap<String, ApiEndpoint<ConsensusApi>>>,
528    module_api: Arc<BTreeMap<ModuleInstanceId, BTreeMap<String, ApiEndpoint<DynServerModule>>>>,
529    mut send_stream: SendStream,
530    mut recv_stream: RecvStream,
531    _request_permit: tokio::sync::OwnedSemaphorePermit,
532) -> anyhow::Result<()> {
533    let request = recv_stream.read_to_end(100_000).await?;
534
535    let request = serde_json::from_slice::<IrohApiRequest>(&request)?;
536
537    let response = await_response(consensus_api, core_api, module_api, request).await;
538
539    let response = serde_json::to_vec(&response)?;
540
541    send_stream.write_all(&response).await?;
542
543    send_stream.finish()?;
544
545    Ok(())
546}

548async fn await_response(
549    consensus_api: Arc<ConsensusApi>,
550    core_api: Arc<BTreeMap<String, ApiEndpoint<ConsensusApi>>>,
551    module_api: Arc<BTreeMap<ModuleInstanceId, BTreeMap<String, ApiEndpoint<DynServerModule>>>>,
552    request: IrohApiRequest,
553) -> Result<Value, ApiError> {
554    match request.method {
555        ApiMethod::Core(method) => {
556            let endpoint = core_api.get(&method).ok_or(ApiError::not_found(method))?;
557
558            let (state, context) = consensus_api.context(&request.request, None).await;
559
560            (endpoint.handler)(state, context, request.request).await
561        }
562        ApiMethod::Module(module_id, method) => {
563            let endpoint = module_api
564                .get(&module_id)
565                .ok_or(ApiError::not_found(module_id.to_string()))?
566                .get(&method)
567                .ok_or(ApiError::not_found(method))?;
568
569            let (state, context) = consensus_api
570                .context(&request.request, Some(module_id))
571                .await;
572
573            (endpoint.handler)(state, context, request.request).await
574        }
575    }
