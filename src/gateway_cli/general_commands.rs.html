3use clap::Subcommand;

10    ConnectFedPayload, LeaveFedPayload, PaymentLogPayload, PaymentSummaryPayload,

13use crate::print_response;

15#[derive(Subcommand)]
16pub enum GeneralCommands {
17    /// Display the version hash of the CLI.
18    VersionHash,
19    /// Display high-level information about the gateway.
20    Info,
21    /// Get the total on-chain, lightning, and eCash balances of the gateway.
22    GetBalances,
23    /// Register the gateway with a federation.
24    ConnectFed {
25        /// Invite code to connect to the federation
26        invite_code: String,
27        /// Activate usage of Tor (or not) as the connector for the federation
28        /// client
29        #[cfg(feature = "tor")]
30        use_tor: Option<bool>,
31        /// Indicates if the client should be recovered from a mnemonic
32        #[clap(long)]
33        recover: Option<bool>,
34    },
35    /// Leave a federation.
36    LeaveFed {
37        #[clap(long)]
38        federation_id: FederationId,
39    },
40    /// Prints the seed phrase for the gateway
41    Seed,
42    /// Safely stop the gateway
43    Stop,
44    /// List the fedimint transactions that the gateway has processed
45    PaymentLog {
46        #[clap(long)]
47        end_position: Option<EventLogId>,
48
49        #[clap(long, default_value_t = 25)]
50        pagination_size: usize,
51
52        #[clap(long)]
53        federation_id: FederationId,
54
55        #[clap(long)]
56        event_kinds: Vec<EventKind>,
57    },
58    /// Create a bcrypt hash of a password, for use in gateway deployment
59    CreatePasswordHash {
60        password: String,
61
62        /// The bcrypt cost factor to use when hashing the password
63        #[clap(long)]
64        cost: Option<u32>,
65    },
66    /// List a payment summary for the last day
67    PaymentSummary {
68        #[clap(long)]
69        start: Option<u64>,
70
71        #[clap(long)]
72        end: Option<u64>,
73    },
74}
75
76impl GeneralCommands {
77    #[allow(clippy::too_many_lines)]
78    pub async fn handle(self, client: &GatewayRpcClient) -> anyhow::Result<()> {
79        match self {
80            Self::VersionHash => {
81                println!("{}", fedimint_build_code_version_env!());
82            }
83            Self::Info => {
84                let response = client.get_info().await?;
85                print_response(response);
86            }
87            Self::GetBalances => {
88                let response = client.get_balances().await?;
89                print_response(response);
90            }
91            Self::ConnectFed {
92                invite_code,
93                #[cfg(feature = "tor")]
94                use_tor,
95                recover,
96            } => {
97                let response = client
98                    .connect_federation(ConnectFedPayload {
99                        invite_code,
100                        #[cfg(feature = "tor")]
101                        use_tor,
102                        #[cfg(not(feature = "tor"))]
103                        use_tor: None,
104                        recover,
105                    })
106                    .await?;
107
108                print_response(response);
109            }
110            Self::LeaveFed { federation_id } => {
111                let response = client
112                    .leave_federation(LeaveFedPayload { federation_id })
113                    .await?;
114                print_response(response);
115            }
116            Self::Seed => {
117                let response = client.get_mnemonic().await?;
118                print_response(response);
119            }
120            Self::Stop => {
121                client.stop().await?;
122            }
123            Self::PaymentLog {
124                end_position,
125                pagination_size,
126                federation_id,
127                event_kinds,
128            } => {
129                let payment_log = client
130                    .payment_log(PaymentLogPayload {
131                        end_position,
132                        pagination_size,
133                        federation_id,
134                        event_kinds,
135                    })
136                    .await?;
137                print_response(payment_log);
138            }
139            Self::CreatePasswordHash { password, cost } => print_response(
140                bcrypt::hash(password, cost.unwrap_or(bcrypt::DEFAULT_COST))
141                    .expect("Unable to create bcrypt hash"),
142            ),
143            Self::PaymentSummary { start, end } => {
144                let now = now();
145                let now_millis = now
146                    .duration_since(UNIX_EPOCH)
147                    .expect("Before unix epoch")
148                    .as_millis()
149                    .try_into()?;
150                let one_day_ago = now
151                    .checked_sub(Duration::from_secs(60 * 60 * 24))
152                    .expect("Before unix epoch");
153                let one_day_ago_millis = one_day_ago
154                    .duration_since(UNIX_EPOCH)
155                    .expect("Before unix epoch")
156                    .as_millis()
157                    .try_into()?;
158                let end_millis = end.unwrap_or(now_millis);
159                let start_millis = start.unwrap_or(one_day_ago_millis);
160                let payment_summary = client
161                    .payment_summary(PaymentSummaryPayload {
162                        start_millis,
163                        end_millis,
164                    })
165                    .await?;
166                print_response(payment_summary);
167            }
168        }
169
170        Ok(())
171    }
172}
