31#[apply(async_trait_maybe_send!)]
32pub trait ServerModule: Debug + Sized {
33    type Common: ModuleCommon;
34
35    type Init: ServerModuleInit;
36
37    fn module_kind() -> ModuleKind {
38        // Note: All modules should define kinds as &'static str, so this doesn't
39        // allocate
40        <Self::Init as ModuleInit>::Common::KIND
41    }
42
43    /// Returns a decoder for the following associated types of this module:
44    /// * `ClientConfig`
45    /// * `Input`
46    /// * `Output`
47    /// * `OutputOutcome`
48    /// * `ConsensusItem`
49    /// * `InputError`
50    /// * `OutputError`
51    fn decoder() -> Decoder {
52        Self::Common::decoder_builder().build()
53    }
54
55    /// This module's contribution to the next consensus proposal. This method
56    /// is only guaranteed to be called once every few seconds. Consensus items
57    /// are not meant to be latency critical; do not create them as
58    /// a response to a processed transaction. Only use consensus items to
59    /// establish consensus on a value that is required to verify
60    /// transactions, like unix time, block heights and feerates, and model all
61    /// other state changes trough transactions. The intention for this method
62    /// is to always return all available consensus items even if they are
63    /// redundant while process_consensus_item returns an error for the
64    /// redundant proposals.
65    ///
66    /// If you think you actually do require latency critical consensus items or
67    /// have trouble designing your module in order to avoid them please contact
68    /// the Fedimint developers.
69    async fn consensus_proposal<'a>(
70        &'a self,
71        dbtx: &mut DatabaseTransaction<'_>,
72    ) -> Vec<<Self::Common as ModuleCommon>::ConsensusItem>;
73
74    /// This function is called once for every consensus item. The function
75    /// should return Ok if and only if the consensus item changes
76    /// the system state. *Therefore this method should return an error in case
77    /// of merely redundant consensus items such that they will be purged from
78    /// the history of the federation.* This enables consensus_proposal to
79    /// return all available consensus item without wasting disk
80    /// space with redundant consensus items.
81    async fn process_consensus_item<'a, 'b>(
82        &'a self,
83        dbtx: &mut DatabaseTransaction<'b>,
84        consensus_item: <Self::Common as ModuleCommon>::ConsensusItem,
85        peer_id: PeerId,
86    ) -> anyhow::Result<()>;
87
88    // Use this function to parallelise stateless cryptographic verification of
89    // inputs across a transaction. All inputs of a transaction are verified
90    // before any input is processed.
91    fn verify_input(
92        &self,
93        _input: &<Self::Common as ModuleCommon>::Input,
94    ) -> Result<(), <Self::Common as ModuleCommon>::InputError> {
95        Ok(())
96    }
97
98    /// Try to spend a transaction input. On success all necessary updates will
99    /// be part of the database transaction. On failure (e.g. double spend)
100    /// the database transaction is rolled back and the operation will take
101    /// no effect.
102    async fn process_input<'a, 'b, 'c>(
103        &'a self,
104        dbtx: &mut DatabaseTransaction<'c>,
105        input: &'b <Self::Common as ModuleCommon>::Input,
106        in_point: InPoint,
107    ) -> Result<InputMeta, <Self::Common as ModuleCommon>::InputError>;
108
109    /// Try to create an output (e.g. issue notes, peg-out BTC, …). On success
110    /// all necessary updates to the database will be part of the database
111    /// transaction. On failure (e.g. double spend) the database transaction
112    /// is rolled back and the operation will take no effect.
113    ///
114    /// The supplied `out_point` identifies the operation (e.g. a peg-out or
115    /// note issuance) and can be used to retrieve its outcome later using
116    /// `output_status`.
117    async fn process_output<'a, 'b>(
118        &'a self,
119        dbtx: &mut DatabaseTransaction<'b>,
120        output: &'a <Self::Common as ModuleCommon>::Output,
121        out_point: OutPoint,
122    ) -> Result<TransactionItemAmount, <Self::Common as ModuleCommon>::OutputError>;
123
124    /// **Deprecated**: Modules should not be using it. Instead, they should
125    /// implement their own custom endpoints with semantics, versioning,
126    /// serialization, etc. that suits them. Potentially multiple or none.
127    ///
128    /// Depending on the module this might contain data needed by the client to
129    /// access funds or give an estimate of when funds will be available.
130    ///
131    /// Returns `None` if the output is unknown, **NOT** if it is just not ready
132    /// yet.
133    ///
134    /// Since this has become deprecated you may return `None` even if the
135    /// output is known as long as the output outcome is not used inside the
136    /// module.
137    #[deprecated(note = "https://github.com/fedimint/fedimint/issues/6671")]
138    async fn output_status(
139        &self,
140        dbtx: &mut DatabaseTransaction<'_>,
141        out_point: OutPoint,
142    ) -> Option<<Self::Common as ModuleCommon>::OutputOutcome>;
143
144    /// Verify submission-only checks for an input
145    ///
146    /// Most modules should not need to know or implement it, so the default
147    /// implementation just returns OK.
148    ///
149    /// In special circumstances it is useful to enforce requirements on the
150    /// included transaction outside of the consensus, in a similar way
151    /// Bitcoin enforces mempool policies.
152    ///
153    /// This functionality might be removed in the future versions, as more
154    /// checks become part of the consensus, so it is advised not to use it.
155    #[doc(hidden)]
156    async fn verify_input_submission<'a, 'b, 'c>(
157        &'a self,
158        _dbtx: &mut DatabaseTransaction<'c>,
159        _input: &'b <Self::Common as ModuleCommon>::Input,
160    ) -> Result<(), <Self::Common as ModuleCommon>::InputError> {
161        Ok(())
162    }
163
164    /// Verify submission-only checks for an output
165    ///
166    /// See [`Self::verify_input_submission`] for more information.
167    #[doc(hidden)]
168    async fn verify_output_submission<'a, 'b>(
169        &'a self,
170        _dbtx: &mut DatabaseTransaction<'b>,
171        _output: &'a <Self::Common as ModuleCommon>::Output,
172        _out_point: OutPoint,
173    ) -> Result<(), <Self::Common as ModuleCommon>::OutputError> {
174        Ok(())
175    }
176
177    /// Queries the database and returns all assets and liabilities of the
178    /// module.
179    ///
180    /// Summing over all modules, if liabilities > assets then an error has
181    /// occurred in the database and consensus should halt.
182    async fn audit(
183        &self,
184        dbtx: &mut DatabaseTransaction<'_>,
185        audit: &mut Audit,
186        module_instance_id: ModuleInstanceId,
187    );
188
189    /// Returns a list of custom API endpoints defined by the module. These are
190    /// made available both to users as well as to other modules. They thus
191    /// should be deterministic, only dependant on their input and the
192    /// current epoch.
193    fn api_endpoints(&self) -> Vec<ApiEndpoint<Self>>;
194}

196/// Backend side module interface
197///
198/// Server side Fedimint module needs to implement this trait.
199#[apply(async_trait_maybe_send!)]
200pub trait IServerModule: Debug {
201    fn as_any(&self) -> &dyn Any;
202
203    /// Returns the decoder belonging to the server module
204    fn decoder(&self) -> Decoder;
205
206    fn module_kind(&self) -> ModuleKind;
207
208    /// This module's contribution to the next consensus proposal
209    async fn consensus_proposal(
210        &self,
211        dbtx: &mut DatabaseTransaction<'_>,
212        module_instance_id: ModuleInstanceId,
213    ) -> Vec<DynModuleConsensusItem>;
214
215    /// This function is called once for every consensus item. The function
216    /// returns an error if any only if the consensus item does not change
217    /// our state and therefore may be safely discarded by the atomic broadcast.
218    async fn process_consensus_item<'a, 'b>(
219        &self,
220        dbtx: &mut DatabaseTransaction<'a>,
221        consensus_item: &'b DynModuleConsensusItem,
222        peer_id: PeerId,
223    ) -> anyhow::Result<()>;
224
225    // Use this function to parallelise stateless cryptographic verification of
226    // inputs across a transaction. All inputs of a transaction are verified
227    // before any input is processed.
228    fn verify_input(&self, input: &DynInput) -> Result<(), DynInputError>;
229
230    /// Try to spend a transaction input. On success all necessary updates will
231    /// be part of the database transaction. On failure (e.g. double spend)
232    /// the database transaction is rolled back and the operation will take
233    /// no effect.
234    async fn process_input<'a, 'b, 'c>(
235        &'a self,
236        dbtx: &mut DatabaseTransaction<'c>,
237        input: &'b DynInput,
238        in_point: InPoint,
239    ) -> Result<InputMeta, DynInputError>;
240
241    /// Try to create an output (e.g. issue notes, peg-out BTC, …). On success
242    /// all necessary updates to the database will be part of the database
243    /// transaction. On failure (e.g. double spend) the database transaction
244    /// is rolled back and the operation will take no effect.
245    ///
246    /// The supplied `out_point` identifies the operation (e.g. a peg-out or
247    /// note issuance) and can be used to retrieve its outcome later using
248    /// `output_status`.
249    async fn process_output<'a>(
250        &self,
251        dbtx: &mut DatabaseTransaction<'a>,
252        output: &DynOutput,
253        out_point: OutPoint,
254    ) -> Result<TransactionItemAmount, DynOutputError>;
255
256    /// See [`ServerModule::verify_input_submission`]
257    #[doc(hidden)]
258    async fn verify_input_submission<'a, 'b, 'c>(
259        &'a self,
260        dbtx: &mut DatabaseTransaction<'c>,
261        input: &'b DynInput,
262    ) -> Result<(), DynInputError>;
263
264    /// See [`ServerModule::verify_output_submission`]
265    #[doc(hidden)]
266    async fn verify_output_submission<'a>(
267        &self,
268        _dbtx: &mut DatabaseTransaction<'a>,
269        _output: &DynOutput,
270        _out_point: OutPoint,
271    ) -> Result<(), DynOutputError>;
272
273    /// See [`ServerModule::output_status`]
274    #[deprecated(note = "https://github.com/fedimint/fedimint/issues/6671")]
275    async fn output_status(
276        &self,
277        dbtx: &mut DatabaseTransaction<'_>,
278        out_point: OutPoint,
279        module_instance_id: ModuleInstanceId,
280    ) -> Option<DynOutputOutcome>;
281
282    /// Queries the database and returns all assets and liabilities of the
283    /// module.
284    ///
285    /// Summing over all modules, if liabilities > assets then an error has
286    /// occurred in the database and consensus should halt.
287    async fn audit(
288        &self,
289        dbtx: &mut DatabaseTransaction<'_>,
290        audit: &mut Audit,
291        module_instance_id: ModuleInstanceId,
292    );
293
294    /// Returns a list of custom API endpoints defined by the module. These are
295    /// made available both to users as well as to other modules. They thus
296    /// should be deterministic, only dependant on their input and the
297    /// current epoch.
298    fn api_endpoints(&self) -> Vec<ApiEndpoint<DynServerModule>>;
299}

301dyn_newtype_define!(
302    #[derive(Clone)]
303    pub DynServerModule(Arc<IServerModule>)

305
306#[apply(async_trait_maybe_send!)]
307impl<T> IServerModule for T
308where
309    T: ServerModule + 'static + Sync,
310{
311    fn decoder(&self) -> Decoder {
312        <T::Common as ModuleCommon>::decoder_builder().build()
313    }
314
315    fn as_any(&self) -> &dyn Any {
316        self
317    }
318
319    fn module_kind(&self) -> ModuleKind {
320        <Self as ServerModule>::module_kind()
321    }
322
323    /// This module's contribution to the next consensus proposal
324    async fn consensus_proposal(
325        &self,
326        dbtx: &mut DatabaseTransaction<'_>,
327        module_instance_id: ModuleInstanceId,
328    ) -> Vec<DynModuleConsensusItem> {
329        <Self as ServerModule>::consensus_proposal(self, dbtx)
330            .await
331            .into_iter()
332            .map(|v| DynModuleConsensusItem::from_typed(module_instance_id, v))
333            .collect()
334    }
335
336    /// This function is called once for every consensus item. The function
337    /// returns an error if any only if the consensus item does not change
338    /// our state and therefore may be safely discarded by the atomic broadcast.
339    async fn process_consensus_item<'a, 'b>(
340        &self,
341        dbtx: &mut DatabaseTransaction<'a>,
342        consensus_item: &'b DynModuleConsensusItem,
343        peer_id: PeerId,
344    ) -> anyhow::Result<()> {
345        <Self as ServerModule>::process_consensus_item(
346            self,
347            dbtx,
348            Clone::clone(
349                consensus_item.as_any()
350                    .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::ConsensusItem>()
351                    .expect("incorrect consensus item type passed to module plugin"),
352            ),
353            peer_id
354        )
355        .await
356    }
357
358    // Use this function to parallelise stateless cryptographic verification of
359    // inputs across a transaction. All inputs of a transaction are verified
360    // before any input is processed.
361    fn verify_input(&self, input: &DynInput) -> Result<(), DynInputError> {
362        <Self as ServerModule>::verify_input(
363            self,
364            input
365                .as_any()
366                .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::Input>()
367                .expect("incorrect input type passed to module plugin"),
368        )
369        .map_err(|v| DynInputError::from_typed(input.module_instance_id(), v))
370    }
371
372    /// Try to spend a transaction input. On success all necessary updates will
373    /// be part of the database transaction. On failure (e.g. double spend)
374    /// the database transaction is rolled back and the operation will take
375    /// no effect.
376    async fn process_input<'a, 'b, 'c>(
377        &'a self,
378        dbtx: &mut DatabaseTransaction<'c>,
379        input: &'b DynInput,
380        in_point: InPoint,
381    ) -> Result<InputMeta, DynInputError> {
382        <Self as ServerModule>::process_input(
383            self,
384            dbtx,
385            input
386                .as_any()
387                .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::Input>()
388                .expect("incorrect input type passed to module plugin"),
389            in_point,
390        )
391        .await
392        .map_err(|v| DynInputError::from_typed(input.module_instance_id(), v))
393    }
394
395    /// Try to create an output (e.g. issue notes, peg-out BTC, …). On success
396    /// all necessary updates to the database will be part of the database
397    /// transaction. On failure (e.g. double spend) the database transaction
398    /// is rolled back and the operation will take no effect.
399    ///
400    /// The supplied `out_point` identifies the operation (e.g. a peg-out or
401    /// note issuance) and can be used to retrieve its outcome later using
402    /// `output_status`.
403    async fn process_output<'a>(
404        &self,
405        dbtx: &mut DatabaseTransaction<'a>,
406        output: &DynOutput,
407        out_point: OutPoint,
408    ) -> Result<TransactionItemAmount, DynOutputError> {
409        <Self as ServerModule>::process_output(
410            self,
411            dbtx,
412            output
413                .as_any()
414                .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::Output>()
415                .expect("incorrect output type passed to module plugin"),
416            out_point,
417        )
418        .await
419        .map_err(|v| DynOutputError::from_typed(output.module_instance_id(), v))
420    }
421
422    async fn verify_input_submission<'a, 'b, 'c>(
423        &'a self,
424        dbtx: &mut DatabaseTransaction<'c>,
425        input: &'b DynInput,
426    ) -> Result<(), DynInputError> {
427        <Self as ServerModule>::verify_input_submission(
428            self,
429            dbtx,
430            input
431                .as_any()
432                .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::Input>()
433                .expect("incorrect input type passed to module plugin"),
434        )
435        .await
436        .map_err(|v| DynInputError::from_typed(input.module_instance_id(), v))
437    }
438
439    async fn verify_output_submission<'a>(
440        &self,
441        dbtx: &mut DatabaseTransaction<'a>,
442        output: &DynOutput,
443        out_point: OutPoint,
444    ) -> Result<(), DynOutputError> {
445        <Self as ServerModule>::verify_output_submission(
446            self,
447            dbtx,
448            output
449                .as_any()
450                .downcast_ref::<<<Self as ServerModule>::Common as ModuleCommon>::Output>()
451                .expect("incorrect output type passed to module plugin"),
452            out_point,
453        )
454        .await
455        .map_err(|v| DynOutputError::from_typed(output.module_instance_id(), v))
456    }
457
458    /// See [`ServerModule::output_status`]
459    async fn output_status(
460        &self,
461        dbtx: &mut DatabaseTransaction<'_>,
462        out_point: OutPoint,
463        module_instance_id: ModuleInstanceId,
464    ) -> Option<DynOutputOutcome> {
465        #[allow(deprecated)]
466        <Self as ServerModule>::output_status(self, dbtx, out_point)
467            .await
468            .map(|v| DynOutputOutcome::from_typed(module_instance_id, v))
469    }
470
471    /// Queries the database and returns all assets and liabilities of the
472    /// module.
473    ///
474    /// Summing over all modules, if liabilities > assets then an error has
475    /// occurred in the database and consensus should halt.
476    async fn audit(
477        &self,
478        dbtx: &mut DatabaseTransaction<'_>,
479        audit: &mut Audit,
480        module_instance_id: ModuleInstanceId,
481    ) {
482        <Self as ServerModule>::audit(self, dbtx, audit, module_instance_id).await;
483    }
484
485    fn api_endpoints(&self) -> Vec<ApiEndpoint<DynServerModule>> {
486        <Self as ServerModule>::api_endpoints(self)
487            .into_iter()
488            .map(|ApiEndpoint { path, handler }| ApiEndpoint {
489                path,
490                handler: Box::new(
491                    move |module: &DynServerModule,
492                          context: ApiEndpointContext<'_>,
493                          value: ApiRequestErased| {
494                        let typed_module = module
495                            .as_any()
496                            .downcast_ref::<T>()
497                            .expect("the dispatcher should always call with the right module");
498                        Box::pin(handler(typed_module, context, value))
499                    },
500                ),
501            })
502            .collect()
503    }
504}

506/// Collection of server modules
507pub type ServerModuleRegistry = ModuleRegistry<DynServerModule>;
508
509pub trait ServerModuleRegistryExt {
510    fn decoder_registry(&self) -> ModuleDecoderRegistry;
511}
512
513impl ServerModuleRegistryExt for ServerModuleRegistry {
514    /// Generate a `ModuleDecoderRegistry` from this `ModuleRegistry`
515    fn decoder_registry(&self) -> ModuleDecoderRegistry {
516        // TODO: cache decoders
517        self.iter_modules()
518            .map(|(id, kind, module)| (id, kind.clone(), module.decoder()))
519            .collect::<ModuleDecoderRegistry>()
520    }
