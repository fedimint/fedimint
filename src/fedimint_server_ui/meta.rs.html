142// Handler for submitting a JSON meta configuration
143pub async fn submit_meta_value(
144    State(state): State<AuthState<DynDashboardApi>>,
145    jar: CookieJar,
146    Form(form): Form<MetaValueForm>,
147) -> impl IntoResponse {
148    // Check authentication
149    if !check_auth(&state.auth_cookie_name, &state.auth_cookie_value, &jar).await {
150        return Redirect::to("/login").into_response();
151    }
152
153    let meta_module = state.api.get_module::<Meta>().unwrap();
154
155    if let Ok(json_value) = serde_json::from_str::<Value>(&form.json_content) {
156        meta_module.handle_submit_request_ui(json_value).await.ok();
157    }
158
159    Redirect::to("/").into_response()
160}
