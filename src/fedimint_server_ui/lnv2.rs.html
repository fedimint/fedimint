17pub async fn render(lightning: &fedimint_lnv2_server::Lightning) -> Markup {

25    html! {

37                                }

43                        }

45                        // Gateway management

47                            h5 { "Gateway Management" }

114// Handler for adding a new gateway
115pub async fn add_gateway(
116    State(state): State<AuthState<DynDashboardApi>>,
117    jar: CookieJar,
118    Form(form): Form<GatewayForm>,
119) -> impl IntoResponse {
120    if !check_auth(&state.auth_cookie_name, &state.auth_cookie_value, &jar).await {
121        return Redirect::to("/login").into_response();
122    }
123
124    state
125        .api
126        .get_module::<fedimint_lnv2_server::Lightning>()
127        .expect("Route only mounted when Lightning V2 module exists")
128        .add_gateway_ui(form.gateway_url)
129        .await;
130
131    Redirect::to("/").into_response()
132}
