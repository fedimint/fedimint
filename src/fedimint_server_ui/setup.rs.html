133// POST handler for the /setup route (process the password setup form)
134async fn setup_submit(
135    State(state): State<UiState<DynSetupApi>>,
136    Form(input): Form<SetupInput>,
137) -> impl IntoResponse {
138    // Only use federation_name if is_lead is true
139    let federation_name = if input.is_lead {
140        Some(input.federation_name)
141    } else {
142        None
143    };
144
145    // Only use disable_base_fees if is_lead is true
146    let disable_base_fees = if input.is_lead {
147        Some(input.disable_base_fees)
148    } else {
149        None
150    };
151
152    match state
153        .api
154        .set_local_parameters(
155            ApiAuth(input.password),
156            input.name,
157            federation_name,
158            disable_base_fees,
159        )
160        .await
161    {
162        Ok(_) => Redirect::to(LOGIN_ROUTE).into_response(),
163        Err(e) => {
164            let content = html! {
165                div class="alert alert-danger" { (e.to_string()) }
166                div class="button-container" {
167                    a href=(ROOT_ROUTE) class="btn btn-primary setup-btn" { "Return to Setup" }
168                }
169            };
170
171            Html(setup_layout("Setup Error", content).into_string()).into_response()
172        }
173    }
174}

185// POST handler for the /login route (authenticate and set session cookie)
186async fn login_submit(
187    State(state): State<UiState<DynSetupApi>>,
188    jar: CookieJar,
189    Form(input): Form<LoginInput>,
190) -> impl IntoResponse {
191    let auth = match state.api.auth().await {
192        Some(auth) => auth,
193        None => return Redirect::to(ROOT_ROUTE).into_response(),
194    };
195
196    login_submit_response(
197        auth,
198        state.auth_cookie_name,
199        state.auth_cookie_value,
200        jar,
201        input,
202    )
203    .into_response()
204}

373pub fn router(api: DynSetupApi) -> Router {
374    Router::new()
375        .route(ROOT_ROUTE, get(setup_form).post(setup_submit))
376        .route(LOGIN_ROUTE, get(login_form).post(login_submit))
377        .route(FEDERATION_SETUP_ROUTE, get(federation_setup))
378        .route(ADD_SETUP_CODE_ROUTE, post(post_add_setup_code))
379        .route(RESET_SETUP_CODES_ROUTE, post(post_reset_setup_codes))
380        .route(START_DKG_ROUTE, post(post_start_dkg))
381        .with_static_routes()
382        .with_state(UiState::new(api))
383}
