11        .spendable_utxos

13        .map(|utxo| utxo.amount.to_sat())

18        .unsigned_peg_out_txos
19        .iter()

22        .sum();
23
24    // Calculate pending incoming change

26        .unsigned_change_utxos
27        .iter()
28        .chain(wallet_summary.unconfirmed_change_utxos.iter())
29        .map(|txo| txo.amount.to_sat())
