119// Main dashboard view
120async fn dashboard_view(
121    State(state): State<AuthState<DynDashboardApi>>,
122    jar: CookieJar,
123) -> impl IntoResponse {
124    if !check_auth(&state.auth_cookie_name, &state.auth_cookie_value, &jar).await {
125        return Redirect::to("/login").into_response();
126    }
127
128    let guardian_names = state.api.guardian_names().await;
129    let federation_name = state.api.federation_name().await;
130    let session_count = state.api.session_count().await;
131    let consensus_ord_latency = state.api.consensus_ord_latency().await;
132    let p2p_connection_status = state.api.p2p_connection_status().await;
133    let invite_code = state.api.federation_invite_code().await;
134    let audit_summary = state.api.federation_audit().await;
135
136    let mut content = html! {
137        div class="row gy-4" {
138            div class="col-md-6" {
139                div class="card h-100" {
140                    div class="card-header dashboard-header" { (federation_name) }
141                    div class="card-body" {
142                        div class="alert alert-info" {
143                            "Session Count: " strong { (session_count) }
144                        }
145                        table class="table table-sm mb-0" {
146                            thead {
147                                tr {
148                                    th { "Guardian ID" }
149                                    th { "Guardian Name" }
150                                }
151                            }
152                            tbody {
153                                @for (guardian_id, name) in guardian_names {
154                                    tr {
155                                        td { (guardian_id.to_string()) }
156                                        td { (name) }
157                                    }
158                                }
159                            }
160                        }
161                    }
162                }
163            }
164
165            // Invite Code Column
166            div class="col-md-6" {
167                (invite_code::invite_code_card())
168            }
169        }
170
171        // Render the invite code modal
172        (invite_code::invite_code_modal(&invite_code))
173
174        // Second row: Audit Summary and Peer Status
175        div class="row gy-4 mt-2" {
176            // Audit Information Column
177            div class="col-lg-6" {
178                (audit::render(&audit_summary))
179            }
180
181            // Peer Connection Status Column
182            div class="col-lg-6" {
183                (latency::render(consensus_ord_latency, &p2p_connection_status))
184            }
185        }
186    };
187
188    // Conditionally add Lightning V2 UI if the module is available
189    if let Some(lightning) = state.api.get_module::<fedimint_lnv2_server::Lightning>() {
190        content = html! {
191            (content)
192            (lnv2::render(lightning).await)
193        };
194    }
195
196    // Conditionally add Wallet UI if the module is available
197    if let Some(wallet_module) = state.api.get_module::<fedimint_wallet_server::Wallet>() {
198        content = html! {
199            (content)
200            (wallet::render(wallet_module).await)
201        };
202    }
203
204    // Conditionally add Meta UI if the module is available
205    if let Some(meta_module) = state.api.get_module::<fedimint_meta_server::Meta>() {
206        content = html! {
207            (content)
208            (meta::render(meta_module).await)
209        };
210    }
211
212    Html(dashboard_layout(content).into_string()).into_response()
213}

215pub fn start(
216    api: DynDashboardApi,
217    ui_bind: SocketAddr,
218    task_handle: TaskHandle,
219) -> Pin<Box<dyn Future<Output = ()> + Send>> {
220    // Create a basic router with core routes
221    let mut app = Router::new()
222        .route("/", get(dashboard_view))
223        .route("/login", get(login_form).post(login_submit));
224
225    // Only add LNv2 gateway routes if the module exists
226    if api
227        .get_module::<fedimint_lnv2_server::Lightning>()
228        .is_some()
229    {
230        app = app
231            .route("/lnv2_gateway_add", post(lnv2::add_gateway))
232            .route("/lnv2_gateway_remove", post(lnv2::remove_gateway));
233    }
234
235    // Only add Meta module routes if the module exists
236    if api.get_module::<fedimint_meta_server::Meta>().is_some() {
237        app = app.route("/meta/submit", post(meta::submit_meta_value))
238    }
239
240    // Finalize the router with state
241    let app = app.with_state(AuthState::new(api));
242
243    Box::pin(async move {
244        let listener = TcpListener::bind(ui_bind)
245            .await
246            .expect("Failed to bind dashboard UI");
247
248        axum::serve(listener, app.into_make_service())
249            .with_graceful_shutdown(task_handle.make_shutdown_rx())
250            .await
251            .expect("Failed to serve dashboard UI");
252    })
253}
