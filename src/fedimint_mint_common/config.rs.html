12// The maximum size of an E-Cash note (1,000,000 coins)
13// Changing this value is considered a breaking change because it is not saved

15const MAX_DENOMINATION_SIZE: Amount = Amount::from_bitcoins(1_000_000);

100#[derive(Debug, Clone, Eq, PartialEq, Hash, Serialize, Deserialize, Encodable, Decodable)]
101pub struct FeeConsensus {
102    base: Amount,
103    parts_per_million: u64,
104}
105
106impl FeeConsensus {
107    /// The mint module will charge a non-configurable base fee of one hundred
108    /// millisatoshis per transaction input and output to account for the costs
109    /// incurred by the federation for processing the transaction. On top of
110    /// that the federation may charge a additional relative fee per input and
111    /// output of up to one thousand parts per million which is equal to one
112    /// tenth of one percent.
113    ///
114    /// # Errors
115    /// - This constructor returns an error if the relative fee is in excess of
116    ///   one thousand parts per million.
117    pub fn new(parts_per_million: u64) -> anyhow::Result<Self> {
118        anyhow::ensure!(
119            parts_per_million <= 1_000,
120            "Relative fee over one thousand parts per million is excessive"
121        );
122
123        Ok(Self {
124            base: Amount::from_msats(100),
125            parts_per_million,
126        })
127    }
128
129    pub fn zero() -> Self {
130        Self {
131            base: Amount::ZERO,
132            parts_per_million: 0,
133        }
134    }
135
136    pub fn fee(&self, amount: Amount) -> Amount {
137        Amount::from_msats(self.fee_msats(amount.msats))
138    }
139
140    fn fee_msats(&self, msats: u64) -> u64 {
141        msats
142            .saturating_mul(self.parts_per_million)
143            .saturating_div(1_000_000)
144            .checked_add(self.base.msats)
145            .expect("The division creates sufficient headroom to add the base fee")
146    }
147}

149#[test]
150fn test_fee_consensus() {
151    let fee_consensus = FeeConsensus::new(1_000).expect("Relative fee is within range");
152
153    assert_eq!(
154        fee_consensus.fee(Amount::from_msats(999)),
155        Amount::from_msats(100)
156    );
157
158    assert_eq!(
159        fee_consensus.fee(Amount::from_sats(1)),
160        Amount::from_msats(100) + Amount::from_msats(1)
161    );
162
163    assert_eq!(
164        fee_consensus.fee(Amount::from_sats(1000)),
165        Amount::from_sats(1) + Amount::from_msats(100)
166    );
167
168    assert_eq!(
169        fee_consensus.fee(Amount::from_bitcoins(1)),
170        Amount::from_sats(100_000) + Amount::from_msats(100)
171    );
172
173    assert_eq!(
174        fee_consensus.fee(Amount::from_bitcoins(100_000)),
175        Amount::from_bitcoins(100) + Amount::from_msats(100)
176    );
177}
