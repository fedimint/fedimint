29fn random_test_dir_suffix() -> String {
30    rand::thread_rng()
31        .sample_iter(&Alphanumeric)
32        .filter(u8::is_ascii_digit)
33        .take(3)
34        .map(char::from)
35        .collect::<String>()
36}

68impl CommonArgs {
69    pub fn mk_test_dir(&self) -> Result<PathBuf> {
70        if self.skip_setup {
71            ensure!(
72                self.test_dir.is_some(),
73                "When using `--skip-setup`, `--test-dir` must be set"
74            );
75        }
76        let path = self.test_dir();
77
78        std::fs::create_dir_all(&path)
79            .with_context(|| format!("Creating tmp directory {}", path.display()))?;
80
81        Ok(path)
82    }

85        self.test_dir.clone().unwrap_or_else(|| {

87                "devimint-{}-{}",
88                std::process::id(),
89                self.link_test_dir_suffix
90            ))
91        })
92    }

124pub async fn setup(arg: CommonArgs) -> Result<(ProcessManager, TaskGroup)> {
125    let test_dir = &arg.mk_test_dir()?;
126    mkdir(test_dir.clone()).await?;
127    let logs_dir: PathBuf = test_dir.join("logs");
128    mkdir(logs_dir.clone()).await?;
129
130    let log_file = fs::OpenOptions::new()
131        .write(true)
132        .create(true)
133        .append(true)
134        .open(logs_dir.join("devimint.log"))
135        .await?
136        .into_std()
137        .await;
138
139    fedimint_logging::TracingSetup::default()
140        .with_file(Some(log_file))
141        // jsonrpsee is expected to fail during startup
142        .with_directive("jsonrpsee-client=off")
143        .init()?;
144
145    let globals =
146        vars::Global::new(test_dir, arg.num_feds, arg.fed_size, arg.offline_nodes).await?;
147
148    if let Some(link_test_dir) = arg.link_test_dir.as_ref() {
149        update_test_dir_link(link_test_dir, &arg.test_dir()).await?;
150    }
151    info!(target: LOG_DEVIMINT, path=%globals.FM_DATA_DIR.display() , "Devimint data dir");
152
153    let mut env_string = String::new();
154    for (var, value) in globals.vars() {
155        debug!(var, value, "Env variable set");
156        writeln!(env_string, r#"export {var}="{value}""#)?; // hope that value doesn't contain a "
157        // TODO: Audit that the environment access only happens in single-threaded code.
158        unsafe { std::env::set_var(var, value) };
159    }
160    write_overwrite_async(globals.FM_TEST_DIR.join("env"), env_string).await?;
161    let process_mgr = ProcessManager::new(globals);
162    let task_group = TaskGroup::new();
163    task_group.install_kill_handler();
164    Ok((process_mgr, task_group))
165}

167pub async fn update_test_dir_link(
168    link_test_dir: &Path,
169    test_dir: &Path,
170) -> Result<(), anyhow::Error> {
171    let make_link = match fs::read_link(link_test_dir).await {
172        Ok(existing) => {
173            if existing == test_dir {
174                false
175            } else {
176                debug!(
177                    old = %existing.display(),
178                    new = %test_dir.display(),
179                    link = %link_test_dir.display(),
180                    "Updating exinst test dir link"
181                );
182
183                fs::remove_file(link_test_dir).await?;
184                true
185            }
186        }
187        _ => true,
188    };
189    if make_link {
190        debug!(src = %test_dir.display(), dst = %link_test_dir.display(), "Linking test dir");
191        fs::symlink(&test_dir, link_test_dir).await?;
192    }
193    Ok(())
194}

196pub async fn cleanup_on_exit<T>(
197    main_process: impl futures::Future<Output = Result<T>>,
198    task_group: TaskGroup,
199) -> Result<Option<T>> {
200    match task_group
201        .make_handle()
202        .cancel_on_shutdown(main_process)
203        .await
204    {
205        Err(_) => {
206            info!("Received shutdown signal before finishing main process, exiting early");
207            Ok(None)
208        }
209        Ok(Ok(v)) => {
210            debug!(target: LOG_DEVIMINT, "Main process finished successfully, shutting down task group");
211            task_group
212                .shutdown_join_all(Duration::from_secs(30))
213                .await?;
214
215            // the caller can drop the v after shutdown
216            Ok(Some(v))
217        }
218        Ok(Err(err)) => {
219            warn!(target: LOG_DEVIMINT, err = %err.fmt_compact_anyhow(), "Main process failed, will shutdown");
220            Err(err)
221        }
222    }
223}

234pub async fn handle_command(cmd: Cmd, common_args: CommonArgs) -> Result<()> {
235    match cmd {
236        Cmd::ExternalDaemons { exec } => {
237            let (process_mgr, task_group) = setup(common_args).await?;
238            let _daemons =
239                write_ready_file(&process_mgr.globals, external_daemons(&process_mgr).await)
240                    .await?;
241            if let Some(exec) = exec {
242                exec_user_command(exec).await?;
243                task_group.shutdown();
244            }
245            task_group.make_handle().make_shutdown_rx().await;
246        }
247        Cmd::DevFed { exec } => {
248            trace!(target: LOG_DEVIMINT, "Starting dev fed");
249            let start_time = Instant::now();
250            let skip_setup = common_args.skip_setup;
251            let (process_mgr, task_group) = setup(common_args).await?;
252            let main = {
253                let task_group = task_group.clone();
254                async move {
255                    let dev_fed = DevJitFed::new(&process_mgr, skip_setup)?;
256
257                    let pegin_start_time = Instant::now();
258                    debug!(target: LOG_DEVIMINT, "Peging in client and gateways");
259
260                    if !skip_setup {
261                        const GW_PEGIN_AMOUNT: u64 = 1_000_000;
262                        const CLIENT_PEGIN_AMOUNT: u64 = 1_000_000;
263
264                        let (operation_id, (), ()) = tokio::try_join!(
265                            async {
266                                let (address, operation_id) =
267                                    dev_fed.internal_client().await?.get_deposit_addr().await?;
268                                debug!(
269                                    target: LOG_DEVIMINT,
270                                    %address,
271                                    %operation_id,
272                                    "Sending funds to client deposit addr"
273                                );
274                                dev_fed
275                                    .bitcoind()
276                                    .await?
277                                    .send_to(address, CLIENT_PEGIN_AMOUNT)
278                                    .await?;
279                                Ok(operation_id)
280                            },
281                            async {
282                                let address = dev_fed
283                                    .gw_lnd_registered()
284                                    .await?
285                                    .get_pegin_addr(&dev_fed.fed().await?.calculate_federation_id())
286                                    .await?;
287                                debug!(
288                                    target: LOG_DEVIMINT,
289                                    %address,
290                                    "Sending funds to LND deposit addr"
291                                );
292                                dev_fed
293                                    .bitcoind()
294                                    .await?
295                                    .send_to(address, GW_PEGIN_AMOUNT)
296                                    .await
297                                    .map(|_| ())
298                            },
299                            async {
300                                if crate::util::supports_lnv2() {
301                                    let gw_ldk = dev_fed.gw_ldk_connected().await?;
302                                    let address = gw_ldk
303                                        .get_pegin_addr(
304                                            &dev_fed.fed().await?.calculate_federation_id(),
305                                        )
306                                        .await?;
307                                    debug!(
308                                        target: LOG_DEVIMINT,
309                                        %address,
310                                        "Sending funds to LDK deposit addr"
311                                    );
312                                    dev_fed
313                                        .bitcoind()
314                                        .await?
315                                        .send_to(address, GW_PEGIN_AMOUNT)
316                                        .await
317                                        .map(|_| ())
318                                } else {
319                                    Ok(())
320                                }
321                            },
322                        )?;
323
324                        dev_fed.bitcoind().await?.mine_blocks_no_wait(11).await?;
325                        dev_fed
326                            .internal_client()
327                            .await?
328                            .await_deposit(&operation_id)
329                            .await?;
330
331                        info!(
332                            target: LOG_DEVIMINT,
333                            elapsed_ms = %pegin_start_time.elapsed().as_millis(),
334                            "Pegins completed"
335                        );
336                    }
337
338                    // TODO: Audit that the environment access only happens in single-threaded code.
339                    unsafe {
340                        std::env::set_var(FM_INVITE_CODE_ENV, dev_fed.fed().await?.invite_code()?);
341                    };
342
343                    dev_fed.finalize(&process_mgr).await?;
344
345                    let daemons = write_ready_file(&process_mgr.globals, Ok(dev_fed)).await?;
346
347                    info!(target: LOG_DEVIMINT, elapsed_ms = %start_time.elapsed().as_millis(), "Devfed ready");
348                    if let Some(exec) = exec {
349                        debug!(target: LOG_DEVIMINT, "Starting exec command");
350                        exec_user_command(exec).await?;
351                        task_group.shutdown();
352                    }
353
354                    debug!(target: LOG_DEVIMINT, "Waiting for group task shutdown");
355                    task_group.make_handle().make_shutdown_rx().await;
356
357                    Ok::<_, anyhow::Error>(daemons)
358                }
359            };
360            if let Some(fed) = cleanup_on_exit(main, task_group).await? {
361                fed.fast_terminate().await;
362            }
363        }
364        Cmd::Rpc(rpc_cmd) => rpc_command(rpc_cmd, common_args).await?,
365        Cmd::RunUi => {
366            let (process_mgr, task_group) = setup(common_args).await?;
367            let task_group_clone = task_group.clone();
368            let main = async {
369                let result = run_ui(&process_mgr).await;
370                let daemons = write_ready_file(&process_mgr.globals, result).await?;
371
372                debug!(target: LOG_DEVIMINT, "Waiting for group task shutdown");
373                task_group_clone.make_handle().make_shutdown_rx().await;
374
375                Ok::<_, anyhow::Error>(daemons)
376            };
377            Box::pin(cleanup_on_exit(main, task_group)).await?;
378        }
379    }
380    Ok(())
381}

383pub async fn exec_user_command(path: Vec<ffi::OsString>) -> Result<(), anyhow::Error> {
384    let cmd_str = path
385        .join(ffi::OsStr::new(" "))
386        .to_string_lossy()
387        .to_string();
388
389    let path_with_aliases = if let Some(existing_path) = env::var_os("PATH") {
390        let mut path = devimint_static_data_dir();
391        path.push("/aliases:");
392        path.push(existing_path);
393        path
394    } else {
395        let mut path = devimint_static_data_dir();
396        path.push("/aliases");
397        path
398    };
399    debug!(target: LOG_DEVIMINT, cmd = %cmd_str, "Executing user command");
400    if !tokio::process::Command::new(&path[0])
401        .args(&path[1..])
402        .env("PATH", path_with_aliases)
403        .kill_on_drop(true)
404        .status()
405        .await
406        .with_context(|| format!("Executing user command failed: {cmd_str}"))?
407        .success()
408    {
409        error!(cmd = %cmd_str, "User command failed");
410        return Err(anyhow!("User command failed: {cmd_str}"));
411    }
412    Ok(())
413}

426pub async fn rpc_command(rpc: RpcCmd, common: CommonArgs) -> Result<()> {
427    fedimint_logging::TracingSetup::default().init()?;
428    match rpc {
429        RpcCmd::Env => {
430            let env_file = common.test_dir().join("env");
431            poll("env file", || async {
432                if fs::try_exists(&env_file)
433                    .await
434                    .context("env file")
435                    .map_err(ControlFlow::Continue)?
436                {
437                    Ok(())
438                } else {
439                    Err(ControlFlow::Continue(anyhow!("env file not found")))
440                }
441            })
442            .await?;
443            let env = fs::read_to_string(&env_file).await?;
444            print!("{env}");
445            Ok(())
446        }
447        RpcCmd::Wait => {
448            let ready_file = common.test_dir().join("ready");
449            poll("ready file", || async {
450                if fs::try_exists(&ready_file)
451                    .await
452                    .context("ready file")
453                    .map_err(ControlFlow::Continue)?
454                {
455                    Ok(())
456                } else {
457                    Err(ControlFlow::Continue(anyhow!("ready file not found")))
458                }
459            })
460            .await?;
461            let env = fs::read_to_string(&ready_file).await?;
462            print!("{env}");
463
464            // Append invite code to devimint env
465            let test_dir = &common.test_dir();
466            let env_file = test_dir.join("env");
467            let invite_file = test_dir.join("cfg/invite-code");
468            if fs::try_exists(&env_file).await.ok().unwrap_or(false)
469                && fs::try_exists(&invite_file).await.ok().unwrap_or(false)
470            {
471                let invite = fs::read_to_string(&invite_file).await?;
472                let mut env_string = fs::read_to_string(&env_file).await?;
473                writeln!(env_string, r#"export FM_INVITE_CODE="{invite}""#)?;
474                // TODO: Audit that the environment access only happens in single-threaded code.
475                unsafe { std::env::set_var(FM_INVITE_CODE_ENV, invite) };
476                write_overwrite_async(env_file, env_string).await?;
477            }
478
479            Ok(())
480        }
481    }
482}

484async fn run_ui(process_mgr: &ProcessManager) -> Result<(Vec<Fedimintd>, ExternalDaemons)> {
485    let externals = external_daemons(process_mgr).await?;
486    let fed_size = process_mgr.globals.FM_FED_SIZE;
487    let fedimintds = futures::future::try_join_all((0..fed_size).map(|peer| {
488        let bitcoind = externals.bitcoind.clone();
489        async move {
490            let peer_port = 10000 + 8137 + peer * 2;
491            let api_port = peer_port + 1;
492            let metrics_port = 3510 + peer;
493
494            // TODO: we should use this, and override ports that need to be fixed by an env
495            // var and/or just scap `run-ui` altogether
496            // let peer_id = PeerId::new(peer as u16);
497            // let peer_env_vars = vars::Fedimintd::init(
498            //     &process_mgr.globals,
499            //     "ui-test-federation".to_string(),
500            //     peer_id,
501            //     process_mgr.globals.fedimintd_overrides.peer_expect(peer_id),
502            // )
503            // .await?;
504            let vars = vars::Fedimintd {
505                FM_BIND_P2P: format!("127.0.0.1:{peer_port}"),
506                FM_P2P_URL: format!("fedimint://127.0.0.1:{peer_port}"),
507                FM_BIND_API: format!("127.0.0.1:{api_port}"),
508                FM_API_URL: format!("ws://127.0.0.1:{api_port}"),
509                FM_DATA_DIR: process_mgr
510                    .globals
511                    .FM_DATA_DIR
512                    .join(format!("fedimintd-{peer}")),
513                FM_BIND_METRICS_API: format!("127.0.0.1:{metrics_port}"),
514                FM_FORCE_BITCOIN_RPC_URL: format!(
515                    "http://bitcoin:bitcoin@127.0.0.1:{}",
516                    process_mgr.globals.FM_PORT_BTC_RPC
517                ),
518                FM_FORCE_BITCOIN_RPC_KIND: "bitcoind".into(),
519                FM_IROH_P2P_SECRET_KEY_OVERRIDE: String::new(),
520                FM_IROH_API_SECRET_KEY_OVERRIDE: String::new(),
521            };
522            let fm = Fedimintd::new(
523                process_mgr,
524                bitcoind.clone(),
525                peer,
526                &vars,
527                "default".to_string(),
528            )
529            .await?;
530            let server_addr = &vars.FM_BIND_API;
531
532            poll("waiting for api startup", || async {
533                TcpStream::connect(server_addr)
534                    .await
535                    .context("connect to api")
536                    .map_err(ControlFlow::Continue)
537            })
538            .await?;
539
540            anyhow::Ok(fm)
541        }
542    }))
543    .await?;
544
545    Ok((fedimintds, externals))
