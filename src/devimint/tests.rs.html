41pub struct Stats {
42    pub min: Duration,
43    pub avg: Duration,
44    pub median: Duration,
45    pub p90: Duration,
46    pub max: Duration,
47    pub sum: Duration,

50impl std::fmt::Display for Stats {
51    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
52        write!(f, "min: {:.1}s", self.min.as_secs_f32())?;
53        write!(f, ", avg: {:.1}s", self.avg.as_secs_f32())?;
54        write!(f, ", median: {:.1}s", self.median.as_secs_f32())?;
55        write!(f, ", p90: {:.1}s", self.p90.as_secs_f32())?;
56        write!(f, ", max: {:.1}s", self.max.as_secs_f32())?;
57        write!(f, ", sum: {:.1}s", self.sum.as_secs_f32())?;
58        Ok(())
59    }
60}

62pub fn stats_for(mut v: Vec<Duration>) -> Stats {
63    assert!(!v.is_empty());
64    v.sort();
65    let n = v.len();
66    let min = v.first().unwrap().to_owned();
67    let max = v.iter().last().unwrap().to_owned();
68    let median = v[n / 2];
69    let sum: Duration = v.iter().sum();
70    let avg = sum / n as u32;
71    let p90 = v[(n as f32 * 0.9) as usize];
72    Stats {
73        min,
74        avg,
75        median,
76        p90,
77        max,
78        sum,
79    }
80}

110pub async fn latency_tests(
111    dev_fed: DevFed,
112    r#type: LatencyTest,
113    upgrade_clients: Option<&UpgradeClients>,
114    iterations: usize,
115    assert_thresholds: bool,
116) -> Result<()> {
117    log_binary_versions().await?;
118
119    let DevFed {
120        fed,
121        gw_lnd,
122        gw_ldk,
123        ..
124    } = dev_fed;
125
126    let max_p90_factor = 5.0;
127    let p90_median_factor = 10;
128
129    let client = match upgrade_clients {
130        Some(c) => match r#type {
131            LatencyTest::Reissue => c.reissue_client.clone(),
132            LatencyTest::LnSend => c.ln_send_client.clone(),
133            LatencyTest::LnReceive => c.ln_receive_client.clone(),
134            LatencyTest::FmPay => c.fm_pay_client.clone(),
135            LatencyTest::Restore => bail!("no reusable upgrade client for restore"),
136        },
137        None => fed.new_joined_client("latency-tests-client").await?,
138    };
139
140    let initial_balance_sats = 100_000_000;
141    fed.pegin_client(initial_balance_sats, &client).await?;
142
143    let lnd_gw_id = gw_lnd.gateway_id().await?;
144
145    match r#type {
146        LatencyTest::Reissue => {
147            info!("Testing latency of reissue");
148            let mut reissues = Vec::with_capacity(iterations);
149            let amount_per_iteration_msats =
150                // use a highest 2^-1 amount that fits, to try to use as many notes as possible
151                ((initial_balance_sats * 1000 / iterations as u64).next_power_of_two() >> 1) - 1;
152            for _ in 0..iterations {
153                let notes = cmd!(client, "spend", amount_per_iteration_msats.to_string())
154                    .out_json()
155                    .await?["notes"]
156                    .as_str()
157                    .context("note must be a string")?
158                    .to_owned();
159
160                let start_time = Instant::now();
161                cmd!(client, "reissue", notes).run().await?;
162                reissues.push(start_time.elapsed());
163            }
164            let reissue_stats = stats_for(reissues);
165            println!("### LATENCY REISSUE: {reissue_stats}");
166
167            if assert_thresholds {
168                assert!(reissue_stats.median < Duration::from_secs(10));
169                assert!(reissue_stats.p90 < reissue_stats.median * p90_median_factor);
170                assert!(
171                    reissue_stats.max.as_secs_f64()
172                        < reissue_stats.p90.as_secs_f64() * max_p90_factor
173                );
174            }
175        }
176        LatencyTest::LnSend => {
177            info!("Testing latency of ln send");
178            let mut ln_sends = Vec::with_capacity(iterations);
179            for _ in 0..iterations {
180                let invoice = gw_ldk.create_invoice(1_000_000).await?;
181                let start_time = Instant::now();
182                ln_pay(&client, invoice.to_string(), lnd_gw_id.clone()).await?;
183                gw_ldk
184                    .wait_bolt11_invoice(invoice.payment_hash().consensus_encode_to_vec())
185                    .await?;
186                ln_sends.push(start_time.elapsed());
187            }
188            let ln_sends_stats = stats_for(ln_sends);
189            println!("### LATENCY LN SEND: {ln_sends_stats}");
190
191            if assert_thresholds {
192                assert!(ln_sends_stats.median < Duration::from_secs(10));
193                assert!(ln_sends_stats.p90 < ln_sends_stats.median * p90_median_factor);
194                assert!(
195                    ln_sends_stats.max.as_secs_f64()
196                        < ln_sends_stats.p90.as_secs_f64() * max_p90_factor
197                );
198            }
199        }
200        LatencyTest::LnReceive => {
201            info!("Testing latency of ln receive");
202            let mut ln_receives = Vec::with_capacity(iterations);
203
204            // give lnd some funds
205            let invoice = gw_ldk.create_invoice(10_000_000).await?;
206            ln_pay(&client, invoice.to_string(), lnd_gw_id.clone()).await?;
207
208            for _ in 0..iterations {
209                let invoice = ln_invoice(
210                    &client,
211                    Amount::from_msats(100_000),
212                    "latency-over-lnd-gw".to_string(),
213                    lnd_gw_id.clone(),
214                )
215                .await?
216                .invoice;
217
218                let start_time = Instant::now();
219                gw_ldk
220                    .pay_invoice(
221                        Bolt11Invoice::from_str(&invoice).expect("Could not parse invoice"),
222                    )
223                    .await?;
224                ln_receives.push(start_time.elapsed());
225            }
226            let ln_receives_stats = stats_for(ln_receives);
227            println!("### LATENCY LN RECV: {ln_receives_stats}");
228
229            if assert_thresholds {
230                assert!(ln_receives_stats.median < Duration::from_secs(10));
231                assert!(ln_receives_stats.p90 < ln_receives_stats.median * p90_median_factor);
232                assert!(
233                    ln_receives_stats.max.as_secs_f64()
234                        < ln_receives_stats.p90.as_secs_f64() * max_p90_factor
235                );
236            }
237        }
238        LatencyTest::FmPay => {
239            info!("Testing latency of internal payments within a federation");
240            let mut fm_internal_pay = Vec::with_capacity(iterations);
241            let sender = fed.new_joined_client("internal-swap-sender").await?;
242            fed.pegin_client(10_000_000, &sender).await?;
243            for _ in 0..iterations {
244                let recv = cmd!(
245                    client,
246                    "ln-invoice",
247                    "--amount=1000000msat",
248                    "--description=internal-swap-invoice",
249                    "--force-internal"
250                )
251                .out_json()
252                .await?;
253
254                let invoice = recv["invoice"]
255                    .as_str()
256                    .context("invoice must be string")?
257                    .to_owned();
258                let recv_op = recv["operation_id"]
259                    .as_str()
260                    .context("operation id must be string")?
261                    .to_owned();
262
263                let start_time = Instant::now();
264                cmd!(sender, "ln-pay", invoice, "--force-internal")
265                    .run()
266                    .await?;
267
268                cmd!(client, "await-invoice", recv_op).run().await?;
269                fm_internal_pay.push(start_time.elapsed());
270            }
271            let fm_pay_stats = stats_for(fm_internal_pay);
272
273            println!("### LATENCY FM PAY: {fm_pay_stats}");
274
275            if assert_thresholds {
276                assert!(fm_pay_stats.median < Duration::from_secs(15));
277                assert!(fm_pay_stats.p90 < fm_pay_stats.median * p90_median_factor);
278                assert!(
279                    fm_pay_stats.max.as_secs_f64()
280                        < fm_pay_stats.p90.as_secs_f64() * max_p90_factor
281                );
282            }
283        }
284        LatencyTest::Restore => {
285            info!("Testing latency of restore");
286            let backup_secret = cmd!(client, "print-secret").out_json().await?["secret"]
287                .as_str()
288                .map(ToOwned::to_owned)
289                .unwrap();
290            if !is_env_var_set(FM_DEVIMINT_RUN_DEPRECATED_TESTS_ENV) {
291                info!("Skipping tests, as in previous versions restore was very slow to test");
292                return Ok(());
293            }
294
295            let start_time = Instant::now();
296            let restore_client = Client::create("restore").await?;
297            cmd!(
298                restore_client,
299                "restore",
300                "--mnemonic",
301                &backup_secret,
302                "--invite-code",
303                fed.invite_code()?
304            )
305            .run()
306            .await?;
307            let restore_time = start_time.elapsed();
308
309            println!("### LATENCY RESTORE: {restore_time:?}");
310
311            if assert_thresholds {
312                if crate::util::is_backwards_compatibility_test() {
313                    assert!(restore_time < Duration::from_secs(160));
314                } else {
315                    assert!(restore_time < Duration::from_secs(30));
316                }
317            }
318        }
319    }
320
321    Ok(())
322}

333async fn stress_test_fed(dev_fed: &DevFed, clients: Option<&UpgradeClients>) -> anyhow::Result<()> {
334    use futures::FutureExt;
335
336    // local environments can fail due to latency thresholds, however this shouldn't
337    // cause the upgrade test to fail
338    let assert_thresholds = false;
339
340    // running only one iteration greatly improves the total test time while still
341    // testing the same types of database entries
342    let iterations = 1;
343
344    // skip restore test for client upgrades, since restoring a client doesn't
345    // require a persistent data dir
346    let restore_test = if clients.is_some() {
347        futures::future::ok(()).right_future()
348    } else {
349        latency_tests(
350            dev_fed.clone(),
351            LatencyTest::Restore,
352            clients,
353            iterations,
354            assert_thresholds,
355        )
356        .left_future()
357    };
358
359    // tests are run in sequence so parallelism is controlled using gnu `parallel`
360    // in `upgrade-test.sh`
361    latency_tests(
362        dev_fed.clone(),
363        LatencyTest::Reissue,
364        clients,
365        iterations,
366        assert_thresholds,
367    )
368    .await?;
369
370    latency_tests(
371        dev_fed.clone(),
372        LatencyTest::LnSend,
373        clients,
374        iterations,
375        assert_thresholds,
376    )
377    .await?;
378
379    latency_tests(
380        dev_fed.clone(),
381        LatencyTest::LnReceive,
382        clients,
383        iterations,
384        assert_thresholds,
385    )
386    .await?;
387
388    latency_tests(
389        dev_fed.clone(),
390        LatencyTest::FmPay,
391        clients,
392        iterations,
393        assert_thresholds,
394    )
395    .await?;
396
397    restore_test.await?;
398
399    Ok(())
400}

542                info!(
543                    ?gatewayd_version,
544                    ?gateway_cli_version,
545                    "### gateway passed stress test for version",
546                );
547            }
548
549            info!("## gatewayd upgraded all binaries successfully");
550        }
551    }
552    Ok(())
553}

1892/// Test that client can detect federation config changes when servers restart
1893/// with new module configurations
1894///
1895/// This test starts a fresh federation, dumps the client config, then stops all
1896/// servers and modifies their configs by adding a new meta module instance. The

1898pub async fn test_client_config_change_detection(
1899    dev_fed: DevFed,
1900    process_mgr: &ProcessManager,
1901) -> Result<()> {

1905    let fedimintd_version = crate::util::FedimintdCmd::version_or_default().await;
1906
1907    if fedimint_cli_version < *VERSION_0_9_0_ALPHA {
1908        info!(target: LOG_DEVIMINT, "Skipping the test - fedimint-cli too old");
1909        return Ok(());
1910    }
1911
1912    if fedimintd_version < *VERSION_0_9_0_ALPHA {
1913        info!(target: LOG_DEVIMINT, "Skipping the test - fedimintd too old");
1914        return Ok(());
1915    }
1916
1917    let DevFed { mut fed, .. } = dev_fed;
1918    let peer_ids: Vec<_> = fed.member_ids().collect();
1919
1920    fed.await_all_peers().await?;
1921
1922    let client = fed.new_joined_client("config-change-test-client").await?;
1923
1924    info!(target: LOG_DEVIMINT, "Getting initial client configuration...");
1925    let initial_config = cmd!(client, "config")
1926        .out_json()
1927        .await
1928        .context("Failed to get initial client config")?;
1929
1930    info!(target: LOG_DEVIMINT, "Initial config modules: {:?}", initial_config["modules"].as_object().unwrap().keys().collect::<Vec<_>>());
1931
1932    let data_dir = env::var(FM_DATA_DIR_ENV)?;
1933    let config_dir = PathBuf::from(&data_dir);
1934
1935    // Shutdown all federation servers
1936    //
1937    // In prod. one would probably use a coordinated shutdown, just to be
1938    // careful, but since the change is only adding a new module that does
1939    // not submit CIs without user/admin interaction, there is
1940    // no way for the consensus to diverge.
1941    info!(target: LOG_DEVIMINT, "Shutting down all federation servers...");
1942    fed.terminate_all_servers().await?;
1943
1944    // Wait for servers to fully shutdown
1945    fedimint_core::task::sleep_in_test("wait for federation shutdown", Duration::from_secs(2))
1946        .await;
1947
1948    info!(target: LOG_DEVIMINT, "Modifying server configurations to add new meta module...");
1949    modify_server_configs(&config_dir, &peer_ids).await?;
1950
1951    // Restart all servers with modified configs
1952    info!(target: LOG_DEVIMINT, "Restarting all servers with modified configurations...");
1953    for peer_id in peer_ids {
1954        fed.start_server(process_mgr, peer_id.to_usize()).await?;
1955    }
1956
1957    // Wait for federation to stabilize
1958    info!(target: LOG_DEVIMINT, "Wait for peers to get back up");
1959    fed.await_all_peers().await?;
1960
1961    // Use fedimint-cli dev wait to let the client read the new config in background
1962    info!(target: LOG_DEVIMINT, "Waiting for client to fetch updated configuration...");
1963    cmd!(client, "dev", "wait", "3")
1964        .run()
1965        .await
1966        .context("Failed to wait for client config update")?;
1967
1968    // Test that client switched to the new config
1969    info!(target: LOG_DEVIMINT, "Testing client detection of configuration changes...");
1970    let updated_config = cmd!(client, "config")
1971        .out_json()
1972        .await
1973        .context("Failed to get updated client config")?;
1974
1975    info!(target: LOG_DEVIMINT, "Updated config modules: {:?}", updated_config["modules"].as_object().unwrap().keys().collect::<Vec<_>>());
1976
1977    // Verify that the configuration has changed (new meta module should be present)
1978    let initial_modules = initial_config["modules"].as_object().unwrap();
1979    let updated_modules = updated_config["modules"].as_object().unwrap();
1980
1981    anyhow::ensure!(
1982        updated_modules.len() > initial_modules.len(),
1983        "Expected more modules in updated config. Initial: {}, Updated: {}",
1984        initial_modules.len(),
1985        updated_modules.len()
1986    );
1987
1988    // Check if a new meta module was added
1989    let new_meta_module = updated_modules.iter().find(|(module_id, module_config)| {
1990        module_config["kind"].as_str() == Some("meta") && !initial_modules.contains_key(*module_id)
1991    });
1992
1993    let new_meta_module_id = new_meta_module
1994        .map(|(id, _)| id)
1995        .with_context(|| "Expected to find new meta module in updated configuration")?;
1996
1997    info!(target: LOG_DEVIMINT, "Found new meta module with id: {}", new_meta_module_id);
1998
1999    // Verify client operations still work with the new configuration
2000    info!(target: LOG_DEVIMINT, "Verifying client operations work with new configuration...");
2001    let final_info = cmd!(client, "info")
2002        .out_json()
2003        .await
2004        .context("Client info command failed with updated configuration")?;
2005
2006    info!(target: LOG_DEVIMINT, "Client successfully adapted to configuration changes: {:?}", final_info["federation_id"]);
2007
2008    Ok(())
2009}

2019/// Modify configuration files for a single peer to add a new meta module
2020/// instance
2021async fn modify_single_peer_config(config_dir: &Path, peer_id: PeerId) -> Result<()> {
2022    use fedimint_aead::{encrypted_write, get_encryption_key};
2023    use fedimint_core::core::ModuleInstanceId;
2024    use fedimint_server::config::io::read_server_config;
2025    use serde_json::Value;
2026
2027    info!(target: LOG_DEVIMINT, %peer_id, "Modifying config for peer");
2028    let peer_dir = config_dir.join(format!("fedimintd-default-{}", peer_id.to_usize()));
2029
2030    // Read consensus config
2031    let consensus_config_path = peer_dir.join("consensus.json");
2032    let consensus_config_content = fs::read_to_string(&consensus_config_path)
2033        .await
2034        .with_context(|| format!("Failed to read consensus config for peer {peer_id}"))?;
2035
2036    let mut consensus_config: Value = serde_json::from_str(&consensus_config_content)
2037        .with_context(|| format!("Failed to parse consensus config for peer {peer_id}"))?;
2038
2039    // Read the encrypted private config using the server config reader
2040    let password = "pass"; // Default password used in devimint
2041    let server_config = read_server_config(password, &peer_dir)
2042        .with_context(|| format!("Failed to read server config for peer {peer_id}"))?;
2043
2044    // Find existing meta module in configs to use as template
2045    let consensus_config_modules = consensus_config["modules"]
2046        .as_object()
2047        .with_context(|| format!("No modules found in consensus config for peer {peer_id}"))?;
2048
2049    // Look for existing meta module to copy its configuration
2050    let existing_meta_consensus = consensus_config_modules
2051        .values()
2052        .find(|module_config| module_config["kind"].as_str() == Some("meta"));
2053
2054    let existing_meta_consensus = existing_meta_consensus
2055        .with_context(|| {
2056            format!("No existing meta module found in consensus config for peer {peer_id}")
2057        })?
2058        .clone();
2059
2060    // Find existing meta module in private config
2061    let existing_meta_instance_id = server_config
2062        .consensus
2063        .modules
2064        .iter()
2065        .find(|(_, config)| config.kind.as_str() == "meta")
2066        .map(|(id, _)| *id)
2067        .with_context(|| {
2068            format!("No existing meta module found in private config for peer {peer_id}")
2069        })?;
2070
2071    let existing_meta_private = server_config
2072        .private
2073        .modules
2074        .get(&existing_meta_instance_id)
2075        .with_context(|| format!("Failed to get existing meta private config for peer {peer_id}"))?
2076        .clone();
2077
2078    // Find the highest existing module ID for the new module
2079    let last_existing_module_id = consensus_config_modules
2080        .keys()
2081        .filter_map(|id| id.parse::<u32>().ok())
2082        .max()
2083        .unwrap_or(0);
2084
2085    let new_module_id = (last_existing_module_id + 1).to_string();
2086    let new_module_instance_id = ModuleInstanceId::from((last_existing_module_id + 1) as u16);
2087
2088    info!(
2089        "Adding new meta module with id {} for peer {} (copying existing meta module config)",
2090        new_module_id, peer_id
2091    );
2092
2093    // Add new meta module to consensus config by copying existing meta module
2094    if let Some(modules) = consensus_config["modules"].as_object_mut() {
2095        modules.insert(new_module_id.clone(), existing_meta_consensus);
2096    }
2097
2098    // Add new meta module to private config
2099    let mut updated_private_config = server_config.private.clone();
2100    updated_private_config
2101        .modules
2102        .insert(new_module_instance_id, existing_meta_private);
2103
2104    // Write back the modified consensus and client configs
2105    let updated_consensus_content = serde_json::to_string_pretty(&consensus_config)
2106        .with_context(|| format!("Failed to serialize consensus config for peer {peer_id}"))?;
2107
2108    write_overwrite_async(&consensus_config_path, updated_consensus_content)
2109        .await
2110        .with_context(|| format!("Failed to write consensus config for peer {peer_id}"))?;
2111
2112    // Write back the modified private config using direct encryption
2113    let salt = std::fs::read_to_string(peer_dir.join("private.salt"))
2114        .with_context(|| format!("Failed to read salt file for peer {peer_id}"))?;
2115    let key = get_encryption_key(password, &salt)
2116        .with_context(|| format!("Failed to get encryption key for peer {peer_id}"))?;
2117
2118    let private_config_bytes = serde_json::to_string(&updated_private_config)
2119        .with_context(|| format!("Failed to serialize private config for peer {peer_id}"))?
2120        .into_bytes();
2121
2122    // Remove the existing encrypted file first
2123    let encrypted_private_path = peer_dir.join("private.encrypt");
2124    if encrypted_private_path.exists() {
2125        std::fs::remove_file(&encrypted_private_path)
2126            .with_context(|| format!("Failed to remove old private config for peer {peer_id}"))?;
2127    }
2128
2129    encrypted_write(private_config_bytes, &key, encrypted_private_path)
2130        .with_context(|| format!("Failed to write encrypted private config for peer {peer_id}"))?;
2131
2132    info!("Successfully modified configs for peer {}", peer_id);
2133    Ok(())
2134}
