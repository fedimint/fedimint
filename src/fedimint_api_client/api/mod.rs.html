153    /// Make an aggregate request to federation, using `strategy` to logically
154    /// merge the responses.
155    #[instrument(target = LOG_CLIENT_NET_API, skip_all, fields(method=method))]
156    async fn request_with_strategy<PR: DeserializeOwned, FR: Debug>(
157        &self,
158        mut strategy: impl QueryStrategy<PR, FR> + MaybeSend,
159        method: String,
160        params: ApiRequestErased,
161    ) -> FederationResult<FR> {
162        // NOTE: `FuturesUnorderded` is a footgun, but all we do here is polling
163        // completed results from it and we don't do any `await`s when
164        // processing them, it should be totally OK.
165        #[cfg(not(target_family = "wasm"))]
166        let mut futures = FuturesUnordered::<Pin<Box<dyn Future<Output = _> + Send>>>::new();
167        #[cfg(target_family = "wasm")]
168        let mut futures = FuturesUnordered::<Pin<Box<dyn Future<Output = _>>>>::new();
169
170        for peer in self.all_peers() {
171            futures.push(Box::pin({
172                let method = &method;
173                let params = &params;
174                async move {
175                    let result = self
176                        .request_single_peer(method.clone(), params.clone(), *peer)
177                        .await;
178
179                    (*peer, result)
180                }
181            }));
182        }
183
184        let mut peer_errors = BTreeMap::new();
185        let peer_error_threshold = self.all_peers().to_num_peers().one_honest();
186
187        loop {
188            let (peer, result) = futures
189                .next()
190                .await
191                .expect("Query strategy ran out of peers to query without returning a result");
192
193            match result {
194                Ok(response) => match strategy.process(peer, response) {
195                    QueryStep::Retry(peers) => {
196                        for peer in peers {
197                            futures.push(Box::pin({
198                                let method = &method;
199                                let params = &params;
200                                async move {
201                                    let result = self
202                                        .request_single_peer(method.clone(), params.clone(), peer)
203                                        .await;
204
205                                    (peer, result)
206                                }
207                            }));
208                        }
209                    }
210                    QueryStep::Success(response) => return Ok(response),
211                    QueryStep::Failure(e) => {
212                        peer_errors.insert(peer, e);
213                    }
214                    QueryStep::Continue => {}
215                },
216                Err(e) => {
217                    e.report_if_unusual(peer, "RequestWithStrategy");
218                    peer_errors.insert(peer, e);
219                }
220            }
221
222            if peer_errors.len() == peer_error_threshold {
223                return Err(FederationError::peer_errors(
224                    method.clone(),
225                    params.params.clone(),
226                    peer_errors,
227                ));
228            }
229        }
230    }
231
232    #[instrument(target = LOG_CLIENT_NET_API, level = "debug", skip(self, strategy))]
233    async fn request_with_strategy_retry<PR: DeserializeOwned + MaybeSend, FR: Debug>(
234        &self,
235        mut strategy: impl QueryStrategy<PR, FR> + MaybeSend,
236        method: String,
237        params: ApiRequestErased,
238    ) -> FR {
239        // NOTE: `FuturesUnorderded` is a footgun, but all we do here is polling
240        // completed results from it and we don't do any `await`s when
241        // processing them, it should be totally OK.
242        #[cfg(not(target_family = "wasm"))]
243        let mut futures = FuturesUnordered::<Pin<Box<dyn Future<Output = _> + Send>>>::new();
244        #[cfg(target_family = "wasm")]
245        let mut futures = FuturesUnordered::<Pin<Box<dyn Future<Output = _>>>>::new();
246
247        for peer in self.all_peers() {
248            futures.push(Box::pin({
249                let method = &method;
250                let params = &params;
251                async move {
252                    let response = util::retry(
253                        format!("api-request-{method}-{peer}"),
254                        api_networking_backoff(),
255                        || async {
256                            self.request_single_peer(method.clone(), params.clone(), *peer)
257                                .await
258                                .inspect_err(|e| {
259                                    e.report_if_unusual(*peer, "QueryWithStrategyRetry");
260                                })
261                                .map_err(|e| anyhow!(e.to_string()))
262                        },
263                    )
264                    .await
265                    .expect("Number of retries has no limit");
266
267                    (*peer, response)
268                }
269            }));
270        }
271
272        loop {
273            let (peer, response) = match futures.next().await {
274                Some(t) => t,
275                None => pending().await,
276            };
277
278            match strategy.process(peer, response) {
279                QueryStep::Retry(peers) => {
280                    for peer in peers {
281                        futures.push(Box::pin({
282                            let method = &method;
283                            let params = &params;
284                            async move {
285                                let response = util::retry(
286                                    format!("api-request-{method}-{peer}"),
287                                    api_networking_backoff(),
288                                    || async {
289                                        self.request_single_peer(
290                                            method.clone(),
291                                            params.clone(),
292                                            peer,
293                                        )
294                                        .await
295                                        .inspect_err(|err| {
296                                            if err.is_unusual() {
297                                                debug!(target: LOG_CLIENT_NET_API, err = %err.fmt_compact(), "Unusual peer error");
298                                            }
299                                        })
300                                        .map_err(|e| anyhow!(e.to_string()))
301                                    },
302                                )
303                                .await
304                                .expect("Number of retries has no limit");
305
306                                (peer, response)
307                            }
308                        }));
309                    }
310                }
311                QueryStep::Success(response) => return response,
312                QueryStep::Failure(e) => {
313                    warn!(target: LOG_CLIENT_NET_API, "Query strategy returned non-retryable failure for peer {peer}: {e}");
314                }
315                QueryStep::Continue => {}
316            }
317        }
318    }

575pub fn deserialize_outcome<R>(
576    outcome: &SerdeOutputOutcome,
577    module_decoder: &Decoder,
578) -> OutputOutcomeResult<R>
579where
580    R: OutputOutcome + MaybeSend,
581{
582    let dyn_outcome = outcome
583        .try_into_inner_known_module_kind(module_decoder)
584        .map_err(|e| OutputOutcomeError::ResponseDeserialization(e.into()))?;
585
586    let source_instance = dyn_outcome.module_instance_id();
587
588    dyn_outcome.as_any().downcast_ref().cloned().ok_or_else(|| {
589        let target_type = std::any::type_name::<R>();
590        OutputOutcomeError::ResponseDeserialization(anyhow!(
591            "Could not downcast output outcome with instance id {source_instance} to {target_type}"
592        ))
593    })
594}

596/// Federation API client
597///
598/// The core underlying object used to make API requests to a federation.
599///
600/// It has an `connectors` handle to actually making outgoing connections
601/// to given URLs, and knows which peers there are and what URLs to connect to
602/// to reach them.
603// TODO: As it is currently it mixes a bit the role of connecting to "peers" with
604// general purpose outgoing connection. Not a big deal, but might need refactor
605// in the future.
606#[derive(Clone, Debug)]
607pub struct FederationApi {
608    /// Map of known URLs to use to connect to peers
609    peers: BTreeMap<PeerId, SafeUrl>,
610    /// List of peer ids, redundant to avoid collecting all the time
611    peers_keys: BTreeSet<PeerId>,
612    /// Our own [`PeerId`] to use when making admin apis
613    admin_id: Option<PeerId>,
614    /// Set when this API is used to communicate with a module
615    module_id: Option<ModuleInstanceId>,
616    /// Api secret of the federation
617    api_secret: Option<String>,
618    /// Connection pool
619    connection_pool: ConnectionPool<dyn IGuardianConnection>,
620}
621
622impl FederationApi {
623    pub fn new(
624        connectors: ConnectorRegistry,
625        peers: BTreeMap<PeerId, SafeUrl>,
626        admin_peer_id: Option<PeerId>,
627        api_secret: Option<&str>,
628    ) -> Self {
629        Self {
630            peers_keys: peers.keys().copied().collect(),
631            peers,
632            admin_id: admin_peer_id,
633            module_id: None,
634            api_secret: api_secret.map(ToOwned::to_owned),
635            connection_pool: ConnectionPool::new(connectors),
636        }
637    }
638
639    async fn get_or_create_connection(
640        &self,
641        url: &SafeUrl,
642        api_secret: Option<&str>,
643    ) -> ServerResult<DynGuaridianConnection> {
644        self.connection_pool
645            .get_or_create_connection(url, api_secret, |url, api_secret, connectors| async move {
646                let conn = connectors
647                    .connect_guardian(&url, api_secret.as_deref())
648                    .await?;
649                Ok(conn)
650            })
651            .await
652    }
653
654    async fn request(
655        &self,
656        peer: PeerId,
657        method: ApiMethod,
658        request: ApiRequestErased,
659    ) -> ServerResult<Value> {
660        trace!(target: LOG_CLIENT_NET_API, %peer, %method, "Api request");
661        let url = self
662            .peers
663            .get(&peer)
664            .ok_or_else(|| ServerError::InvalidPeerId { peer_id: peer })?;
665        let conn = self
666            .get_or_create_connection(url, self.api_secret.as_deref())
667            .await
668            .context("Failed to connect to peer")
669            .map_err(ServerError::Connection)?;
670
671        let method_str = method.to_string();
672        let peer_str = peer.to_string();
673        let timer = CLIENT_API_REQUEST_DURATION_SECONDS
674            .with_label_values(&[&method_str, &peer_str])
675            .start_timer_ext();
676
677        let res = conn.request(method.clone(), request).await;
678
679        timer.observe_duration();
680
681        let result_label = if res.is_ok() { "success" } else { "error" }.to_string();
682        CLIENT_API_REQUESTS_TOTAL
683            .with_label_values(&[&method_str, &peer_str, &result_label])
684            .inc();
685
686        trace!(target: LOG_CLIENT_NET_API, ?method, res_ok = res.is_ok(), "Api response");
687
688        res
689    }

691    /// Get receiver for changes in the active connections
692    ///
693    /// This allows real-time monitoring of connection status.
694    pub fn get_active_connection_receiver(&self) -> watch::Receiver<BTreeSet<SafeUrl>> {
695        self.connection_pool.get_active_connection_receiver()
696    }
