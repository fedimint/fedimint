28#[derive(Debug, Clone)]
29pub struct WebsocketEndpoint {
30    /// Connection pool for websocket connections
31    #[allow(clippy::type_complexity)]
32    connections:
33        Arc<tokio::sync::Mutex<HashMap<SafeUrl, Arc<tokio::sync::OnceCell<Arc<WsClient>>>>>>,
34}
35
36impl WebsocketEndpoint {
37    pub fn new() -> Self {
38        Self {
39            connections: Arc::new(tokio::sync::Mutex::new(HashMap::new())),
40        }
41    }
42
43    async fn get_or_create_connection(
44        &self,
45        url: &SafeUrl,
46        api_secret: Option<&str>,
47    ) -> PeerResult<Arc<WsClient>> {
48        let mut pool_lock = self.connections.lock().await;
49
50        let entry_arc = pool_lock
51            .entry(url.to_owned())
52            .and_modify(|entry_arc| {
53                // Check if existing connection is disconnected and remove it
54                if let Some(existing_conn) = entry_arc.get()
55                    && !existing_conn.is_connected() {
56                        trace!(target: LOG_NET_WS, %url, "Existing connection is disconnected, removing from pool");
57                        *entry_arc = Arc::new(OnceCell::new());
58                    }
59            })
60            .or_insert_with(|| Arc::new(OnceCell::new()))
61            .clone();
62
63        // Drop the pool lock so other connections can work in parallel
64        drop(pool_lock);
65
66        let conn = entry_arc
67            .get_or_try_init(|| async {
68                trace!(target: LOG_NET_WS, %url, "Creating new websocket connection");
69
70                #[cfg(not(target_family = "wasm"))]
71                let mut client = {
72                    use jsonrpsee_ws_client::{CustomCertStore, WsClientBuilder};
73                    use tokio_rustls::rustls::RootCertStore;
74
75                    install_crypto_provider().await;
76                    let webpki_roots = webpki_roots::TLS_SERVER_ROOTS.iter().cloned();
77                    let mut root_certs = RootCertStore::empty();
78                    root_certs.extend(webpki_roots);
79
80                    let tls_cfg = CustomCertStore::builder()
81                        .with_root_certificates(root_certs)
82                        .with_no_client_auth();
83
84                    WsClientBuilder::default()
85                        .max_concurrent_requests(u16::MAX as usize)
86                        .with_custom_cert_store(tls_cfg)
87                };
88
89                #[cfg(target_family = "wasm")]
90                let client = WsClientBuilder::default().max_concurrent_requests(u16::MAX as usize);
91
92                if let Some(api_secret) = api_secret {
93                    #[cfg(not(target_family = "wasm"))]
94                    {
95                        // on native platforms, jsonrpsee-client ignores `user:pass@...` in the Url,
96                        // but we can set up the headers manually
97
98                        use base64::Engine as _;
99                        use jsonrpsee_ws_client::{HeaderMap, HeaderValue};
100                        let mut headers = HeaderMap::new();
101
102                        let auth = base64::engine::general_purpose::STANDARD
103                            .encode(format!("fedimint:{api_secret}"));
104
105                        headers.insert(
106                            "Authorization",
107                            HeaderValue::from_str(&format!("Basic {auth}")).expect("Can't fail"),
108                        );
109
110                        client = client.set_headers(headers);
111                    }
112                    #[cfg(target_family = "wasm")]
113                    {
114                        // on wasm, url will be handled by the browser, which should take care of
115                        // `user:pass@...`
116                        let mut url = url.clone();
117                        url.set_username("fedimint")
118                            .map_err(|_| PeerError::InvalidEndpoint(anyhow!("invalid username")))?;
119                        url.set_password(Some(&api_secret))
120                            .map_err(|_| PeerError::InvalidEndpoint(anyhow!("invalid secret")))?;
121
122                        let client = client
123                            .build(url.as_str())
124                            .await
125                            .map_err(|err| PeerError::InternalClientError(err.into()))?;
126
127                        return Ok(Arc::new(client));
128                    }
129                }
130
131                let client = client
132                    .build(url.as_str())
133                    .await
134                    .map_err(|err| PeerError::InternalClientError(err.into()))?;
135
136                Ok(Arc::new(client))
137            })
138            .await?;
139
140        trace!(target: LOG_NET_WS, %url, "Using websocket connection");
141        Ok(conn.clone())
142    }
143}

151#[async_trait::async_trait]
152impl Connector for WebsocketEndpoint {
153    async fn connect(
154        &self,
155        url: &SafeUrl,
156        api_secret: Option<&str>,
157    ) -> PeerResult<DynClientConnection> {
158        let client = self.get_or_create_connection(url, api_secret).await?;
159        Ok(client.into_dyn())
160    }

163#[async_trait]
164impl IClientConnection for WsClient {
165    async fn request(&self, method: ApiMethod, request: ApiRequestErased) -> PeerResult<Value> {
166        let method = match method {
167            ApiMethod::Core(method) => method,
168            ApiMethod::Module(module_id, method) => format!("module_{module_id}_{method}"),
169        };
170
171        Ok(ClientT::request(self, &method, [request.to_json()])
172            .await
173            .map_err(jsonrpc_error_to_peer_error)?)
174    }
175
176    async fn await_disconnection(&self) {
177        self.on_disconnect().await;
178    }
179}

201fn jsonrpc_error_to_peer_error(jsonrpc_error: JsonRpcClientError) -> PeerError {
202    match jsonrpc_error {
203        JsonRpcClientError::Call(error_object) => {
204            let error = anyhow!(error_object.message().to_owned());
205            match ErrorCode::from(error_object.code()) {
206                ErrorCode::ParseError | ErrorCode::OversizedRequest | ErrorCode::InvalidRequest => {
207                    PeerError::InvalidRequest(error)
208                }
209                ErrorCode::MethodNotFound => PeerError::InvalidRpcId(error),
210                ErrorCode::InvalidParams => PeerError::InvalidRequest(error),
211                ErrorCode::InternalError | ErrorCode::ServerIsBusy | ErrorCode::ServerError(_) => {
212                    PeerError::ServerError(error)
213                }
214            }
215        }
216        JsonRpcClientError::Transport(error) => PeerError::Transport(anyhow!(error)),
217        JsonRpcClientError::RestartNeeded(arc) => PeerError::Transport(anyhow!(arc)),
218        JsonRpcClientError::ParseError(error) => PeerError::InvalidResponse(anyhow!(error)),
219        JsonRpcClientError::InvalidSubscriptionId => {
220            PeerError::Transport(anyhow!("Invalid subscription id"))
221        }
222        JsonRpcClientError::InvalidRequestId(invalid_request_id) => {
223            PeerError::InvalidRequest(anyhow!(invalid_request_id))
224        }
225        JsonRpcClientError::RequestTimeout => PeerError::Transport(anyhow!("Request timeout")),
226        JsonRpcClientError::Custom(e) => PeerError::Transport(anyhow!(e)),
227        JsonRpcClientError::HttpNotImplemented => {
228            PeerError::ServerError(anyhow!("Http not implemented"))
229        }
230        JsonRpcClientError::EmptyBatchRequest(empty_batch_request) => {
231            PeerError::InvalidRequest(anyhow!(empty_batch_request))
232        }
233        JsonRpcClientError::RegisterMethod(register_method_error) => {
234            PeerError::InvalidResponse(anyhow!(register_method_error))
235        }
236    }
