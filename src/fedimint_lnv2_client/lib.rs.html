121/// The state of an operation sending a payment over lightning.
122///
123/// ```mermaid
124/// graph LR
125/// classDef virtual fill:#fff,stroke-dasharray: 5 5
126///
127///     Funding -- funding transaction is rejected --> Rejected
128///     Funding -- funding transaction is accepted --> Funded
129///     Funded -- payment is confirmed  --> Success
130///     Funded -- payment attempt expires --> Refunding
131///     Funded -- gateway cancels payment attempt --> Refunding
132///     Refunding -- payment is confirmed --> Success
133///     Refunding -- ecash is minted --> Refunded
134///     Refunding -- minting ecash fails --> Failure
135/// ```
136/// The transition from Refunding to Success is only possible if the gateway
137/// misbehaves.
138#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
139pub enum SendOperationState {
140    /// We are funding the contract to incentivize the gateway.
141    Funding,
142    /// We are waiting for the gateway to complete the payment.
143    Funded,
144    /// The payment was successful.
145    Success([u8; 32]),
146    /// The payment has failed and we are refunding the contract.
147    Refunding,
148    /// The payment has been refunded.
149    Refunded,
150    /// Either a programming error has occurred or the federation is malicious.
151    Failure,
152}
153
154/// The final state of an operation sending a payment over lightning.
155#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
156pub enum FinalSendOperationState {
157    /// The payment was successful.
158    Success,
159    /// The payment has been refunded.
160    Refunded,
161    /// Either a programming error has occurred or the federation is malicious.
162    Failure,
163}

332fn generate_ephemeral_tweak(static_pk: PublicKey) -> ([u8; 32], PublicKey) {
333    let keypair = Keypair::new(secp256k1::SECP256K1, &mut rand::thread_rng());
334
335    let tweak = ecdh::SharedSecret::new(&static_pk, &keypair.secret_key());
336
337    (tweak.secret_bytes(), keypair.public_key())
338}

391    async fn update_gateway_map(
392        federation_id: &FederationId,
393        client_ctx: &ClientContext<Self>,
394        module_api: &DynModuleApi,
395        gateway_conn: &Arc<dyn GatewayConnection + Send + Sync>,
396    ) {
397        // Update the mapping from lightning node public keys to gateway api
398        // endpoints maintained in the module database. When paying an invoice this
399        // enables the client to select the gateway that has created the invoice,
400        // if possible, such that the payment does not go over lightning, reducing
401        // fees and latency.
402
403        if let Ok(gateways) = module_api.gateways().await {
404            let mut dbtx = client_ctx.module_db().begin_transaction().await;
405
406            for gateway in gateways {
407                if let Ok(Some(routing_info)) = gateway_conn
408                    .routing_info(gateway.clone(), federation_id)
409                    .await
410                {
411                    dbtx.insert_entry(&GatewayKey(routing_info.lightning_public_key), &gateway)
412                        .await;
413                }
414            }
415
416            if let Err(e) = dbtx.commit_tx_result().await {
417                warn!("Failed to commit the updated gateway mapping to the database: {e}");
418            }
419        }
420    }

422    pub async fn select_gateway(
423        &self,
424        invoice: Option<Bolt11Invoice>,
425    ) -> Result<(SafeUrl, RoutingInfo), SelectGatewayError> {
426        let gateways = self
427            .module_api
428            .gateways()
429            .await
430            .map_err(|e| SelectGatewayError::FederationError(e.to_string()))?;
431
432        if gateways.is_empty() {
433            return Err(SelectGatewayError::NoVettedGateways);
434        }
435
436        if let Some(invoice) = invoice
437            && let Some(gateway) = self
438                .client_ctx
439                .module_db()
440                .begin_transaction_nc()
441                .await
442                .get_value(&GatewayKey(invoice.recover_payee_pub_key()))
443                .await
444                .filter(|gateway| gateways.contains(gateway))
445            && let Ok(Some(routing_info)) = self.routing_info(&gateway).await
446        {
447            return Ok((gateway, routing_info));
448        }
449
450        for gateway in gateways {
451            if let Ok(Some(routing_info)) = self.routing_info(&gateway).await {
452                return Ok((gateway, routing_info));
453            }
454        }
455
456        Err(SelectGatewayError::FailedToFetchRoutingInfo)
457    }

468    /// Pay an invoice. For testing you can optionally specify a gateway to
469    /// route with, otherwise a gateway will be selected automatically. If the
470    /// invoice was created by a gateway connected to our federation, the same
471    /// gateway will be selected to allow for a direct ecash swap. Otherwise we
472    /// select a random online gateway.
473    ///
474    /// The fee for this payment may depend on the selected gateway but
475    /// will be limited to one and a half percent plus one hundred satoshis.
476    /// This fee accounts for the fee charged by the gateway as well as
477    /// the additional fee required to reliably route this payment over
478    /// lightning if necessary. Since the gateway has been vetted by at least
479    /// one guardian we trust it to set a reasonable fee and only enforce a
480    /// rather high limit.
481    ///
482    /// The absolute fee for a payment can be calculated from the operation meta
483    /// to be shown to the user in the transaction history.
484    pub async fn send(
485        &self,
486        invoice: Bolt11Invoice,
487        gateway: Option<SafeUrl>,
488        custom_meta: Value,
489    ) -> Result<OperationId, SendPaymentError> {
490        let amount = invoice
491            .amount_milli_satoshis()
492            .ok_or(SendPaymentError::InvoiceMissingAmount)?;
493
494        if invoice.is_expired() {
495            return Err(SendPaymentError::InvoiceExpired);
496        }
497
498        if self.cfg.network != invoice.currency().into() {
499            return Err(SendPaymentError::WrongCurrency {
500                invoice_currency: invoice.currency(),
501                federation_currency: self.cfg.network.into(),
502            });
503        }
504
505        let operation_id = self.get_next_operation_id(&invoice).await?;
506
507        let (ephemeral_tweak, ephemeral_pk) = generate_ephemeral_tweak(self.keypair.public_key());
508
509        let refund_keypair = SecretKey::from_slice(&ephemeral_tweak)
510            .expect("32 bytes, within curve order")
511            .keypair(secp256k1::SECP256K1);
512
513        let (gateway_api, routing_info) = match gateway {
514            Some(gateway_api) => (
515                gateway_api.clone(),
516                self.routing_info(&gateway_api)
517                    .await
518                    .map_err(SendPaymentError::GatewayConnectionError)?
519                    .ok_or(SendPaymentError::UnknownFederation)?,
520            ),
521            None => self
522                .select_gateway(Some(invoice.clone()))
523                .await
524                .map_err(SendPaymentError::FailedToSelectGateway)?,
525        };
526
527        let (send_fee, expiration_delta) = routing_info.send_parameters(&invoice);
528
529        if !send_fee.le(&PaymentFee::SEND_FEE_LIMIT) {
530            return Err(SendPaymentError::PaymentFeeExceedsLimit);
531        }
532
533        if EXPIRATION_DELTA_LIMIT < expiration_delta {
534            return Err(SendPaymentError::ExpirationDeltaExceedsLimit);
535        }
536
537        let consensus_block_count = self
538            .module_api
539            .consensus_block_count()
540            .await
541            .map_err(|e| SendPaymentError::FederationError(e.to_string()))?;
542
543        let contract = OutgoingContract {
544            payment_image: PaymentImage::Hash(*invoice.payment_hash()),
545            amount: send_fee.add_to(amount),
546            expiration: consensus_block_count + expiration_delta + CONTRACT_CONFIRMATION_BUFFER,
547            claim_pk: routing_info.module_public_key,
548            refund_pk: refund_keypair.public_key(),
549            ephemeral_pk,
550        };
551
552        let contract_clone = contract.clone();
553        let gateway_api_clone = gateway_api.clone();
554        let invoice_clone = invoice.clone();
555
556        let client_output = ClientOutput::<LightningOutput> {
557            output: LightningOutput::V0(LightningOutputV0::Outgoing(contract.clone())),
558            amount: contract.amount,
559        };
560        let client_output_sm = ClientOutputSM::<LightningClientStateMachines> {
561            state_machines: Arc::new(move |range: OutPointRange| {
562                vec![LightningClientStateMachines::Send(SendStateMachine {
563                    common: SendSMCommon {
564                        operation_id,
565                        outpoint: range.into_iter().next().unwrap(),
566                        contract: contract_clone.clone(),
567                        gateway_api: Some(gateway_api_clone.clone()),
568                        invoice: Some(LightningInvoice::Bolt11(invoice_clone.clone())),
569                        refund_keypair,
570                    },
571                    state: SendSMState::Funding,
572                })]
573            }),
574        };
575
576        let client_output = self.client_ctx.make_client_outputs(ClientOutputBundle::new(
577            vec![client_output],
578            vec![client_output_sm],
579        ));
580        let transaction = TransactionBuilder::new().with_outputs(client_output);
581
582        self.client_ctx
583            .finalize_and_submit_transaction(
584                operation_id,
585                LightningCommonInit::KIND.as_str(),
586                move |change_outpoint_range| {
587                    LightningOperationMeta::Send(SendOperationMeta {
588                        change_outpoint_range,
589                        gateway: gateway_api.clone(),
590                        contract: contract.clone(),
591                        invoice: LightningInvoice::Bolt11(invoice.clone()),
592                        custom_meta: custom_meta.clone(),
593                    })
594                },
595                transaction,
596            )
597            .await
598            .map_err(|e| SendPaymentError::FinalizationError(e.to_string()))?;
599
600        Ok(operation_id)
601    }

603    async fn get_next_operation_id(
604        &self,
605        invoice: &Bolt11Invoice,
606    ) -> Result<OperationId, SendPaymentError> {
607        for payment_attempt in 0..u64::MAX {
608            let operation_id = OperationId::from_encodable(&(invoice.clone(), payment_attempt));
609
610            if !self.client_ctx.operation_exists(operation_id).await {
611                return Ok(operation_id);
612            }
613
614            if self.client_ctx.has_active_states(operation_id).await {
615                return Err(SendPaymentError::PendingPreviousPayment(operation_id));
616            }
617
618            let mut stream = self
619                .subscribe_send_operation_state_updates(operation_id)
620                .await
621                .expect("operation_id exists")
622                .into_stream();
623
624            // This will not block since we checked for active states and there were none,
625            // so by definition a final state has to have been assumed already.
626            while let Some(state) = stream.next().await {
627                if let SendOperationState::Success(_) = state {
628                    return Err(SendPaymentError::SuccessfulPreviousPayment(operation_id));
629                }
630            }
631        }
632
633        panic!("We could not find an unused operation id for sending a lightning payment");
634    }

636    /// Subscribe to all state updates of the send operation.
637    pub async fn subscribe_send_operation_state_updates(
638        &self,
639        operation_id: OperationId,
640    ) -> anyhow::Result<UpdateStreamOrOutcome<SendOperationState>> {
641        let operation = self.client_ctx.get_operation(operation_id).await?;
642        let mut stream = self.notifier.subscribe(operation_id).await;
643        let client_ctx = self.client_ctx.clone();
644        let module_api = self.module_api.clone();
645
646        Ok(self.client_ctx.outcome_or_updates(operation, operation_id, move || {
647            stream! {
648                loop {
649                    if let Some(LightningClientStateMachines::Send(state)) = stream.next().await {
650                        match state.state {
651                            SendSMState::Funding => yield SendOperationState::Funding,
652                            SendSMState::Funded => yield SendOperationState::Funded,
653                            SendSMState::Success(preimage) => {
654                                // the preimage has been verified by the state machine previously
655                                assert!(state.common.contract.verify_preimage(&preimage));
656
657                                yield SendOperationState::Success(preimage);
658                                return;
659                            },
660                            SendSMState::Refunding(out_points) => {
661                                yield SendOperationState::Refunding;
662
663                                if client_ctx.await_primary_module_outputs(operation_id, out_points.clone()).await.is_ok() {
664                                    yield SendOperationState::Refunded;
665                                    return;
666                                }
667
668                                // The gateway may have incorrectly claimed the outgoing contract thereby causing
669                                // our refund transaction to be rejected. Therefore, we check one last time if
670                                // the preimage is available before we enter the failure state.
671                                if let Some(preimage) = module_api.await_preimage(
672                                    state.common.outpoint,
673                                    0
674                                ).await
675                                    && state.common.contract.verify_preimage(&preimage) {
676                                        yield SendOperationState::Success(preimage);
677                                        return;
678                                    }
679
680                                yield SendOperationState::Failure;
681                                return;
682                            },
683                            SendSMState::Rejected(..) => {
684                                yield SendOperationState::Failure;
685                                return;
686                            },
687                        }
688                    }
689                }
690            }
691        }))
692    }

718    /// Request an invoice. For testing you can optionally specify a gateway to
719    /// generate the invoice, otherwise a random online gateway will be selected
720    /// automatically.
721    ///
722    /// The total fee for this payment may depend on the chosen gateway but
723    /// will be limited to half of one percent plus fifty satoshis. Since the
724    /// selected gateway has been vetted by at least one guardian we trust it to
725    /// set a reasonable fee and only enforce a rather high limit.
726    ///
727    /// The absolute fee for a payment can be calculated from the operation meta
728    /// to be shown to the user in the transaction history.
729    pub async fn receive(
730        &self,
731        amount: Amount,
732        expiry_secs: u32,
733        description: Bolt11InvoiceDescription,
734        gateway: Option<SafeUrl>,
735        custom_meta: Value,
736    ) -> Result<(Bolt11Invoice, OperationId), ReceiveError> {
737        let (gateway, contract, invoice) = self
738            .create_contract_and_fetch_invoice(
739                self.keypair.public_key(),
740                amount,
741                expiry_secs,
742                description,
743                gateway,
744            )
745            .await?;
746
747        let operation_id = self
748            .receive_incoming_contract(gateway, contract, invoice.clone(), custom_meta)
749            .await
750            .expect("The contract has been generated with our public key");
751
752        Ok((invoice, operation_id))
753    }

755    /// Create an incoming contract locked to a public key derived from the
756    /// recipient's static module public key and fetches the corresponding
757    /// invoice.
758    async fn create_contract_and_fetch_invoice(
759        &self,
760        recipient_static_pk: PublicKey,
761        amount: Amount,
762        expiry_secs: u32,
763        description: Bolt11InvoiceDescription,
764        gateway: Option<SafeUrl>,
765    ) -> Result<(SafeUrl, IncomingContract, Bolt11Invoice), ReceiveError> {
766        let (ephemeral_tweak, ephemeral_pk) = generate_ephemeral_tweak(recipient_static_pk);
767
768        let encryption_seed = ephemeral_tweak
769            .consensus_hash::<sha256::Hash>()
770            .to_byte_array();
771
772        let preimage = encryption_seed
773            .consensus_hash::<sha256::Hash>()
774            .to_byte_array();
775
776        let (gateway, routing_info) = match gateway {
777            Some(gateway) => (
778                gateway.clone(),
779                self.routing_info(&gateway)
780                    .await
781                    .map_err(ReceiveError::GatewayConnectionError)?
782                    .ok_or(ReceiveError::UnknownFederation)?,
783            ),
784            None => self
785                .select_gateway(None)
786                .await
787                .map_err(ReceiveError::FailedToSelectGateway)?,
788        };
789
790        if !routing_info.receive_fee.le(&PaymentFee::RECEIVE_FEE_LIMIT) {
791            return Err(ReceiveError::PaymentFeeExceedsLimit);
792        }
793
794        let contract_amount = routing_info.receive_fee.subtract_from(amount.msats);
795
796        // The dust limit ensures that the incoming contract can be claimed without
797        // additional funds as the contracts amount is sufficient to cover the fees
798        if contract_amount < Amount::from_sats(50) {
799            return Err(ReceiveError::DustAmount);
800        }
801
802        let expiration = duration_since_epoch()
803            .as_secs()
804            .saturating_add(u64::from(expiry_secs));
805
806        let claim_pk = recipient_static_pk
807            .mul_tweak(
808                secp256k1::SECP256K1,
809                &Scalar::from_be_bytes(ephemeral_tweak).expect("Within curve order"),
810            )
811            .expect("Tweak is valid");
812
813        let contract = IncomingContract::new(
814            self.cfg.tpe_agg_pk,
815            encryption_seed,
816            preimage,
817            PaymentImage::Hash(preimage.consensus_hash()),
818            contract_amount,
819            expiration,
820            claim_pk,
821            routing_info.module_public_key,
822            ephemeral_pk,
823        );
824
825        let invoice = self
826            .gateway_conn
827            .bolt11_invoice(
828                gateway.clone(),
829                self.federation_id,
830                contract.clone(),
831                amount,
832                description,
833                expiry_secs,
834            )
835            .await
836            .map_err(ReceiveError::GatewayConnectionError)?;
837
838        if invoice.payment_hash() != &preimage.consensus_hash() {
839            return Err(ReceiveError::InvalidInvoicePaymentHash);
840        }
841
842        if invoice.amount_milli_satoshis() != Some(amount.msats) {
843            return Err(ReceiveError::InvalidInvoiceAmount);
844        }
845
846        Ok((gateway, contract, invoice))
847    }

849    // Receive an incoming contract locked to a public key derived from our
850    // static module public key.
851    async fn receive_incoming_contract(
852        &self,
853        gateway: SafeUrl,
854        contract: IncomingContract,
855        invoice: Bolt11Invoice,
856        custom_meta: Value,
857    ) -> Option<OperationId> {
858        let operation_id = OperationId::from_encodable(&contract.clone());
859
860        let (claim_keypair, agg_decryption_key) = self.recover_contract_keys(&contract)?;
861
862        let receive_sm = LightningClientStateMachines::Receive(ReceiveStateMachine {
863            common: ReceiveSMCommon {
864                operation_id,
865                contract: contract.clone(),
866                claim_keypair,
867                agg_decryption_key,
868            },
869            state: ReceiveSMState::Pending,
870        });
871
872        // this may only fail if the operation id is already in use, in which case we
873        // ignore the error such that the method is idempotent
874        self.client_ctx
875            .manual_operation_start(
876                operation_id,
877                LightningCommonInit::KIND.as_str(),
878                LightningOperationMeta::Receive(ReceiveOperationMeta {
879                    gateway,
880                    contract,
881                    invoice: LightningInvoice::Bolt11(invoice),
882                    custom_meta,
883                }),
884                vec![self.client_ctx.make_dyn_state(receive_sm)],
885            )
886            .await
887            .ok();
888
889        Some(operation_id)
890    }
891
892    fn recover_contract_keys(
893        &self,
894        contract: &IncomingContract,
895    ) -> Option<(Keypair, AggregateDecryptionKey)> {
896        let ephemeral_tweak = ecdh::SharedSecret::new(
897            &contract.commitment.ephemeral_pk,
898            &self.keypair.secret_key(),
899        )
900        .secret_bytes();
901
902        let encryption_seed = ephemeral_tweak
903            .consensus_hash::<sha256::Hash>()
904            .to_byte_array();
905
906        let claim_keypair = self
907            .keypair
908            .secret_key()
909            .mul_tweak(&Scalar::from_be_bytes(ephemeral_tweak).expect("Within curve order"))
910            .expect("Tweak is valid")
911            .keypair(secp256k1::SECP256K1);
912
913        if claim_keypair.public_key() != contract.commitment.claim_pk {
914            return None; // The claim key is not derived from our pk
915        }
916
917        let agg_decryption_key = derive_agg_dk(&self.cfg.tpe_agg_pk, &encryption_seed);
918
919        if !contract.verify_agg_decryption_key(&self.cfg.tpe_agg_pk, &agg_decryption_key) {
920            return None; // The decryption key is not derived from our pk
921        }
922
923        contract.decrypt_preimage(&agg_decryption_key)?;
924
925        Some((claim_keypair, agg_decryption_key))
926    }

928    /// Subscribe to all state updates of the receive operation.
929    pub async fn subscribe_receive_operation_state_updates(
930        &self,
931        operation_id: OperationId,
932    ) -> anyhow::Result<UpdateStreamOrOutcome<ReceiveOperationState>> {
933        let operation = self.client_ctx.get_operation(operation_id).await?;
934        let mut stream = self.notifier.subscribe(operation_id).await;
935        let client_ctx = self.client_ctx.clone();
936
937        Ok(self.client_ctx.outcome_or_updates(operation, operation_id, move || {
938            stream! {
939                loop {
940                    if let Some(LightningClientStateMachines::Receive(state)) = stream.next().await {
941                        match state.state {
942                            ReceiveSMState::Pending => yield ReceiveOperationState::Pending,
943                            ReceiveSMState::Claiming(out_points) => {
944                                yield ReceiveOperationState::Claiming;
945
946                                if client_ctx.await_primary_module_outputs(operation_id, out_points).await.is_ok() {
947                                    yield ReceiveOperationState::Claimed;
948                                } else {
949                                    yield ReceiveOperationState::Failure;
950                                }
951                                return;
952                            },
953                            ReceiveSMState::Expired => {
954                                yield ReceiveOperationState::Expired;
955                                return;
956                            }
957                        }
958                    }
959                }
960            }
961        }))
962    }

965    pub async fn await_final_receive_operation_state(
966        &self,
967        operation_id: OperationId,
968    ) -> anyhow::Result<FinalReceiveOperationState> {

977                    ReceiveOperationState::Failure => Some(FinalReceiveOperationState::Failure),
978                    _ => None,
979                })
980            })
981            .next()
982            .await
983            .expect("Stream contains one final state");
984
985        Ok(state)
986    }
