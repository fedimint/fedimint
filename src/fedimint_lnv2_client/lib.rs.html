122/// The state of an operation sending a payment over lightning.
123///
124/// ```mermaid
125/// graph LR
126/// classDef virtual fill:#fff,stroke-dasharray: 5 5
127///
128///     Funding -- funding transaction is rejected --> Rejected
129///     Funding -- funding transaction is accepted --> Funded
130///     Funded -- payment is confirmed  --> Success
131///     Funded -- payment attempt expires --> Refunding
132///     Funded -- gateway cancels payment attempt --> Refunding
133///     Refunding -- payment is confirmed --> Success
134///     Refunding -- ecash is minted --> Refunded
135///     Refunding -- minting ecash fails --> Failure
136/// ```
137/// The transition from Refunding to Success is only possible if the gateway
138/// misbehaves.
139#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
140pub enum SendOperationState {
141    /// We are funding the contract to incentivize the gateway.
142    Funding,
143    /// We are waiting for the gateway to complete the payment.
144    Funded,
145    /// The payment was successful.
146    Success,
147    /// The payment has failed and we are refunding the contract.
148    Refunding,
149    /// The payment has been refunded.
150    Refunded,
151    /// Either a programming error has occurred or the federation is malicious.
152    Failure,
153}
154
155/// The final state of an operation sending a payment over lightning.
156#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
157pub enum FinalSendOperationState {
158    /// The payment was successful.
159    Success,
160    /// The payment has been refunded.
161    Refunded,
162    /// Either a programming error has occurred or the federation is malicious.
163    Failure,
164}
165
166pub type SendResult = Result<OperationId, SendPaymentError>;

171/// ```mermaid
172/// graph LR

174///
175///     Pending -- payment is confirmed --> Claiming
176///     Pending -- invoice expires --> Expired
177///     Claiming -- ecash is minted --> Claimed
178///     Claiming -- minting ecash fails --> Failure

181pub enum ReceiveOperationState {
182    /// We are waiting for the payment.
183    Pending,
184    /// The payment request has expired.
185    Expired,
186    /// The payment has been confirmed and we are issuing the ecash.
187    Claiming,
188    /// The payment has been successful.
189    Claimed,
190    /// Either a programming error has occurred or the federation is malicious.
191    Failure,
192}

333fn generate_ephemeral_tweak(static_pk: PublicKey) -> ([u8; 32], PublicKey) {
334    let keypair = Keypair::new(secp256k1::SECP256K1, &mut rand::thread_rng());
335
336    let tweak = ecdh::SharedSecret::new(&static_pk, &keypair.secret_key());
337
338    (tweak.secret_bytes(), keypair.public_key())
339}

382            let mut interval = tokio::time::interval(Duration::from_secs(24 * 60 * 60));

392    async fn update_gateway_map(
393        federation_id: &FederationId,
394        client_ctx: &ClientContext<Self>,
395        module_api: &DynModuleApi,
396        gateway_conn: &Arc<dyn GatewayConnection + Send + Sync>,
397    ) {
398        // Update the mapping from lightning node public keys to gateway api
399        // endpoints maintained in the module database. When paying an invoice this
400        // enables the client to select the gateway that has created the invoice,
401        // if possible, such that the payment does not go over lightning, reducing
402        // fees and latency.
403
404        if let Ok(gateways) = module_api.gateways().await {
405            let mut dbtx = client_ctx.module_db().begin_transaction().await;
406
407            for gateway in gateways {
408                if let Ok(Some(routing_info)) = gateway_conn
409                    .routing_info(gateway.clone(), federation_id)
410                    .await
411                {
412                    dbtx.insert_entry(&GatewayKey(routing_info.lightning_public_key), &gateway)
413                        .await;
414                }
415            }
416
417            if let Err(e) = dbtx.commit_tx_result().await {
418                warn!("Failed to commit the updated gateway mapping to the database: {e}");
419            }
420        }
421    }

423    async fn select_gateway(
424        &self,
425        invoice: Option<Bolt11Invoice>,
426    ) -> Result<(SafeUrl, RoutingInfo), SelectGatewayError> {
427        let gateways = self
428            .module_api
429            .gateways()
430            .await
431            .map_err(|e| SelectGatewayError::FederationError(e.to_string()))?;
432
433        if gateways.is_empty() {
434            return Err(SelectGatewayError::NoVettedGateways);
435        }
436
437        if let Some(invoice) = invoice {
438            if let Some(gateway) = self
439                .client_ctx
440                .module_db()
441                .begin_transaction_nc()
442                .await
443                .get_value(&GatewayKey(invoice.recover_payee_pub_key()))
444                .await
445                .filter(|gateway| gateways.contains(gateway))
446            {
447                if let Ok(Some(routing_info)) = self.routing_info(&gateway).await {
448                    return Ok((gateway, routing_info));
449                }
450            }
451        }
452
453        for gateway in gateways {
454            if let Ok(Some(routing_info)) = self.routing_info(&gateway).await {
455                return Ok((gateway, routing_info));
456            }
457        }
458
459        Err(SelectGatewayError::FailedToFetchRoutingInfo)
460    }

462    async fn routing_info(

464        gateway: &SafeUrl,
465    ) -> Result<Option<RoutingInfo>, GatewayConnectionError> {

470
471    /// Pay an invoice. For testing you can optionally specify a gateway to
472    /// route with, otherwise a gateway will be selected automatically. If the
473    /// invoice was created by a gateway connected to our federation, the same
474    /// gateway will be selected to allow for a direct ecash swap. Otherwise we
475    /// select a random online gateway.
476    ///
477    /// The fee for this payment may depend on the selected gateway but
478    /// will be limited to one and a half percent plus one hundred satoshis.
479    /// This fee accounts for the fee charged by the gateway as well as
480    /// the additional fee required to reliably route this payment over
481    /// lightning if necessary. Since the gateway has been vetted by at least
482    /// one guardian we trust it to set a reasonable fee and only enforce a
483    /// rather high limit.
484    ///
485    /// The absolute fee for a payment can be calculated from the operation meta
486    /// to be shown to the user in the transaction history.
487    pub async fn send(
488        &self,
489        invoice: Bolt11Invoice,
490        gateway: Option<SafeUrl>,
491        custom_meta: Value,
492    ) -> Result<OperationId, SendPaymentError> {
493        let amount = invoice
494            .amount_milli_satoshis()
495            .ok_or(SendPaymentError::InvoiceMissingAmount)?;
496
497        if invoice.is_expired() {
498            return Err(SendPaymentError::InvoiceExpired);
499        }
500
501        if self.cfg.network != invoice.currency().into() {
502            return Err(SendPaymentError::WrongCurrency {
503                invoice_currency: invoice.currency(),
504                federation_currency: self.cfg.network.into(),
505            });
506        }
507
508        let operation_id = self.get_next_operation_id(&invoice).await?;
509
510        let (ephemeral_tweak, ephemeral_pk) = generate_ephemeral_tweak(self.keypair.public_key());
511
512        let refund_keypair = SecretKey::from_slice(&ephemeral_tweak)
513            .expect("32 bytes, within curve order")
514            .keypair(secp256k1::SECP256K1);
515
516        let (gateway_api, routing_info) = match gateway {
517            Some(gateway_api) => (
518                gateway_api.clone(),
519                self.routing_info(&gateway_api)
520                    .await
521                    .map_err(SendPaymentError::GatewayConnectionError)?
522                    .ok_or(SendPaymentError::UnknownFederation)?,
523            ),
524            None => self
525                .select_gateway(Some(invoice.clone()))
526                .await
527                .map_err(SendPaymentError::FailedToSelectGateway)?,
528        };
529
530        let (send_fee, expiration_delta) = routing_info.send_parameters(&invoice);
531
532        if !send_fee.le(&PaymentFee::SEND_FEE_LIMIT) {
533            return Err(SendPaymentError::PaymentFeeExceedsLimit);
534        }
535
536        if EXPIRATION_DELTA_LIMIT < expiration_delta {
537            return Err(SendPaymentError::ExpirationDeltaExceedsLimit);
538        }
539
540        let consensus_block_count = self
541            .module_api
542            .consensus_block_count()
543            .await
544            .map_err(|e| SendPaymentError::FederationError(e.to_string()))?;
545
546        let contract = OutgoingContract {
547            payment_image: PaymentImage::Hash(*invoice.payment_hash()),
548            amount: send_fee.add_to(amount),
549            expiration: consensus_block_count + expiration_delta + CONTRACT_CONFIRMATION_BUFFER,
550            claim_pk: routing_info.module_public_key,
551            refund_pk: refund_keypair.public_key(),
552            ephemeral_pk,
553        };
554
555        let contract_clone = contract.clone();
556        let gateway_api_clone = gateway_api.clone();
557        let invoice_clone = invoice.clone();
558
559        let client_output = ClientOutput::<LightningOutput> {
560            output: LightningOutput::V0(LightningOutputV0::Outgoing(contract.clone())),
561            amount: contract.amount,
562        };
563        let client_output_sm = ClientOutputSM::<LightningClientStateMachines> {
564            state_machines: Arc::new(move |out_point_range: OutPointRange| {
565                vec![LightningClientStateMachines::Send(SendStateMachine {
566                    common: SendSMCommon {
567                        operation_id,
568                        funding_txid: out_point_range.txid(),
569                        gateway_api: gateway_api_clone.clone(),
570                        contract: contract_clone.clone(),
571                        invoice: LightningInvoice::Bolt11(invoice_clone.clone()),
572                        refund_keypair,
573                    },
574                    state: SendSMState::Funding,
575                })]
576            }),
577        };
578
579        let client_output = self.client_ctx.make_client_outputs(ClientOutputBundle::new(
580            vec![client_output],
581            vec![client_output_sm],
582        ));
583        let transaction = TransactionBuilder::new().with_outputs(client_output);
584
585        self.client_ctx
586            .finalize_and_submit_transaction(
587                operation_id,
588                LightningCommonInit::KIND.as_str(),
589                move |change_range| {
590                    LightningOperationMeta::Send(SendOperationMeta {
591                        funding_txid: change_range.txid(),
592                        funding_change_outpoints: change_range.into_iter().collect(),
593                        gateway: gateway_api.clone(),
594                        contract: contract.clone(),
595                        invoice: LightningInvoice::Bolt11(invoice.clone()),
596                        custom_meta: custom_meta.clone(),
597                    })
598                },
599                transaction,
600            )
601            .await
602            .map_err(|e| SendPaymentError::FinalizationError(e.to_string()))?;
603
604        Ok(operation_id)
605    }

607    async fn get_next_operation_id(
608        &self,
609        invoice: &Bolt11Invoice,
610    ) -> Result<OperationId, SendPaymentError> {
611        for payment_attempt in 0..u64::MAX {
612            let operation_id = OperationId::from_encodable(&(invoice.clone(), payment_attempt));
613
614            if !self.client_ctx.operation_exists(operation_id).await {
615                return Ok(operation_id);
616            }
617
618            if self.client_ctx.has_active_states(operation_id).await {
619                return Err(SendPaymentError::PendingPreviousPayment(operation_id));
620            }
621
622            let mut stream = self
623                .subscribe_send_operation_state_updates(operation_id)
624                .await
625                .expect("operation_id exists")
626                .into_stream();
627
628            // This will not block since we checked for active states and there were none,
629            // so by definition a final state has to have been assumed already.
630            while let Some(state) = stream.next().await {
631                if let SendOperationState::Success = state {
632                    return Err(SendPaymentError::SuccessfulPreviousPayment(operation_id));
633                }
634            }
635        }
636
637        panic!("We could not find an unused operation id for sending a lightning payment");
638    }

640    /// Subscribe to all state updates of the send operation.
641    pub async fn subscribe_send_operation_state_updates(
642        &self,
643        operation_id: OperationId,
644    ) -> anyhow::Result<UpdateStreamOrOutcome<SendOperationState>> {
645        let operation = self.client_ctx.get_operation(operation_id).await?;
646        let mut stream = self.notifier.subscribe(operation_id).await;
647        let client_ctx = self.client_ctx.clone();
648        let module_api = self.module_api.clone();
649
650        Ok(self.client_ctx.outcome_or_updates(operation, operation_id, move || {
651            stream! {
652                loop {
653                    if let Some(LightningClientStateMachines::Send(state)) = stream.next().await {
654                        match state.state {
655                            SendSMState::Funding => yield SendOperationState::Funding,
656                            SendSMState::Funded => yield SendOperationState::Funded,
657                            SendSMState::Success(preimage) => {
658                                // the preimage has been verified by the state machine previously
659                                assert!(state.common.contract.verify_preimage(&preimage));
660
661                                yield SendOperationState::Success;
662                                return;
663                            },
664                            SendSMState::Refunding(out_points) => {
665                                yield SendOperationState::Refunding;
666
667                                if client_ctx.await_primary_module_outputs(operation_id, out_points.clone()).await.is_ok() {
668                                    yield SendOperationState::Refunded;
669                                    return;
670                                }
671
672                                // The gateway may have incorrectly claimed the outgoing contract thereby causing
673                                // our refund transaction to be rejected. Therefore, we check one last time if
674                                // the preimage is available before we enter the failure state.
675                                if let Some(preimage) = module_api.await_preimage(
676                                    &state.common.contract.contract_id(),
677                                    0
678                                ).await {
679                                    if state.common.contract.verify_preimage(&preimage) {
680                                        yield SendOperationState::Success;
681                                        return;
682                                    }
683                                }
684
685                                yield SendOperationState::Failure;
686                                return;
687                            },
688                            SendSMState::Rejected(..) => {
689                                yield SendOperationState::Failure;
690                                return;
691                            },
692                        }
693                    }
694                }
695            }
696        }))
697    }

723    /// Request an invoice. For testing you can optionally specify a gateway to
724    /// generate the invoice, otherwise a random online gateway will be selected
725    /// automatically.
726    ///
727    /// The total fee for this payment may depend on the chosen gateway but
728    /// will be limited to half of one percent plus fifty satoshis. Since the
729    /// selected gateway has been vetted by at least one guardian we trust it to
730    /// set a reasonable fee and only enforce a rather high limit.
731    ///
732    /// The absolute fee for a payment can be calculated from the operation meta
733    /// to be shown to the user in the transaction history.
734    pub async fn receive(
735        &self,
736        amount: Amount,
737        expiry_secs: u32,
738        description: Bolt11InvoiceDescription,
739        gateway: Option<SafeUrl>,
740        custom_meta: Value,
741    ) -> Result<(Bolt11Invoice, OperationId), ReceiveError> {
742        let (gateway, contract, invoice) = self
743            .create_contract_and_fetch_invoice(
744                self.keypair.public_key(),
745                amount,
746                expiry_secs,
747                description,
748                gateway,
749            )
750            .await?;
751
752        let operation_id = self
753            .receive_incoming_contract(gateway, contract, invoice.clone(), custom_meta)
754            .await
755            .expect("The contract has been generated with our public key");
756
757        Ok((invoice, operation_id))
758    }

760    /// Create an incoming contract locked to a public key derived from the
761    /// recipient's static module public key and fetches the corresponding
762    /// invoice.
763    async fn create_contract_and_fetch_invoice(
764        &self,
765        recipient_static_pk: PublicKey,
766        amount: Amount,
767        expiry_secs: u32,
768        description: Bolt11InvoiceDescription,
769        gateway: Option<SafeUrl>,
770    ) -> Result<(SafeUrl, IncomingContract, Bolt11Invoice), ReceiveError> {
771        let (ephemeral_tweak, ephemeral_pk) = generate_ephemeral_tweak(recipient_static_pk);
772
773        let encryption_seed = ephemeral_tweak
774            .consensus_hash::<sha256::Hash>()
775            .to_byte_array();
776
777        let preimage = encryption_seed
778            .consensus_hash::<sha256::Hash>()
779            .to_byte_array();
780
781        let (gateway, routing_info) = match gateway {
782            Some(gateway) => (
783                gateway.clone(),
784                self.routing_info(&gateway)
785                    .await
786                    .map_err(ReceiveError::GatewayConnectionError)?
787                    .ok_or(ReceiveError::UnknownFederation)?,
788            ),
789            None => self
790                .select_gateway(None)
791                .await
792                .map_err(ReceiveError::FailedToSelectGateway)?,
793        };
794
795        if !routing_info.receive_fee.le(&PaymentFee::RECEIVE_FEE_LIMIT) {
796            return Err(ReceiveError::PaymentFeeExceedsLimit);
797        }
798
799        let contract_amount = routing_info.receive_fee.subtract_from(amount.msats);
800
801        // The dust limit ensures that the incoming contract can be claimed without
802        // additional funds as the contracts amount is sufficient to cover the fees
803        if contract_amount < Amount::from_sats(50) {
804            return Err(ReceiveError::DustAmount);
805        }
806
807        let expiration = duration_since_epoch()
808            .as_secs()
809            .saturating_add(u64::from(expiry_secs));
810
811        let claim_pk = recipient_static_pk
812            .mul_tweak(
813                secp256k1::SECP256K1,
814                &Scalar::from_be_bytes(ephemeral_tweak).expect("Within curve order"),
815            )
816            .expect("Tweak is valid");
817
818        let contract = IncomingContract::new(
819            self.cfg.tpe_agg_pk,
820            encryption_seed,
821            preimage,
822            PaymentImage::Hash(preimage.consensus_hash()),
823            contract_amount,
824            expiration,
825            claim_pk,
826            routing_info.module_public_key,
827            ephemeral_pk,
828        );
829
830        let invoice = self
831            .gateway_conn
832            .bolt11_invoice(
833                gateway.clone(),
834                self.federation_id,
835                contract.clone(),
836                amount,
837                description,
838                expiry_secs,
839            )
840            .await
841            .map_err(ReceiveError::GatewayConnectionError)?;
842
843        if invoice.payment_hash() != &preimage.consensus_hash() {
844            return Err(ReceiveError::InvalidInvoicePaymentHash);
845        }
846
847        if invoice.amount_milli_satoshis() != Some(amount.msats) {
848            return Err(ReceiveError::InvalidInvoiceAmount);
849        }
850
851        Ok((gateway, contract, invoice))
852    }

854    // Receive an incoming contract locked to a public key derived from our
855    // static module public key.
856    async fn receive_incoming_contract(
857        &self,
858        gateway: SafeUrl,
859        contract: IncomingContract,
860        invoice: Bolt11Invoice,
861        custom_meta: Value,
862    ) -> Option<OperationId> {
863        let operation_id = OperationId::from_encodable(&contract.clone());
864
865        let (claim_keypair, agg_decryption_key) = self.recover_contract_keys(&contract)?;
866
867        let receive_sm = LightningClientStateMachines::Receive(ReceiveStateMachine {
868            common: ReceiveSMCommon {
869                operation_id,
870                contract: contract.clone(),
871                claim_keypair,
872                agg_decryption_key,
873            },
874            state: ReceiveSMState::Pending,
875        });
876
877        // this may only fail if the operation id is already in use, in which case we
878        // ignore the error such that the method is idempotent
879        self.client_ctx
880            .manual_operation_start(
881                operation_id,
882                LightningCommonInit::KIND.as_str(),
883                LightningOperationMeta::Receive(ReceiveOperationMeta {
884                    gateway,
885                    contract,
886                    invoice: LightningInvoice::Bolt11(invoice),
887                    custom_meta,
888                }),
889                vec![self.client_ctx.make_dyn_state(receive_sm)],
890            )
891            .await
892            .ok();
893
894        Some(operation_id)
895    }
896
897    fn recover_contract_keys(
898        &self,
899        contract: &IncomingContract,
900    ) -> Option<(Keypair, AggregateDecryptionKey)> {
901        let ephemeral_tweak = ecdh::SharedSecret::new(
902            &contract.commitment.ephemeral_pk,
903            &self.keypair.secret_key(),
904        )
905        .secret_bytes();
906
907        let encryption_seed = ephemeral_tweak
908            .consensus_hash::<sha256::Hash>()
909            .to_byte_array();
910
911        let claim_keypair = self
912            .keypair
913            .secret_key()
914            .mul_tweak(&Scalar::from_be_bytes(ephemeral_tweak).expect("Within curve order"))
915            .expect("Tweak is valid")
916            .keypair(secp256k1::SECP256K1);
917
918        if claim_keypair.public_key() != contract.commitment.claim_pk {
919            return None; // The claim key is not derived from our pk
920        }
921
922        let agg_decryption_key = derive_agg_dk(&self.cfg.tpe_agg_pk, &encryption_seed);
923
924        if !contract.verify_agg_decryption_key(&self.cfg.tpe_agg_pk, &agg_decryption_key) {
925            return None; // The decryption key is not derived from our pk
926        }
927
928        contract.decrypt_preimage(&agg_decryption_key)?;
929
930        Some((claim_keypair, agg_decryption_key))
931    }

933    /// Subscribe to all state updates of the receive operation.
934    pub async fn subscribe_receive_operation_state_updates(
935        &self,
936        operation_id: OperationId,
937    ) -> anyhow::Result<UpdateStreamOrOutcome<ReceiveOperationState>> {
938        let operation = self.client_ctx.get_operation(operation_id).await?;
939        let mut stream = self.notifier.subscribe(operation_id).await;
940        let client_ctx = self.client_ctx.clone();
941
942        Ok(self.client_ctx.outcome_or_updates(operation, operation_id, move || {
943            stream! {
944                loop {
945                    if let Some(LightningClientStateMachines::Receive(state)) = stream.next().await {
946                        match state.state {
947                            ReceiveSMState::Pending => yield ReceiveOperationState::Pending,
948                            ReceiveSMState::Claiming(out_points) => {
949                                yield ReceiveOperationState::Claiming;
950
951                                if client_ctx.await_primary_module_outputs(operation_id, out_points).await.is_ok() {
952                                    yield ReceiveOperationState::Claimed;
953                                } else {
954                                    yield ReceiveOperationState::Failure;
955                                }
956                                return;
957                            },
958                            ReceiveSMState::Expired => {
959                                yield ReceiveOperationState::Expired;
960                                return;
961                            }
962                        }
963                    }
964                }
965            }
966        }))
967    }

970    pub async fn await_final_receive_operation_state(
971        &self,
972        operation_id: OperationId,
973    ) -> anyhow::Result<FinalReceiveOperationState> {

975            .subscribe_receive_operation_state_updates(operation_id)
976            .await?

982                    ReceiveOperationState::Failure => Some(FinalReceiveOperationState::Failure),
983                    _ => None,
984                })
985            })
986            .next()
987            .await
988            .expect("Stream contains one final state");
989
990        Ok(state)
991    }
