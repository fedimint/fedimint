87    async fn connect(&self) -> Result<LndClient, LightningRpcError> {
88        let mut retries = 0;
89        let client = loop {
90            if retries >= MAX_LIGHTNING_RETRIES {
91                return Err(LightningRpcError::FailedToConnect);
92            }
93
94            retries += 1;
95
96            match connect(
97                self.address.clone(),
98                self.tls_cert.clone(),
99                self.macaroon.clone(),
100            )
101            .await
102            {
103                Ok(client) => break client,
104                Err(e) => {
105                    tracing::debug!("Couldn't connect to LND, retrying in 1 second... {e:?}");
106                    sleep(Duration::from_secs(1)).await;
107                }
108            }
109        };
110
111        Ok(client)
112    }

114    /// Spawns a new background task that subscribes to updates of a specific
115    /// HOLD invoice. When the HOLD invoice is ACCEPTED, we can request the
116    /// preimage from the Gateway. A new task is necessary because LND's
117    /// global `subscribe_invoices` does not currently emit updates for HOLD invoices: <https://github.com/lightningnetwork/lnd/issues/3120>
118    async fn spawn_lnv2_hold_invoice_subscription(
119        &self,
120        task_group: &TaskGroup,
121        gateway_sender: HtlcSubscriptionSender,
122        payment_hash: Vec<u8>,
123    ) -> Result<(), LightningRpcError> {
124        let mut client = self.connect().await?;
125
126        let self_copy = self.clone();
127        let r_hash = payment_hash.clone();
128        task_group.spawn("LND HOLD Invoice Subscription", |handle| async move {
129            let future_stream =
130                client
131                    .invoices()
132                    .subscribe_single_invoice(SubscribeSingleInvoiceRequest {
133                        r_hash: r_hash.clone(),
134                    });
135
136            let mut hold_stream = tokio::select! {
137                stream = future_stream => {
138                    match stream {
139                        Ok(stream) => stream.into_inner(),
140                        Err(e) => {
141                            error!(?e, "Failed to subscribe to hold invoice updates");
142                            return;
143                        }
144                    }
145                },
146                () = handle.make_shutdown_rx() => {
147                    info!("LND HOLD Invoice Subscription received shutdown signal");
148                    return;
149                }
150            };
151
152            while let Some(hold) = tokio::select! {
153                () = handle.make_shutdown_rx() => {
154                    None
155                }
156                hold_update = hold_stream.message() => {
157                    match hold_update {
158                        Ok(hold) => hold,
159                        Err(e) => {
160                            error!(?e, "Error received over hold invoice update stream");
161                            None
162                        }
163                    }
164                }
165            } {
166                debug!(
167                    ?hold,
168                    "LND HOLD Invoice Update {}",
169                    PrettyPaymentHash(&r_hash)
170                );
171
172                if hold.state() == InvoiceState::Accepted {
173                    let intercept = InterceptPaymentRequest {
174                        payment_hash: Hash::from_slice(&hold.r_hash.clone())
175                            .expect("Failed to convert to Hash"),
176                        amount_msat: hold.amt_paid_msat as u64,
177                        // The rest of the fields are not used in LNv2 and can be removed once LNv1
178                        // support is over
179                        expiry: hold.expiry as u32,
180                        short_channel_id: Some(0),
181                        incoming_chan_id: 0,
182                        htlc_id: 0,
183                    };
184
185                    match gateway_sender.send(intercept).await {
186                        Ok(()) => {}
187                        Err(e) => {
188                            error!(
189                                ?e,
190                                "Hold Invoice Subscription failed to send Intercept to gateway"
191                            );
192                            let _ = self_copy.cancel_hold_invoice(hold.r_hash).await;
193                        }
194                    }
195                }
196            }
197        });
198
199        Ok(())
200    }

202    /// Spawns a new background task that subscribes to "add" updates for all
203    /// invoices. This is used to detect when a new invoice has been
204    /// created. If this invoice is a HOLD invoice, it is potentially destined
205    /// for a federation. At this point, we spawn a separate task to monitor the
206    /// status of the HOLD invoice.
207    async fn spawn_lnv2_invoice_subscription(
208        &self,
209        task_group: &TaskGroup,
210        gateway_sender: HtlcSubscriptionSender,
211    ) -> Result<(), LightningRpcError> {
212        let mut client = self.connect().await?;
213
214        // Compute the minimum `add_index` that we need to subscribe to updates for.
215        let add_index = client
216            .lightning()
217            .list_invoices(ListInvoiceRequest {
218                pending_only: true,
219                index_offset: 0,
220                num_max_invoices: u64::MAX,
221                reversed: false,
222            })
223            .await
224            .map_err(|status| {
225                error!(?status, "Failed to list all invoices");
226                LightningRpcError::FailedToRouteHtlcs {
227                    failure_reason: "Failed to list all invoices".to_string(),
228                }
229            })?
230            .into_inner()
231            .first_index_offset;
232
233        let self_copy = self.clone();
234        let hold_group = task_group.make_subgroup();
235        task_group.spawn("LND Invoice Subscription", move |handle| async move {
236            let future_stream = client.lightning().subscribe_invoices(InvoiceSubscription {
237                add_index,
238                settle_index: u64::MAX, // we do not need settle invoice events
239            });
240            let mut invoice_stream = tokio::select! {
241                stream = future_stream => {
242                    match stream {
243                        Ok(stream) => stream.into_inner(),
244                        Err(e) => {
245                            error!(?e, "Failed to subscribe to all invoice updates");
246                            return;
247                        }
248                    }
249                },
250                () = handle.make_shutdown_rx() => {
251                    info!("LND Invoice Subscription received shutdown signal");
252                    return;
253                }
254            };
255
256            info!("LND Invoice Subscription: starting to process invoice updates");
257            while let Some(invoice) = tokio::select! {
258                () = handle.make_shutdown_rx() => {
259                    info!("LND Invoice Subscription task received shutdown signal");
260                    None
261                }
262                invoice_update = invoice_stream.message() => {
263                    match invoice_update {
264                        Ok(invoice) => invoice,
265                        Err(e) => {
266                            error!(?e, "Error received over invoice update stream");
267                            None
268                        }
269                    }
270                }
271            } {
272                // If the `r_preimage` is empty and the invoice is OPEN, this means a new HOLD
273                // invoice has been created, which is potentially an invoice destined for a
274                // federation. We will spawn a new task to monitor the status of
275                // the HOLD invoice.
276                let payment_hash = invoice.r_hash.clone();
277
278                debug!(
279                    ?invoice,
280                    "LND Invoice Update {}",
281                    PrettyPaymentHash(&payment_hash),
282                );
283
284                if invoice.r_preimage.is_empty() && invoice.state() == InvoiceState::Open {
285                    info!(
286                        "Monitoring new LNv2 invoice with {}",
287                        PrettyPaymentHash(&payment_hash)
288                    );
289                    if let Err(e) = self_copy
290                        .spawn_lnv2_hold_invoice_subscription(
291                            &hold_group,
292                            gateway_sender.clone(),
293                            payment_hash.clone(),
294                        )
295                        .await
296                    {
297                        error!(
298                            ?e,
299                            "Failed to spawn HOLD invoice subscription task {}",
300                            PrettyPaymentHash(&payment_hash),
301                        );
302                    }
303                }
304            }
305        });
306
307        Ok(())
308    }

310    /// Spawns a new background task that intercepts HTLCs from the LND node. In
311    /// the LNv1 protocol, this is used as a trigger mechanism for
312    /// requesting the Gateway to retrieve the preimage for a payment.
313    async fn spawn_lnv1_htlc_interceptor(
314        &self,
315        task_group: &TaskGroup,
316        lnd_sender: mpsc::Sender<ForwardHtlcInterceptResponse>,
317        lnd_rx: mpsc::Receiver<ForwardHtlcInterceptResponse>,
318        gateway_sender: HtlcSubscriptionSender,
319    ) -> Result<(), LightningRpcError> {
320        let mut client = self.connect().await?;
321
322        // Verify that LND is reachable via RPC before attempting to spawn a new thread
323        // that will intercept HTLCs.
324        client
325            .lightning()
326            .get_info(GetInfoRequest {})
327            .await
328            .map_err(|status| LightningRpcError::FailedToGetNodeInfo {
329                failure_reason: format!("Failed to get node info {status:?}"),
330            })?;
331
332        task_group.spawn("LND HTLC Subscription", |handle| async move {
333                let future_stream = client
334                    .router()
335                    .htlc_interceptor(ReceiverStream::new(lnd_rx));
336                let mut htlc_stream = tokio::select! {
337                    stream = future_stream => {
338                        match stream {
339                            Ok(stream) => stream.into_inner(),
340                            Err(e) => {
341                                error!("Failed to establish htlc stream");
342                                let e = LightningRpcError::FailedToGetRouteHints {
343                                    failure_reason: format!("Failed to subscribe to LND htlc stream {e:?}"),
344                                };
345                                debug!("Error: {e}");
346                                return;
347                            }
348                        }
349                    },
350                    () = handle.make_shutdown_rx() => {
351                        info!("LND HTLC Subscription received shutdown signal while trying to intercept HTLC stream, exiting...");
352                        return;
353                    }
354                };
355
356                debug!("LND HTLC Subscription: starting to process stream");
357                // To gracefully handle shutdown signals, we need to be able to receive signals
358                // while waiting for the next message from the HTLC stream.
359                //
360                // If we're in the middle of processing a message from the stream, we need to
361                // finish before stopping the spawned task. Checking if the task group is
362                // shutting down at the start of each iteration will cause shutdown signals to
363                // not process until another message arrives from the HTLC stream, which may
364                // take a long time, or never.
365                while let Some(htlc) = tokio::select! {
366                    () = handle.make_shutdown_rx() => {
367                        info!("LND HTLC Subscription task received shutdown signal");
368                        None
369                    }
370                    htlc_message = htlc_stream.message() => {
371                        match htlc_message {
372                            Ok(htlc) => htlc,
373                            Err(e) => {
374                                error!(?e, "Error received over HTLC stream");
375                                None
376                            }
377                    }}
378                } {
379                    trace!("LND HTLC Subscription: handling htlc {htlc:?}");
380
381                    if htlc.incoming_circuit_key.is_none() {
382                        error!("Cannot route htlc with None incoming_circuit_key");
383                        continue;
384                    }
385
386                    let incoming_circuit_key = htlc.incoming_circuit_key.unwrap();
387
388                    // Forward all HTLCs to gatewayd, gatewayd will filter them based on scid
389                    let intercept = InterceptPaymentRequest {
390                        payment_hash: Hash::from_slice(&htlc.payment_hash).expect("Failed to convert payment Hash"),
391                        amount_msat: htlc.outgoing_amount_msat,
392                        expiry: htlc.incoming_expiry,
393                        short_channel_id: Some(htlc.outgoing_requested_chan_id),
394                        incoming_chan_id: incoming_circuit_key.chan_id,
395                        htlc_id: incoming_circuit_key.htlc_id,
396                    };
397
398                    match gateway_sender.send(intercept).await {
399                        Ok(()) => {}
400                        Err(e) => {
401                            error!("Failed to send HTLC to gatewayd for processing: {:?}", e);
402                            let _ = Self::cancel_htlc(incoming_circuit_key, lnd_sender.clone())
403                                .await
404                                .map_err(|e| {
405                                    error!("Failed to cancel HTLC: {:?}", e);
406                                });
407                        }
408                    }
409                }
410            });
411
412        Ok(())
413    }

461    async fn lookup_payment(
462        &self,
463        payment_hash: Vec<u8>,
464        client: &mut LndClient,
465    ) -> Result<Option<String>, LightningRpcError> {
466        // Loop until we successfully get the status of the payment, or determine that
467        // the payment has not been made yet.
468        loop {
469            let payments = client
470                .router()
471                .track_payment_v2(TrackPaymentRequest {
472                    payment_hash: payment_hash.clone(),
473                    no_inflight_updates: true,
474                })
475                .await;
476
477            match payments {
478                Ok(payments) => {
479                    // Block until LND returns the completed payment
480                    if let Some(payment) =
481                        payments.into_inner().message().await.map_err(|status| {
482                            LightningRpcError::FailedPayment {
483                                failure_reason: status.message().to_string(),
484                            }
485                        })?
486                    {
487                        if payment.status() == PaymentStatus::Succeeded {
488                            return Ok(Some(payment.payment_preimage));
489                        }
490
491                        let failure_reason = payment.failure_reason();
492                        return Err(LightningRpcError::FailedPayment {
493                            failure_reason: format!("{failure_reason:?}"),
494                        });
495                    }
496                }
497                Err(e) => {
498                    // Break if we got a response back from the LND node that indicates the payment
499                    // hash was not found.
500                    if e.code() == Code::NotFound {
501                        return Ok(None);
502                    }
503
504                    warn!(
505                        "Could not get the status of payment {payment_hash:?} Error: {e:?}. Trying again in 5 seconds"
506                    );
507                    sleep(Duration::from_secs(5)).await;
508                }
509            }
510        }
511    }

513    /// Settles a HOLD invoice that is specified by the `payment_hash` with the
514    /// given `preimage`. If there is no invoice corresponding to the
515    /// `payment_hash`, this function will return an error.
516    async fn settle_hold_invoice(
517        &self,
518        payment_hash: Vec<u8>,
519        preimage: Preimage,
520    ) -> Result<(), LightningRpcError> {
521        let mut client = self.connect().await?;
522        let invoice = client
523            .invoices()
524            .lookup_invoice_v2(LookupInvoiceMsg {
525                invoice_ref: Some(InvoiceRef::PaymentHash(payment_hash.clone())),
526                lookup_modifier: 0,
527            })
528            .await
529            .map_err(|_| LightningRpcError::FailedToCompleteHtlc {
530                failure_reason: "Hold invoice does not exist".to_string(),
531            })?
532            .into_inner();
533
534        let state = invoice.state();
535        if state != InvoiceState::Accepted {
536            error!(
537                ?state,
538                "HOLD invoice state is not accepted {}",
539                PrettyPaymentHash(&payment_hash)
540            );
541            return Err(LightningRpcError::FailedToCompleteHtlc {
542                failure_reason: "HOLD invoice state is not accepted".to_string(),
543            });
544        }
545
546        client
547            .invoices()
548            .settle_invoice(SettleInvoiceMsg {
549                preimage: preimage.0.to_vec(),
550            })
551            .await
552            .map_err(|e| {
553                error!(
554                    ?e,
555                    "Failed to settle HOLD invoice {}",
556                    PrettyPaymentHash(&payment_hash)
557                );
558                LightningRpcError::FailedToCompleteHtlc {
559                    failure_reason: "Failed to settle HOLD invoice".to_string(),
560                }
561            })?;
562
563        Ok(())
564    }

566    /// Cancels a HOLD invoice that is specified by the `payment_hash`.
567    /// If there is no invoice corresponding to the `payment_hash`, this
568    /// function will return an error.
569    async fn cancel_hold_invoice(&self, payment_hash: Vec<u8>) -> Result<(), LightningRpcError> {
570        let mut client = self.connect().await?;
571        let invoice = client
572            .invoices()
573            .lookup_invoice_v2(LookupInvoiceMsg {
574                invoice_ref: Some(InvoiceRef::PaymentHash(payment_hash.clone())),
575                lookup_modifier: 0,
576            })
577            .await
578            .map_err(|_| LightningRpcError::FailedToCompleteHtlc {
579                failure_reason: "Hold invoice does not exist".to_string(),
580            })?
581            .into_inner();
582
583        let state = invoice.state();
584        if state != InvoiceState::Open {
585            warn!(
586                ?state,
587                "Trying to cancel HOLD invoice with {} that is not OPEN, gateway likely encountered an issue",
588                PrettyPaymentHash(&payment_hash)
589            );
590        }
591
592        client
593            .invoices()
594            .cancel_invoice(CancelInvoiceMsg {
595                payment_hash: payment_hash.clone(),
596            })
597            .await
598            .map_err(|e| {
599                error!(
600                    ?e,
601                    "Failed to cancel HOLD invoice {}",
602                    PrettyPaymentHash(&payment_hash)
603                );
604                LightningRpcError::FailedToCompleteHtlc {
605                    failure_reason: "Failed to cancel HOLD invoice".to_string(),
606                }
607            })?;
608
609        Ok(())
610    }

621    async fn info(&self) -> Result<GetNodeInfoResponse, LightningRpcError> {
622        let mut client = self.connect().await?;
623        let info = client
624            .lightning()
625            .get_info(GetInfoRequest {})
626            .await
627            .map_err(|status| LightningRpcError::FailedToGetNodeInfo {
628                failure_reason: format!("Failed to get node info {status:?}"),
629            })?
630            .into_inner();
631
632        let pub_key: PublicKey =
633            info.identity_pubkey
634                .parse()
635                .map_err(|e| LightningRpcError::FailedToGetNodeInfo {
636                    failure_reason: format!("Failed to parse public key {e:?}"),
637                })?;
638
639        let network = match info
640            .chains
641            .first()
642            .ok_or_else(|| LightningRpcError::FailedToGetNodeInfo {
643                failure_reason: "Failed to parse node network".to_string(),
644            })?
645            .network
646            .as_str()
647        {
648            // LND uses "mainnet", but rust-bitcoin uses "bitcoin".
649            // TODO: create a fedimint `Network` type that understands "mainnet"
650            "mainnet" => "bitcoin",
651            other => other,
652        }
653        .to_string();
654
655        return Ok(GetNodeInfoResponse {
656            pub_key,
657            alias: info.alias,
658            network,
659            block_height: info.block_height,
660            synced_to_chain: info.synced_to_chain,
661        });
662    }

664    async fn routehints(
665        &self,
666        num_route_hints: usize,
667    ) -> Result<GetRouteHintsResponse, LightningRpcError> {
668        let mut client = self.connect().await?;
669        let mut channels = client
670            .lightning()
671            .list_channels(ListChannelsRequest {
672                active_only: true,
673                inactive_only: false,
674                public_only: false,
675                private_only: false,
676                peer: vec![],
677            })
678            .await
679            .map_err(|status| LightningRpcError::FailedToGetRouteHints {
680                failure_reason: format!("Failed to list channels {status:?}"),
681            })?
682            .into_inner()
683            .channels;
684
685        // Take the channels with the largest incoming capacity
686        channels.sort_by(|a, b| b.remote_balance.cmp(&a.remote_balance));
687        channels.truncate(num_route_hints);
688
689        let mut route_hints: Vec<RouteHint> = vec![];
690        for chan in &channels {
691            let info = client
692                .lightning()
693                .get_chan_info(ChanInfoRequest {
694                    chan_id: chan.chan_id,
695                })
696                .await
697                .map_err(|status| LightningRpcError::FailedToGetRouteHints {
698                    failure_reason: format!("Failed to get channel info {status:?}"),
699                })?
700                .into_inner();
701
702            let Some(policy) = info.node1_policy.clone() else {
703                continue;
704            };
705            let src_node_id =
706                PublicKey::from_str(&chan.remote_pubkey).expect("Failed to parse pubkey");
707            let short_channel_id = chan.chan_id;
708            let base_msat = policy.fee_base_msat as u32;
709            let proportional_millionths = policy.fee_rate_milli_msat as u32;
710            let cltv_expiry_delta = policy.time_lock_delta;
711            let htlc_maximum_msat = Some(policy.max_htlc_msat);
712            let htlc_minimum_msat = Some(policy.min_htlc as u64);
713
714            let route_hint_hop = RouteHintHop {
715                src_node_id,
716                short_channel_id,
717                base_msat,
718                proportional_millionths,
719                cltv_expiry_delta: cltv_expiry_delta as u16,
720                htlc_minimum_msat,
721                htlc_maximum_msat,
722            };
723            route_hints.push(RouteHint(vec![route_hint_hop]));
724        }
725
726        Ok(GetRouteHintsResponse { route_hints })
727    }

729    async fn pay_private(
730        &self,
731        invoice: PrunedInvoice,
732        max_delay: u64,
733        max_fee: Amount,
734    ) -> Result<PayInvoiceResponse, LightningRpcError> {
735        let payment_hash = invoice.payment_hash.to_byte_array().to_vec();
736        info!(
737            "LND Paying invoice with {}",
738            PrettyPaymentHash(&payment_hash)
739        );
740        let mut client = self.connect().await?;
741
742        debug!(
743            ?invoice,
744            "pay_private checking if payment for invoice exists"
745        );
746
747        // If the payment exists, that means we've already tried to pay the invoice
748        let preimage: Vec<u8> = match self
749            .lookup_payment(invoice.payment_hash.to_byte_array().to_vec(), &mut client)
750            .await?
751        {
752            Some(preimage) => {
753                info!(
754                    "LND payment already exists for invoice with {}",
755                    PrettyPaymentHash(&payment_hash)
756                );
757                hex::FromHex::from_hex(preimage.as_str()).map_err(|error| {
758                    LightningRpcError::FailedPayment {
759                        failure_reason: format!("Failed to convert preimage {error:?}"),
760                    }
761                })?
762            }
763            _ => {
764                // LND API allows fee limits in the `i64` range, but we use `u64` for
765                // max_fee_msat. This means we can only set an enforceable fee limit
766                // between 0 and i64::MAX
767                let fee_limit_msat: i64 =
768                    max_fee
769                        .msats
770                        .try_into()
771                        .map_err(|error| LightningRpcError::FailedPayment {
772                            failure_reason: format!(
773                                "max_fee_msat exceeds valid LND fee limit ranges {error:?}"
774                            ),
775                        })?;
776
777                let amt_msat = invoice.amount.msats.try_into().map_err(|error| {
778                    LightningRpcError::FailedPayment {
779                        failure_reason: format!("amount exceeds valid LND amount ranges {error:?}"),
780                    }
781                })?;
782                let final_cltv_delta =
783                    invoice.min_final_cltv_delta.try_into().map_err(|error| {
784                        LightningRpcError::FailedPayment {
785                            failure_reason: format!(
786                                "final cltv delta exceeds valid LND range {error:?}"
787                            ),
788                        }
789                    })?;
790                let cltv_limit =
791                    max_delay
792                        .try_into()
793                        .map_err(|error| LightningRpcError::FailedPayment {
794                            failure_reason: format!("max delay exceeds valid LND range {error:?}"),
795                        })?;
796
797                let dest_features = wire_features_to_lnd_feature_vec(&invoice.destination_features)
798                    .map_err(|e| LightningRpcError::FailedPayment {
799                        failure_reason: e.to_string(),
800                    })?;
801
802                debug!(
803                    "LND payment does not exist for invoice with {}, will attempt to pay",
804                    PrettyPaymentHash(&payment_hash)
805                );
806                let payments = client
807                    .router()
808                    .send_payment_v2(SendPaymentRequest {
809                        amt_msat,
810                        dest: invoice.destination.serialize().to_vec(),
811                        dest_features,
812                        payment_hash: invoice.payment_hash.to_byte_array().to_vec(),
813                        payment_addr: invoice.payment_secret.to_vec(),
814                        route_hints: route_hints_to_lnd(&invoice.route_hints),
815                        final_cltv_delta,
816                        cltv_limit,
817                        no_inflight_updates: false,
818                        timeout_seconds: LND_PAYMENT_TIMEOUT_SECONDS,
819                        fee_limit_msat,
820                        ..Default::default()
821                    })
822                    .await
823                    .map_err(|status| {
824                        error!(
825                            "LND payment request failed for invoice with {} with {status:?}",
826                            PrettyPaymentHash(&payment_hash)
827                        );
828                        LightningRpcError::FailedPayment {
829                            failure_reason: format!("Failed to make outgoing payment {status:?}"),
830                        }
831                    })?;
832
833                debug!(
834                    "LND payment request sent for invoice with {}, waiting for payment status...",
835                    PrettyPaymentHash(&payment_hash),
836                );
837                let mut messages = payments.into_inner();
838                loop {
839                    match messages.message().await.map_err(|error| {
840                        LightningRpcError::FailedPayment {
841                            failure_reason: format!("Failed to get payment status {error:?}"),
842                        }
843                    }) {
844                        Ok(Some(payment)) if payment.status() == PaymentStatus::Succeeded => {
845                            info!(
846                                "LND payment succeeded for invoice with {}",
847                                PrettyPaymentHash(&payment_hash)
848                            );
849                            break hex::FromHex::from_hex(payment.payment_preimage.as_str())
850                                .map_err(|error| LightningRpcError::FailedPayment {
851                                    failure_reason: format!("Failed to convert preimage {error:?}"),
852                                })?;
853                        }
854                        Ok(Some(payment)) if payment.status() == PaymentStatus::InFlight => {
855                            debug!(
856                                "LND payment for invoice with {} is inflight",
857                                PrettyPaymentHash(&payment_hash)
858                            );
859                            continue;
860                        }
861                        Ok(Some(payment)) => {
862                            error!(
863                                "LND payment failed for invoice with {} with {payment:?}",
864                                PrettyPaymentHash(&payment_hash)
865                            );
866                            let failure_reason = payment.failure_reason();
867                            return Err(LightningRpcError::FailedPayment {
868                                failure_reason: format!("{failure_reason:?}"),
869                            });
870                        }
871                        Ok(None) => {
872                            error!(
873                                "LND payment failed for invoice with {} with no payment status",
874                                PrettyPaymentHash(&payment_hash)
875                            );
876                            return Err(LightningRpcError::FailedPayment {
877                                failure_reason: format!(
878                                    "Failed to get payment status for payment hash {:?}",
879                                    invoice.payment_hash
880                                ),
881                            });
882                        }
883                        Err(e) => {
884                            error!(
885                                "LND payment failed for invoice with {} with {e:?}",
886                                PrettyPaymentHash(&payment_hash)
887                            );
888                            return Err(e);
889                        }
890                    }
891                }
892            }
893        };
894        Ok(PayInvoiceResponse {
895            preimage: Preimage(preimage.try_into().expect("Failed to create preimage")),
896        })
897    }

905    async fn route_htlcs<'a>(
906        self: Box<Self>,
907        task_group: &TaskGroup,
908    ) -> Result<(RouteHtlcStream<'a>, Arc<dyn ILnRpcClient>), LightningRpcError> {
909        const CHANNEL_SIZE: usize = 100;
910
911        // Channel to send intercepted htlc to the gateway for processing
912        let (gateway_sender, gateway_receiver) =
913            mpsc::channel::<InterceptPaymentRequest>(CHANNEL_SIZE);
914
915        let (lnd_sender, lnd_rx) = mpsc::channel::<ForwardHtlcInterceptResponse>(CHANNEL_SIZE);
916
917        self.spawn_interceptor(
918            task_group,
919            lnd_sender.clone(),
920            lnd_rx,
921            gateway_sender.clone(),
922        )
923        .await?;
924        let new_client = Arc::new(Self {
925            address: self.address.clone(),
926            tls_cert: self.tls_cert.clone(),
927            macaroon: self.macaroon.clone(),
928            lnd_sender: Some(lnd_sender.clone()),
929        });
930        Ok((Box::pin(ReceiverStream::new(gateway_receiver)), new_client))
931    }

933    async fn complete_htlc(&self, htlc: InterceptPaymentResponse) -> Result<(), LightningRpcError> {
934        let InterceptPaymentResponse {
935            action,
936            payment_hash,
937            incoming_chan_id,
938            htlc_id,
939        } = htlc;
940
941        let (action, preimage) = match action {
942            PaymentAction::Settle(preimage) => (ResolveHoldForwardAction::Settle, preimage),
943            PaymentAction::Cancel => (ResolveHoldForwardAction::Fail, Preimage([0; 32])),
944            PaymentAction::Forward => (ResolveHoldForwardAction::Resume, Preimage([0; 32])),
945        };
946
947        // First check if this completion request corresponds to a HOLD LNv2 invoice
948        match action {
949            ResolveHoldForwardAction::Settle => {
950                if let Ok(()) = self
951                    .settle_hold_invoice(payment_hash.to_byte_array().to_vec(), preimage.clone())
952                    .await
953                {
954                    info!("Successfully settled HOLD invoice {}", payment_hash);
955                    return Ok(());
956                }
957            }
958            _ => {
959                if let Ok(()) = self
960                    .cancel_hold_invoice(payment_hash.to_byte_array().to_vec())
961                    .await
962                {
963                    info!("Successfully canceled HOLD invoice {}", payment_hash);
964                    return Ok(());
965                }
966            }
967        }
968
969        // If we can't settle/cancel the payment via LNv2, try LNv1
970        if let Some(lnd_sender) = self.lnd_sender.clone() {
971            let response = ForwardHtlcInterceptResponse {
972                incoming_circuit_key: Some(CircuitKey {
973                    chan_id: incoming_chan_id,
974                    htlc_id,
975                }),
976                action: action.into(),
977                preimage: preimage.0.to_vec(),
978                failure_message: vec![],
979                failure_code: FailureCode::TemporaryChannelFailure.into(),
980            };
981
982            Self::send_lnd_response(lnd_sender, response).await?;
983            return Ok(());
984        }
985
986        error!("Gatewayd has not started to route HTLCs");
987        Err(LightningRpcError::FailedToCompleteHtlc {
988            failure_reason: "Gatewayd has not started to route HTLCs".to_string(),
989        })
990    }

993        &self,
994        create_invoice_request: CreateInvoiceRequest,
995    ) -> Result<CreateInvoiceResponse, LightningRpcError> {
996        let mut client = self.connect().await?;
997        let description = create_invoice_request
998            .description
999            .unwrap_or(InvoiceDescription::Direct(String::new()));
1000
1001        if create_invoice_request.payment_hash.is_none() {
1002            let invoice = match description {
1003                InvoiceDescription::Direct(description) => Invoice {
1004                    memo: description,
1005                    value_msat: create_invoice_request.amount_msat as i64,
1006                    expiry: i64::from(create_invoice_request.expiry_secs),
1007                    ..Default::default()
1008                },
1009                InvoiceDescription::Hash(desc_hash) => Invoice {
1010                    description_hash: desc_hash.to_byte_array().to_vec(),
1011                    value_msat: create_invoice_request.amount_msat as i64,
1012                    expiry: i64::from(create_invoice_request.expiry_secs),
1013                    ..Default::default()
1014                },
1015            };
1016
1017            let add_invoice_response =
1018                client.lightning().add_invoice(invoice).await.map_err(|e| {
1019                    LightningRpcError::FailedToGetInvoice {
1020                        failure_reason: e.to_string(),
1021                    }
1022                })?;
1023
1024            let invoice = add_invoice_response.into_inner().payment_request;
1025            Ok(CreateInvoiceResponse { invoice })
1026        } else {
1027            let payment_hash = create_invoice_request
1028                .payment_hash
1029                .expect("Already checked payment hash")
1030                .to_byte_array()
1031                .to_vec();
1032            let hold_invoice_request = match description {
1033                InvoiceDescription::Direct(description) => AddHoldInvoiceRequest {
1034                    memo: description,
1035                    hash: payment_hash.clone(),

1037                    expiry: i64::from(create_invoice_request.expiry_secs),
1038                    ..Default::default()
1039                },
1040                InvoiceDescription::Hash(desc_hash) => AddHoldInvoiceRequest {
1041                    description_hash: desc_hash.to_byte_array().to_vec(),
1042                    hash: payment_hash.clone(),
1043                    value_msat: create_invoice_request.amount_msat as i64,
1044                    expiry: i64::from(create_invoice_request.expiry_secs),
1045                    ..Default::default()
1046                },
1047            };
1048
1049            let hold_invoice_response = client
1050                .invoices()
1051                .add_hold_invoice(hold_invoice_request)
1052                .await
1053                .map_err(|e| LightningRpcError::FailedToGetInvoice {
1054                    failure_reason: e.to_string(),
1055                })?;
1056
1057            let invoice = hold_invoice_response.into_inner().payment_request;
1058            Ok(CreateInvoiceResponse { invoice })
1059        }
1060    }

1191            }),
1192            Err(e) => Err(LightningRpcError::FailedToOpenChannel {

1194            }),
1195        }
1196    }

1334            .unsettled_local_balance
1335            .unwrap_or_default();
1336        let pending_outbound = channel_balance_response
1337            .pending_open_local_balance
1338            .unwrap_or_default();
1339        let lightning_balance_msats =
1340            total_outbound.msat - unsettled_outbound.msat - pending_outbound.msat;
1341
1342        let total_inbound = channel_balance_response.remote_balance.unwrap_or_default();
1343        let unsettled_inbound = channel_balance_response
1344            .unsettled_remote_balance
1345            .unwrap_or_default();
1346        let pending_inbound = channel_balance_response
1347            .pending_open_remote_balance
1348            .unwrap_or_default();
1349        let inbound_lightning_liquidity_msats =
1350            total_inbound.msat - unsettled_inbound.msat - pending_inbound.msat;
1351
1352        Ok(GetBalancesResponse {
1353            onchain_balance_sats: (wallet_balance_response.total_balance
1354                + wallet_balance_response.reserved_balance_anchor_chan)
1355                as u64,
1356            lightning_balance_msats,
1357            inbound_lightning_liquidity_msats,
1358        })
1359    }

1361    async fn get_invoice(
1362        &self,
1363        get_invoice_request: GetInvoiceRequest,
1364    ) -> Result<Option<GetInvoiceResponse>, LightningRpcError> {
1365        let mut client = self.connect().await?;
1366        let invoice = client
1367            .invoices()
1368            .lookup_invoice_v2(LookupInvoiceMsg {
1369                invoice_ref: Some(InvoiceRef::PaymentHash(
1370                    get_invoice_request.payment_hash.consensus_encode_to_vec(),
1371                )),
1372                ..Default::default()
1373            })
1374            .await;
1375        let invoice = match invoice {
1376            Ok(invoice) => invoice.into_inner(),
1377            Err(_) => return Ok(None),
1378        };
1379        let preimage: [u8; 32] = invoice
1380            .clone()
1381            .r_preimage
1382            .try_into()
1383            .expect("Could not convert preimage");
1384        let status = match &invoice.state() {
1385            InvoiceState::Settled => fedimint_gateway_common::PaymentStatus::Succeeded,
1386            InvoiceState::Canceled => fedimint_gateway_common::PaymentStatus::Failed,
1387            _ => fedimint_gateway_common::PaymentStatus::Pending,
1388        };
1389
1390        Ok(Some(GetInvoiceResponse {
1391            preimage: Some(preimage.consensus_encode_to_hex()),
1392            payment_hash: Some(
1393                sha256::Hash::from_slice(&invoice.r_hash).expect("Could not convert payment hash"),
1394            ),
1395            amount: Amount::from_msats(invoice.value_msat as u64),
1396            created_at: UNIX_EPOCH + Duration::from_secs(invoice.creation_date as u64),
1397            status,
1398        }))
1399    }
1400}

1423fn wire_features_to_lnd_feature_vec(features_wire_encoded: &[u8]) -> anyhow::Result<Vec<i32>> {
1424    ensure!(
1425        features_wire_encoded.len() <= 1_000,
1426        "Will not process feature bit vectors larger than 1000 byte"
1427    );
1428
1429    let lnd_features = features_wire_encoded
1430        .iter()
1431        .rev()
1432        .enumerate()
1433        .flat_map(|(byte_idx, &feature_byte)| {
1434            (0..8).filter_map(move |bit_idx| {
1435                if (feature_byte & (1u8 << bit_idx)) != 0 {
1436                    Some(
1437                        i32::try_from(byte_idx * 8 + bit_idx)
1438                            .expect("Index will never exceed i32::MAX for feature vectors <8MB"),
1439                    )
1440                } else {
1441                    None
1442                }
1443            })
1444        })
1445        .collect::<Vec<_>>();
1446
1447    Ok(lnd_features)
1448}
