88    /// Creates a new `GatewayLdkClient` instance and starts the underlying
89    /// lightning node. All resources, including the lightning node, will be
90    /// cleaned up when the returned `GatewayLdkClient` instance is dropped.
91    /// There's no need to manually stop the node.
92    pub fn new(
93        data_dir: &Path,
94        chain_source_config: GatewayLdkChainSourceConfig,
95        network: Network,
96        lightning_port: u16,
97        alias: String,
98        mnemonic: Mnemonic,
99        runtime: Arc<tokio::runtime::Runtime>,
100    ) -> anyhow::Result<Self> {
101        let mut bytes = [0u8; 32];
102        let alias = if alias.is_empty() {
103            "LDK Gateway".to_string()
104        } else {
105            alias
106        };
107        let alias_bytes = alias.as_bytes();
108        let truncated = &alias_bytes[..alias_bytes.len().min(32)];
109        bytes[..truncated.len()].copy_from_slice(truncated);
110        let node_alias = Some(NodeAlias(bytes));
111
112        let mut node_builder = ldk_node::Builder::from_config(ldk_node::config::Config {
113            network,
114            listening_addresses: Some(vec![SocketAddress::TcpIpV4 {
115                addr: [0, 0, 0, 0],
116                port: lightning_port,
117            }]),
118            node_alias,
119            ..Default::default()
120        });
121
122        node_builder.set_entropy_bip39_mnemonic(mnemonic, None);
123
124        let bitcoind_rpc = create_bitcoind(&chain_source_config.bitcoin_rpc_config())?;
125
126        match chain_source_config {
127            GatewayLdkChainSourceConfig::Bitcoind { server_url } => {
128                node_builder.set_chain_source_bitcoind_rpc(
129                    server_url
130                        .host_str()
131                        .expect("Could not retrieve host from bitcoind RPC url")
132                        .to_string(),
133                    server_url
134                        .port()
135                        .expect("Could not retrieve port from bitcoind RPC url"),
136                    server_url.username().to_string(),
137                    server_url.password().unwrap_or_default().to_string(),
138                );
139            }
140            GatewayLdkChainSourceConfig::Esplora { server_url } => {
141                node_builder.set_chain_source_esplora(server_url.to_string(), None);
142            }
143        };
144        let Some(data_dir_str) = data_dir.to_str() else {
145            return Err(anyhow::anyhow!("Invalid data dir path"));
146        };
147        node_builder.set_storage_dir_path(data_dir_str.to_string());
148
149        let node = Arc::new(node_builder.build()?);
150        node.start_with_runtime(runtime).map_err(|err| {
151            crit!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed to start LDK Node");
152            LightningRpcError::FailedToConnect
153        })?;
154
155        let (htlc_stream_sender, htlc_stream_receiver) = tokio::sync::mpsc::channel(1024);
156        let task_group = TaskGroup::new();
157
158        let node_clone = node.clone();
159        task_group.spawn("ldk lightning node event handler", |handle| async move {
160            loop {
161                Self::handle_next_event(&node_clone, &htlc_stream_sender, &handle).await;
162            }
163        });
164
165        Ok(GatewayLdkClient {
166            node,
167            bitcoind_rpc,
168            task_group,
169            htlc_stream_receiver_or: Some(htlc_stream_receiver),
170            outbound_lightning_payment_lock_pool: lockable::LockPool::new(),
171        })
172    }
173
174    async fn handle_next_event(
175        node: &ldk_node::Node,
176        htlc_stream_sender: &Sender<InterceptPaymentRequest>,
177        handle: &TaskHandle,
178    ) {
179        // We manually check for task termination in case we receive a payment while the
180        // task is shutting down. In that case, we want to finish the payment
181        // before shutting this task down.
182        let event = tokio::select! {
183            event = node.next_event_async() => {
184                event
185            }
186            () = handle.make_shutdown_rx() => {
187                return;
188            }
189        };
190
191        if let ldk_node::Event::PaymentClaimable {
192            payment_id: _,
193            payment_hash,
194            claimable_amount_msat,
195            claim_deadline,
196        } = event
197        {
198            if let Err(err) = htlc_stream_sender
199                .send(InterceptPaymentRequest {
200                    payment_hash: Hash::from_slice(&payment_hash.0).expect("Failed to create Hash"),
201                    amount_msat: claimable_amount_msat,
202                    expiry: claim_deadline.unwrap_or_default(),
203                    short_channel_id: None,
204                    incoming_chan_id: 0,
205                    htlc_id: 0,
206                })
207                .await
208            {
209                warn!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed send InterceptHtlcRequest to stream");
210            }
211        }
212
213        // The `PaymentClaimable` event is the only event type that we are interested
214        // in. We can safely ignore all other events.
215        node.event_handled();
216    }

218    /// Converts a transaction outpoint to a short channel ID by querying the
219    /// blockchain.
220    async fn outpoint_to_scid(&self, funding_txo: OutPoint) -> anyhow::Result<u64> {
221        let block_hash = self
222            .bitcoind_rpc
223            .get_txout_proof(funding_txo.txid)
224            .await?
225            .block_header
226            .block_hash();
227
228        let block_height = self
229            .bitcoind_rpc
230            .get_tx_block_height(&funding_txo.txid)
231            .await?
232            .ok_or(anyhow::anyhow!("Failed to get block height"))?;
233
234        let block = self.bitcoind_rpc.get_block(&block_hash).await?;
235
236        let tx_index = block
237            .txdata
238            .iter()
239            .enumerate()
240            .find(|(_, tx)| tx.compute_txid() == funding_txo.txid)
241            .ok_or(anyhow::anyhow!("Failed to find transaction"))?
242            .0 as u32;
243
244        let output_index = funding_txo.vout;
245
246        scid_from_parts(block_height, u64::from(tx_index), u64::from(output_index))
247            .map_err(|e| anyhow::anyhow!("Failed to convert to short channel ID: {e:?}"))
248    }

251impl Drop for GatewayLdkClient {
252    fn drop(&mut self) {
253        self.task_group.shutdown();
254
255        info!(target: LOG_LIGHTNING, "Stopping LDK Node...");
256        match self.node.stop() {
257            Err(err) => {
258                warn!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed to stop LDK Node");
259            }
260            _ => {
261                info!(target: LOG_LIGHTNING, "LDK Node stopped.");
262            }
263        }
264    }

321    async fn pay(
322        &self,
323        invoice: Bolt11Invoice,
324        max_delay: u64,
325        max_fee: Amount,
326    ) -> Result<PayInvoiceResponse, LightningRpcError> {
327        let payment_id = PaymentId(*invoice.payment_hash().as_byte_array());
328
329        // Lock by the payment hash to prevent multiple simultaneous calls with the same
330        // invoice from executing. This prevents `ldk-node::Bolt11Payment::send()` from
331        // being called multiple times with the same invoice. This is important because
332        // `ldk-node::Bolt11Payment::send()` is not idempotent, but this function must
333        // be idempotent.
334        let _payment_lock_guard = self
335            .outbound_lightning_payment_lock_pool
336            .async_lock(payment_id)
337            .await;
338
339        // If a payment is not known to the node we can initiate it, and if it is known
340        // we can skip calling `ldk-node::Bolt11Payment::send()` and wait for the
341        // payment to complete. The lock guard above guarantees that this block is only
342        // executed once at a time for a given payment hash, ensuring that there is no
343        // race condition between checking if a payment is known and initiating a new
344        // payment if it isn't.
345        if self.node.payment(&payment_id).is_none() {
346            assert_eq!(
347                self.node
348                    .bolt11_payment()
349                    .send(
350                        &invoice,
351                        Some(SendingParameters {
352                            max_total_routing_fee_msat: Some(Some(max_fee.msats)),
353                            max_total_cltv_expiry_delta: Some(max_delay as u32),
354                            max_path_count: None,
355                            max_channel_saturation_power_of_half: None,
356                        }),
357                    )
358                    // TODO: Investigate whether all error types returned by `Bolt11Payment::send()`
359                    // result in idempotency.
360                    .map_err(|e| LightningRpcError::FailedPayment {
361                        failure_reason: format!("LDK payment failed to initialize: {e:?}"),
362                    })?,
363                payment_id
364            );
365        }
366
367        // TODO: Find a way to avoid looping/polling to know when a payment is
368        // completed. `ldk-node` provides `PaymentSuccessful` and `PaymentFailed`
369        // events, but interacting with the node event queue here isn't
370        // straightforward.
371        loop {
372            if let Some(payment_details) = self.node.payment(&payment_id) {
373                match payment_details.status {
374                    PaymentStatus::Pending => {}
375                    PaymentStatus::Succeeded => {
376                        if let PaymentKind::Bolt11 {
377                            preimage: Some(preimage),
378                            ..
379                        } = payment_details.kind
380                        {
381                            return Ok(PayInvoiceResponse {
382                                preimage: Preimage(preimage.0),
383                            });
384                        }
385                    }
386                    PaymentStatus::Failed => {
387                        return Err(LightningRpcError::FailedPayment {
388                            failure_reason: "LDK payment failed".to_string(),
389                        });
390                    }
391                }
392            }
393            fedimint_core::runtime::sleep(Duration::from_millis(100)).await;
394        }
395    }

398        mut self: Box<Self>,

413    async fn complete_htlc(&self, htlc: InterceptPaymentResponse) -> Result<(), LightningRpcError> {
414        let InterceptPaymentResponse {
415            action,
416            payment_hash,
417            incoming_chan_id: _,
418            htlc_id: _,
419        } = htlc;
420
421        let ph = PaymentHash(*payment_hash.clone().as_byte_array());
422
423        // TODO: Get the actual amount from the LDK node. Probably makes the
424        // most sense to pipe it through the `InterceptHtlcResponse` struct.
425        // This value is only used by `ldk-node` to ensure that the amount
426        // claimed isn't less than the amount expected, but we've already
427        // verified that the amount is correct when we intercepted the payment.
428        let claimable_amount_msat = 999_999_999_999_999;
429
430        let ph_hex_str = hex::encode(payment_hash);
431
432        if let PaymentAction::Settle(preimage) = action {
433            self.node
434                .bolt11_payment()
435                .claim_for_hash(ph, claimable_amount_msat, PaymentPreimage(preimage.0))
436                .map_err(|_| LightningRpcError::FailedToCompleteHtlc {
437                    failure_reason: format!("Failed to claim LDK payment with hash {ph_hex_str}"),
438                })?;
439        } else {
440            warn!(target: LOG_LIGHTNING, payment_hash = %ph_hex_str, "Unwinding payment because the action was not `Settle`");
441            self.node.bolt11_payment().fail_for_hash(ph).map_err(|_| {
442                LightningRpcError::FailedToCompleteHtlc {
443                    failure_reason: format!("Failed to unwind LDK payment with hash {ph_hex_str}"),
444                }
445            })?;
446        }
447
448        return Ok(());
449    }

451    async fn create_invoice(
452        &self,
453        create_invoice_request: CreateInvoiceRequest,
454    ) -> Result<CreateInvoiceResponse, LightningRpcError> {
455        let payment_hash_or = if let Some(payment_hash) = create_invoice_request.payment_hash {
456            let ph = PaymentHash(*payment_hash.as_byte_array());
457            Some(ph)
458        } else {
459            None
460        };
461
462        // Currently `ldk-node` only supports direct descriptions.
463        // See https://github.com/lightningdevkit/ldk-node/issues/325.
464        // TODO: Once the above issue is resolved, we should support
465        // description hashes as well.
466        let description_str = match create_invoice_request.description {
467            Some(InvoiceDescription::Direct(desc)) => desc,
468            _ => String::new(),
469        };
470
471        let invoice = match payment_hash_or {
472            Some(payment_hash) => self.node.bolt11_payment().receive_for_hash(
473                create_invoice_request.amount_msat,
474                description_str.as_str(),
475                create_invoice_request.expiry_secs,
476                payment_hash,
477            ),
478            None => self.node.bolt11_payment().receive(
479                create_invoice_request.amount_msat,
480                description_str.as_str(),
481                create_invoice_request.expiry_secs,
482            ),
483        }
484        .map_err(|e| LightningRpcError::FailedToGetInvoice {
485            failure_reason: e.to_string(),
486        })?;
487
488        Ok(CreateInvoiceResponse {
489            invoice: invoice.to_string(),
490        })
491    }

508        &self,

510            address,
511            amount,
512            // TODO: Respect this fee rate once `ldk-node` supports setting a custom fee rate.

535    async fn open_channel(
536        &self,
537        OpenChannelRequest {
538            pubkey,
539            host,
540            channel_size_sats,
541            push_amount_sats,
542        }: OpenChannelRequest,
543    ) -> Result<OpenChannelResponse, LightningRpcError> {
544        let push_amount_msats_or = if push_amount_sats == 0 {
545            None
546        } else {
547            Some(push_amount_sats * 1000)
548        };
549
550        let user_channel_id = self
551            .node
552            .open_announced_channel(
553                pubkey,
554                SocketAddress::from_str(&host).map_err(|e| {
555                    LightningRpcError::FailedToConnectToPeer {
556                        failure_reason: e.to_string(),
557                    }
558                })?,
559                channel_size_sats,
560                push_amount_msats_or,
561                None,
562            )
563            .map_err(|e| LightningRpcError::FailedToOpenChannel {
564                failure_reason: e.to_string(),
565            })?;
566
567        // The channel isn't always visible immediately, so we need to poll for it.
568        for _ in 0..10 {
569            let funding_txid_or = self
570                .node
571                .list_channels()
572                .iter()
573                .find(|channel| channel.user_channel_id == user_channel_id)
574                .and_then(|channel| channel.funding_txo)
575                .map(|funding_txo| funding_txo.txid);
576
577            if let Some(funding_txid) = funding_txid_or {
578                return Ok(OpenChannelResponse {
579                    funding_txid: funding_txid.to_string(),
580                });
581            }
582
583            fedimint_core::runtime::sleep(Duration::from_millis(100)).await;
584        }
585
586        Err(LightningRpcError::FailedToOpenChannel {
587            failure_reason: "Channel could not be opened".to_string(),
588        })
589    }

593        CloseChannelsWithPeerRequest { pubkey }: CloseChannelsWithPeerRequest,
594    ) -> Result<CloseChannelsWithPeerResponse, LightningRpcError> {
595        let mut num_channels_closed = 0;

598            .node

600            .iter()
601            .filter(|channel| channel.counterparty_node_id == pubkey)

606                .is_ok()

609            }
610        }
611

613            num_channels_closed,
614        })

641    async fn get_balances(&self) -> Result<GetBalancesResponse, LightningRpcError> {
642        let balances = self.node.list_balances();
643        let channel_lists = self
644            .node
645            .list_channels()
646            .into_iter()
647            .filter(|chan| chan.is_usable)
648            .collect::<Vec<_>>();
649        // map and get the total inbound_capacity_msat in the channels
650        let total_inbound_liquidity_balance_msat: u64 = channel_lists
651            .iter()
652            .map(|channel| channel.inbound_capacity_msat)
653            .sum();
654
655        Ok(GetBalancesResponse {
656            onchain_balance_sats: balances.total_onchain_balance_sats,
657            lightning_balance_msats: balances.total_lightning_balance_sats * 1000,
658            inbound_lightning_liquidity_msats: total_inbound_liquidity_balance_msat,
659        })
660    }
661
662    async fn get_invoice(
663        &self,
664        get_invoice_request: GetInvoiceRequest,
665    ) -> Result<Option<GetInvoiceResponse>, LightningRpcError> {
666        let invoices = self
667            .node
668            .list_payments_with_filter(|details| {
669                details.direction == PaymentDirection::Inbound
670                    && details.id == PaymentId(get_invoice_request.payment_hash.to_byte_array())
671                    && !matches!(details.kind, PaymentKind::Onchain)
672            })
673            .iter()
674            .map(|details| {
675                let (preimage, payment_hash) = get_preimage_and_payment_hash(&details.kind);
676                let status = match details.status {
677                    PaymentStatus::Failed => fedimint_gateway_common::PaymentStatus::Failed,
678                    PaymentStatus::Succeeded => fedimint_gateway_common::PaymentStatus::Succeeded,
679                    PaymentStatus::Pending => fedimint_gateway_common::PaymentStatus::Pending,
680                };
681                GetInvoiceResponse {
682                    preimage: preimage.map(|p| p.0.consensus_encode_to_hex()),
683                    payment_hash,
684                    amount: Amount::from_msats(
685                        details
686                            .amount_msat
687                            .expect("amountless invoices are not supported"),
688                    ),
689                    created_at: UNIX_EPOCH + Duration::from_secs(details.latest_update_timestamp),
690                    status,
691                }
692            })
693            .collect::<Vec<_>>();
694
695        Ok(invoices.first().cloned())
696    }
697}

699/// Maps LDK's `PaymentKind` to an optional preimage and an optional payment
700/// hash depending on the type of payment.
701fn get_preimage_and_payment_hash(kind: &PaymentKind) -> (Option<Preimage>, Option<sha256::Hash>) {
702    match kind {
703        PaymentKind::Bolt11 {
704            hash,
705            preimage,
706            secret: _,
707        } => (
708            preimage.map(|p| Preimage(p.0)),
709            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
710        ),
711        PaymentKind::Bolt11Jit {
712            hash,
713            preimage,
714            secret: _,
715            lsp_fee_limits: _,
716        } => (
717            preimage.map(|p| Preimage(p.0)),
718            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
719        ),
720        PaymentKind::Bolt12Offer {
721            hash,
722            preimage,
723            secret: _,
724            offer_id: _,
725            payer_note: _,
726            quantity: _,
727        } => (
728            preimage.map(|p| Preimage(p.0)),
729            hash.map(|h| sha256::Hash::from_slice(&h.0).expect("Failed to convert payment hash")),
730        ),
731        PaymentKind::Bolt12Refund {
732            hash,
733            preimage,
734            secret: _,
735            payer_note: _,
736            quantity: _,
737        } => (
738            preimage.map(|p| Preimage(p.0)),
739            hash.map(|h| sha256::Hash::from_slice(&h.0).expect("Failed to convert payment hash")),
740        ),
741        PaymentKind::Spontaneous { hash, preimage } => (
742            preimage.map(|p| Preimage(p.0)),
743            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
744        ),
745        PaymentKind::Onchain => panic!("No preimage for onchain payment"),
746    }
747}
