67impl GatewayLdkClient {
68    /// Creates a new `GatewayLdkClient` instance and starts the underlying
69    /// lightning node. All resources, including the lightning node, will be
70    /// cleaned up when the returned `GatewayLdkClient` instance is dropped.
71    /// There's no need to manually stop the node.
72    pub fn new(
73        data_dir: &Path,
74        chain_source: ChainSource,
75        network: Network,
76        lightning_port: u16,
77        alias: String,
78        mnemonic: Mnemonic,
79        runtime: Arc<tokio::runtime::Runtime>,
80    ) -> anyhow::Result<Self> {
81        let mut bytes = [0u8; 32];
82        let alias = if alias.is_empty() {
83            "LDK Gateway".to_string()
84        } else {
85            alias
86        };
87        let alias_bytes = alias.as_bytes();
88        let truncated = &alias_bytes[..alias_bytes.len().min(32)];
89        bytes[..truncated.len()].copy_from_slice(truncated);
90        let node_alias = Some(NodeAlias(bytes));
91
92        let mut node_builder = ldk_node::Builder::from_config(ldk_node::config::Config {
93            network,
94            listening_addresses: Some(vec![SocketAddress::TcpIpV4 {
95                addr: [0, 0, 0, 0],
96                port: lightning_port,
97            }]),
98            node_alias,
99            ..Default::default()
100        });
101
102        node_builder.set_entropy_bip39_mnemonic(mnemonic, None);
103
104        match chain_source.clone() {
105            ChainSource::Bitcoind {
106                username,
107                password,
108                server_url,
109            } => {
110                node_builder.set_chain_source_bitcoind_rpc(
111                    server_url
112                        .host_str()
113                        .expect("Could not retrieve host from bitcoind RPC url")
114                        .to_string(),
115                    server_url
116                        .port()
117                        .expect("Could not retrieve port from bitcoind RPC url"),
118                    username,
119                    password,
120                );
121            }
122            ChainSource::Esplora { server_url } => {
123                node_builder.set_chain_source_esplora(get_esplora_url(server_url)?, None);
124            }
125        };
126        let Some(data_dir_str) = data_dir.to_str() else {
127            return Err(anyhow::anyhow!("Invalid data dir path"));
128        };
129        node_builder.set_storage_dir_path(data_dir_str.to_string());
130
131        info!(chain_source = %chain_source, data_dir = %data_dir_str, alias = %alias, "Starting LDK Node...");
132        let node = Arc::new(node_builder.build()?);
133        node.start_with_runtime(runtime).map_err(|err| {
134            crit!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed to start LDK Node");
135            LightningRpcError::FailedToConnect
136        })?;
137
138        let (htlc_stream_sender, htlc_stream_receiver) = tokio::sync::mpsc::channel(1024);
139        let task_group = TaskGroup::new();
140
141        let node_clone = node.clone();
142        task_group.spawn("ldk lightning node event handler", |handle| async move {
143            loop {
144                Self::handle_next_event(&node_clone, &htlc_stream_sender, &handle).await;
145            }
146        });
147
148        info!("Successfully started LDK Gateway");
149        Ok(GatewayLdkClient {
150            node,
151            task_group,
152            htlc_stream_receiver_or: Some(htlc_stream_receiver),
153            outbound_lightning_payment_lock_pool: lockable::LockPool::new(),
154            outbound_offer_lock_pool: lockable::LockPool::new(),
155        })
156    }
157
158    async fn handle_next_event(
159        node: &ldk_node::Node,
160        htlc_stream_sender: &Sender<InterceptPaymentRequest>,
161        handle: &TaskHandle,
162    ) {
163        // We manually check for task termination in case we receive a payment while the
164        // task is shutting down. In that case, we want to finish the payment
165        // before shutting this task down.
166        let event = tokio::select! {
167            event = node.next_event_async() => {
168                event
169            }
170            () = handle.make_shutdown_rx() => {
171                return;
172            }
173        };
174
175        if let ldk_node::Event::PaymentClaimable {
176            payment_id: _,
177            payment_hash,
178            claimable_amount_msat,
179            claim_deadline,
180            ..
181        } = event
182            && let Err(err) = htlc_stream_sender
183                .send(InterceptPaymentRequest {
184                    payment_hash: Hash::from_slice(&payment_hash.0).expect("Failed to create Hash"),
185                    amount_msat: claimable_amount_msat,
186                    expiry: claim_deadline.unwrap_or_default(),
187                    short_channel_id: None,
188                    incoming_chan_id: 0,
189                    htlc_id: 0,
190                })
191                .await
192        {
193            warn!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed send InterceptHtlcRequest to stream");
194        }
195
196        // The `PaymentClaimable` event is the only event type that we are interested
197        // in. We can safely ignore all other events.
198        if let Err(err) = node.event_handled() {
199            warn!(err = %err.fmt_compact(), "LDK could not mark event handled");
200        }
201    }
202}

204impl Drop for GatewayLdkClient {
205    fn drop(&mut self) {
206        self.task_group.shutdown();
207
208        info!(target: LOG_LIGHTNING, "Stopping LDK Node...");
209        match self.node.stop() {
210            Err(err) => {
211                warn!(target: LOG_LIGHTNING, err = %err.fmt_compact(), "Failed to stop LDK Node");
212            }
213            _ => {
214                info!(target: LOG_LIGHTNING, "LDK Node stopped.");
215            }
216        }
217    }

261    async fn pay(
262        &self,
263        invoice: Bolt11Invoice,
264        max_delay: u64,
265        max_fee: Amount,
266    ) -> Result<PayInvoiceResponse, LightningRpcError> {
267        let payment_id = PaymentId(*invoice.payment_hash().as_byte_array());
268
269        // Lock by the payment hash to prevent multiple simultaneous calls with the same
270        // invoice from executing. This prevents `ldk-node::Bolt11Payment::send()` from
271        // being called multiple times with the same invoice. This is important because
272        // `ldk-node::Bolt11Payment::send()` is not idempotent, but this function must
273        // be idempotent.
274        let _payment_lock_guard = self
275            .outbound_lightning_payment_lock_pool
276            .async_lock(payment_id)
277            .await;
278
279        // If a payment is not known to the node we can initiate it, and if it is known
280        // we can skip calling `ldk-node::Bolt11Payment::send()` and wait for the
281        // payment to complete. The lock guard above guarantees that this block is only
282        // executed once at a time for a given payment hash, ensuring that there is no
283        // race condition between checking if a payment is known and initiating a new
284        // payment if it isn't.
285        if self.node.payment(&payment_id).is_none() {
286            assert_eq!(
287                self.node
288                    .bolt11_payment()
289                    .send(
290                        &invoice,
291                        Some(SendingParameters {
292                            max_total_routing_fee_msat: Some(Some(max_fee.msats)),
293                            max_total_cltv_expiry_delta: Some(max_delay as u32),
294                            max_path_count: None,
295                            max_channel_saturation_power_of_half: None,
296                        }),
297                    )
298                    // TODO: Investigate whether all error types returned by `Bolt11Payment::send()`
299                    // result in idempotency.
300                    .map_err(|e| LightningRpcError::FailedPayment {
301                        failure_reason: format!("LDK payment failed to initialize: {e:?}"),
302                    })?,
303                payment_id
304            );
305        }
306
307        // TODO: Find a way to avoid looping/polling to know when a payment is
308        // completed. `ldk-node` provides `PaymentSuccessful` and `PaymentFailed`
309        // events, but interacting with the node event queue here isn't
310        // straightforward.
311        loop {
312            if let Some(payment_details) = self.node.payment(&payment_id) {
313                match payment_details.status {
314                    PaymentStatus::Pending => {}
315                    PaymentStatus::Succeeded => {
316                        if let PaymentKind::Bolt11 {
317                            preimage: Some(preimage),
318                            ..
319                        } = payment_details.kind
320                        {
321                            return Ok(PayInvoiceResponse {
322                                preimage: Preimage(preimage.0),
323                            });
324                        }
325                    }
326                    PaymentStatus::Failed => {
327                        return Err(LightningRpcError::FailedPayment {
328                            failure_reason: "LDK payment failed".to_string(),
329                        });
330                    }
331                }
332            }
333            fedimint_core::runtime::sleep(Duration::from_millis(100)).await;
334        }
335    }

353    async fn complete_htlc(&self, htlc: InterceptPaymentResponse) -> Result<(), LightningRpcError> {
354        let InterceptPaymentResponse {
355            action,
356            payment_hash,
357            incoming_chan_id: _,
358            htlc_id: _,
359        } = htlc;
360
361        let ph = PaymentHash(*payment_hash.clone().as_byte_array());
362
363        // TODO: Get the actual amount from the LDK node. Probably makes the
364        // most sense to pipe it through the `InterceptHtlcResponse` struct.
365        // This value is only used by `ldk-node` to ensure that the amount
366        // claimed isn't less than the amount expected, but we've already
367        // verified that the amount is correct when we intercepted the payment.
368        let claimable_amount_msat = 999_999_999_999_999;
369
370        let ph_hex_str = hex::encode(payment_hash);
371
372        if let PaymentAction::Settle(preimage) = action {
373            self.node
374                .bolt11_payment()
375                .claim_for_hash(ph, claimable_amount_msat, PaymentPreimage(preimage.0))
376                .map_err(|_| LightningRpcError::FailedToCompleteHtlc {
377                    failure_reason: format!("Failed to claim LDK payment with hash {ph_hex_str}"),
378                })?;
379        } else {
380            warn!(target: LOG_LIGHTNING, payment_hash = %ph_hex_str, "Unwinding payment because the action was not `Settle`");
381            self.node.bolt11_payment().fail_for_hash(ph).map_err(|_| {
382                LightningRpcError::FailedToCompleteHtlc {
383                    failure_reason: format!("Failed to unwind LDK payment with hash {ph_hex_str}"),
384                }
385            })?;
386        }
387
388        return Ok(());
389    }

391    async fn create_invoice(
392        &self,
393        create_invoice_request: CreateInvoiceRequest,
394    ) -> Result<CreateInvoiceResponse, LightningRpcError> {
395        let payment_hash_or = if let Some(payment_hash) = create_invoice_request.payment_hash {
396            let ph = PaymentHash(*payment_hash.as_byte_array());
397            Some(ph)
398        } else {
399            None
400        };
401
402        let description = match create_invoice_request.description {
403            Some(InvoiceDescription::Direct(desc)) => {
404                Bolt11InvoiceDescription::Direct(Description::new(desc).map_err(|_| {
405                    LightningRpcError::FailedToGetInvoice {
406                        failure_reason: "Invalid description".to_string(),
407                    }
408                })?)
409            }
410            Some(InvoiceDescription::Hash(hash)) => {
411                Bolt11InvoiceDescription::Hash(lightning_invoice::Sha256(hash))
412            }
413            None => Bolt11InvoiceDescription::Direct(Description::empty()),
414        };
415
416        let invoice = match payment_hash_or {
417            Some(payment_hash) => self.node.bolt11_payment().receive_for_hash(
418                create_invoice_request.amount_msat,
419                &description,
420                create_invoice_request.expiry_secs,
421                payment_hash,
422            ),
423            None => self.node.bolt11_payment().receive(
424                create_invoice_request.amount_msat,
425                &description,
426                create_invoice_request.expiry_secs,
427            ),
428        }
429        .map_err(|e| LightningRpcError::FailedToGetInvoice {
430            failure_reason: e.to_string(),
431        })?;
432
433        Ok(CreateInvoiceResponse {
434            invoice: invoice.to_string(),
435        })
436    }

484    async fn open_channel(
485        &self,
486        OpenChannelRequest {
487            pubkey,
488            host,
489            channel_size_sats,
490            push_amount_sats,
491        }: OpenChannelRequest,
492    ) -> Result<OpenChannelResponse, LightningRpcError> {
493        let push_amount_msats_or = if push_amount_sats == 0 {
494            None
495        } else {
496            Some(push_amount_sats * 1000)
497        };
498
499        let user_channel_id = self
500            .node
501            .open_announced_channel(
502                pubkey,
503                SocketAddress::from_str(&host).map_err(|e| {
504                    LightningRpcError::FailedToConnectToPeer {
505                        failure_reason: e.to_string(),
506                    }
507                })?,
508                channel_size_sats,
509                push_amount_msats_or,
510                None,
511            )
512            .map_err(|e| LightningRpcError::FailedToOpenChannel {
513                failure_reason: e.to_string(),
514            })?;
515
516        // The channel isn't always visible immediately, so we need to poll for it.
517        for _ in 0..10 {
518            let funding_txid_or = self
519                .node
520                .list_channels()
521                .iter()
522                .find(|channel| channel.user_channel_id == user_channel_id)
523                .and_then(|channel| channel.funding_txo)
524                .map(|funding_txo| funding_txo.txid);
525
526            if let Some(funding_txid) = funding_txid_or {
527                return Ok(OpenChannelResponse {
528                    funding_txid: funding_txid.to_string(),
529                });
530            }
531
532            fedimint_core::runtime::sleep(Duration::from_millis(100)).await;
533        }
534
535        Err(LightningRpcError::FailedToOpenChannel {
536            failure_reason: "Channel could not be opened".to_string(),
537        })
538    }

540    async fn close_channels_with_peer(
541        &self,
542        CloseChannelsWithPeerRequest { pubkey, force }: CloseChannelsWithPeerRequest,
543    ) -> Result<CloseChannelsWithPeerResponse, LightningRpcError> {
544        let mut num_channels_closed = 0;
545
546        info!(%pubkey, "Closing all channels with peer");
547        for channel_with_peer in self
548            .node
549            .list_channels()
550            .iter()
551            .filter(|channel| channel.counterparty_node_id == pubkey)
552        {
553            if force {
554                match self.node.force_close_channel(
555                    &channel_with_peer.user_channel_id,
556                    pubkey,
557                    Some("User initiated force close".to_string()),
558                ) {
559                    Ok(()) => num_channels_closed += 1,
560                    Err(err) => {
561                        error!(%pubkey, err = %err.fmt_compact(), "Could not force close channel");
562                    }
563                }
564            } else {
565                match self
566                    .node
567                    .close_channel(&channel_with_peer.user_channel_id, pubkey)
568                {
569                    Ok(()) => {
570                        num_channels_closed += 1;
571                    }
572                    Err(err) => {
573                        error!(%pubkey, err = %err.fmt_compact(), "Could not close channel");
574                    }
575                }
576            }
577        }
578
579        Ok(CloseChannelsWithPeerResponse {
580            num_channels_closed,
581        })
582    }

621    async fn get_invoice(
622        &self,
623        get_invoice_request: GetInvoiceRequest,
624    ) -> Result<Option<GetInvoiceResponse>, LightningRpcError> {
625        let invoices = self
626            .node
627            .list_payments_with_filter(|details| {
628                details.direction == PaymentDirection::Inbound
629                    && details.id == PaymentId(get_invoice_request.payment_hash.to_byte_array())
630                    && !matches!(details.kind, PaymentKind::Onchain { .. })
631            })
632            .iter()
633            .map(|details| {
634                let (preimage, payment_hash, _) = get_preimage_and_payment_hash(&details.kind);
635                let status = match details.status {
636                    PaymentStatus::Failed => fedimint_gateway_common::PaymentStatus::Failed,
637                    PaymentStatus::Succeeded => fedimint_gateway_common::PaymentStatus::Succeeded,
638                    PaymentStatus::Pending => fedimint_gateway_common::PaymentStatus::Pending,
639                };
640                GetInvoiceResponse {
641                    preimage: preimage.map(|p| p.to_string()),
642                    payment_hash,
643                    amount: Amount::from_msats(
644                        details
645                            .amount_msat
646                            .expect("amountless invoices are not supported"),
647                    ),
648                    created_at: UNIX_EPOCH + Duration::from_secs(details.latest_update_timestamp),
649                    status,
650                }
651            })
652            .collect::<Vec<_>>();
653
654        Ok(invoices.first().cloned())
655    }
656
657    async fn list_transactions(
658        &self,
659        start_secs: u64,
660        end_secs: u64,
661    ) -> Result<ListTransactionsResponse, LightningRpcError> {
662        let transactions = self
663            .node
664            .list_payments_with_filter(|details| {
665                !matches!(details.kind, PaymentKind::Onchain { .. })
666                    && details.latest_update_timestamp >= start_secs
667                    && details.latest_update_timestamp < end_secs
668            })
669            .iter()
670            .map(|details| {
671                let (preimage, payment_hash, payment_kind) =
672                    get_preimage_and_payment_hash(&details.kind);
673                let direction = match details.direction {
674                    PaymentDirection::Outbound => {
675                        fedimint_gateway_common::PaymentDirection::Outbound
676                    }
677                    PaymentDirection::Inbound => fedimint_gateway_common::PaymentDirection::Inbound,
678                };
679                let status = match details.status {
680                    PaymentStatus::Failed => fedimint_gateway_common::PaymentStatus::Failed,
681                    PaymentStatus::Succeeded => fedimint_gateway_common::PaymentStatus::Succeeded,
682                    PaymentStatus::Pending => fedimint_gateway_common::PaymentStatus::Pending,
683                };
684                fedimint_gateway_common::PaymentDetails {
685                    payment_hash,
686                    preimage: preimage.map(|p| p.to_string()),
687                    payment_kind,
688                    amount: Amount::from_msats(
689                        details
690                            .amount_msat
691                            .expect("amountless invoices are not supported"),
692                    ),
693                    direction,
694                    status,
695                    timestamp_secs: details.latest_update_timestamp,
696                }
697            })
698            .collect::<Vec<_>>();
699        Ok(ListTransactionsResponse { transactions })
700    }

702    fn create_offer(
703        &self,
704        amount: Option<Amount>,
705        description: Option<String>,
706        expiry_secs: Option<u32>,
707        quantity: Option<u64>,
708    ) -> Result<String, LightningRpcError> {
709        let description = description.unwrap_or_default();
710        let offer = if let Some(amount) = amount {
711            self.node
712                .bolt12_payment()
713                .receive(amount.msats, &description, expiry_secs, quantity)
714                .map_err(|err| LightningRpcError::Bolt12Error {
715                    failure_reason: err.to_string(),
716                })?
717        } else {
718            self.node
719                .bolt12_payment()
720                .receive_variable_amount(&description, expiry_secs)
721                .map_err(|err| LightningRpcError::Bolt12Error {
722                    failure_reason: err.to_string(),
723                })?
724        };
725
726        Ok(offer.to_string())
727    }
728
729    async fn pay_offer(
730        &self,
731        offer: String,
732        quantity: Option<u64>,
733        amount: Option<Amount>,
734        payer_note: Option<String>,
735    ) -> Result<Preimage, LightningRpcError> {
736        let offer = Offer::from_str(&offer).map_err(|_| LightningRpcError::Bolt12Error {
737            failure_reason: "Failed to parse Bolt12 Offer".to_string(),
738        })?;
739
740        let _offer_lock_guard = self
741            .outbound_offer_lock_pool
742            .blocking_lock(LdkOfferId(offer.id()));
743
744        let payment_id = if let Some(amount) = amount {
745            self.node
746                .bolt12_payment()
747                .send_using_amount(&offer, amount.msats, quantity, payer_note)
748                .map_err(|err| LightningRpcError::Bolt12Error {
749                    failure_reason: err.to_string(),
750                })?
751        } else {
752            self.node
753                .bolt12_payment()
754                .send(&offer, quantity, payer_note)
755                .map_err(|err| LightningRpcError::Bolt12Error {
756                    failure_reason: err.to_string(),
757                })?
758        };
759
760        loop {
761            if let Some(payment_details) = self.node.payment(&payment_id) {
762                match payment_details.status {
763                    PaymentStatus::Pending => {}
764                    PaymentStatus::Succeeded => match payment_details.kind {
765                        PaymentKind::Bolt12Offer {
766                            preimage: Some(preimage),
767                            ..
768                        } => {
769                            info!(target: LOG_LIGHTNING, offer = %offer, payment_id = %payment_id, preimage = %preimage, "Successfully paid offer");
770                            return Ok(Preimage(preimage.0));
771                        }
772                        _ => {
773                            return Err(LightningRpcError::FailedPayment {
774                                failure_reason: "Unexpected payment kind".to_string(),
775                            });
776                        }
777                    },
778                    PaymentStatus::Failed => {
779                        return Err(LightningRpcError::FailedPayment {
780                            failure_reason: "Bolt12 payment failed".to_string(),
781                        });
782                    }
783                }
784            }
785            fedimint_core::runtime::sleep(Duration::from_millis(100)).await;
786        }
787    }
788}

790/// Maps LDK's `PaymentKind` to an optional preimage and an optional payment
791/// hash depending on the type of payment.
792fn get_preimage_and_payment_hash(
793    kind: &PaymentKind,
794) -> (
795    Option<Preimage>,
796    Option<sha256::Hash>,
797    fedimint_gateway_common::PaymentKind,
798) {
799    match kind {
800        PaymentKind::Bolt11 {
801            hash,
802            preimage,
803            secret: _,
804        } => (
805            preimage.map(|p| Preimage(p.0)),
806            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
807            fedimint_gateway_common::PaymentKind::Bolt11,
808        ),
809        PaymentKind::Bolt11Jit {
810            hash,
811            preimage,
812            secret: _,
813            lsp_fee_limits: _,
814            ..
815        } => (
816            preimage.map(|p| Preimage(p.0)),
817            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
818            fedimint_gateway_common::PaymentKind::Bolt11,
819        ),
820        PaymentKind::Bolt12Offer {
821            hash,
822            preimage,
823            secret: _,
824            offer_id: _,
825            payer_note: _,
826            quantity: _,
827        } => (
828            preimage.map(|p| Preimage(p.0)),
829            hash.map(|h| sha256::Hash::from_slice(&h.0).expect("Failed to convert payment hash")),
830            fedimint_gateway_common::PaymentKind::Bolt12Offer,
831        ),
832        PaymentKind::Bolt12Refund {
833            hash,
834            preimage,
835            secret: _,
836            payer_note: _,
837            quantity: _,
838        } => (
839            preimage.map(|p| Preimage(p.0)),
840            hash.map(|h| sha256::Hash::from_slice(&h.0).expect("Failed to convert payment hash")),
841            fedimint_gateway_common::PaymentKind::Bolt12Refund,
842        ),
843        PaymentKind::Spontaneous { hash, preimage } => (
844            preimage.map(|p| Preimage(p.0)),
845            Some(sha256::Hash::from_slice(&hash.0).expect("Failed to convert payment hash")),
846            fedimint_gateway_common::PaymentKind::Bolt11,
847        ),
848        PaymentKind::Onchain { .. } => (None, None, fedimint_gateway_common::PaymentKind::Onchain),
849    }
850}

852/// When a port is specified in the Esplora URL, the esplora client inside LDK
853/// node cannot connect to the lightning node when there is a trailing slash.
854/// The `SafeUrl::Display` function will always serialize the `SafeUrl` with a
855/// trailing slash, which causes the connection to fail.
856///
857/// To handle this, we explicitly construct the esplora URL when a port is
858/// specified.
859fn get_esplora_url(server_url: SafeUrl) -> anyhow::Result<String> {
860    // Esplora client cannot handle trailing slashes
861    let host = server_url
862        .host_str()
863        .ok_or(anyhow::anyhow!("Missing esplora host"))?;
864    let server_url = if let Some(port) = server_url.port() {
865        format!("{}://{}:{}", server_url.scheme(), host, port)
866    } else {
867        server_url.to_string()
868    };
869    Ok(server_url)
870}
