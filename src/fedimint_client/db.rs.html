40pub enum DbKeyPrefix {
41    EncodedClientSecret = 0x28,
42    ClientSecret = 0x29, // Unused
43    ClientPreRootSecretHash = 0x2a,
44    OperationLog = 0x2c,
45    ChronologicalOperationLog = 0x2d,
46    CommonApiVersionCache = 0x2e,
47    ClientConfig = 0x2f,
48    PendingClientConfig = 0x3b,
49    ClientInviteCode = 0x30, // Unused; clean out remnant data before re-using!
50    ClientInitState = 0x31,
51    ClientMetadata = 0x32,
52    ClientLastBackup = 0x33,
53    ClientMetaField = 0x34,
54    ClientMetaServiceInfo = 0x35,
55    ApiSecret = 0x36,
56    PeerLastApiVersionsSummaryCache = 0x37,
57    ApiUrlAnnouncement = 0x38,
58    EventLog = fedimint_eventlog::DB_KEY_PREFIX_EVENT_LOG,
59    UnorderedEventLog = fedimint_eventlog::DB_KEY_PREFIX_UNORDERED_EVENT_LOG,
60    EventLogTrimable = fedimint_eventlog::DB_KEY_PREFIX_EVENT_LOG_TRIMABLE,
61    ChainId = 0x3c,
62    ClientModuleRecovery = 0x40,
63
64    DatabaseVersion = fedimint_core::db::DbKeyPrefix::DatabaseVersion as u8,
65    ClientBackup = fedimint_core::db::DbKeyPrefix::ClientBackup as u8,
66
67    ActiveStates = ExecutorDbPrefixes::ActiveStates as u8,
68    InactiveStates = ExecutorDbPrefixes::InactiveStates as u8,
69
70    /// Arbitrary data of the applications integrating Fedimint client and
71    /// wanting to store some Federation-specific data in Fedimint client
72    /// database.
73    ///
74    /// New users are encouraged to use this single prefix only.
75    //
76    // TODO: https://github.com/fedimint/fedimint/issues/4444
77    //       in the future, we should make all global access to the db private
78    //       and only expose a getter returning isolated database.
79    UserData = 0xb0,
80    /// Prefixes between 0xb1..=0xcf shall all be considered allocated for
81    /// historical and future external use
82    ExternalReservedStart = 0xb1,
83    /// Prefixes between 0xb1..=0xcf shall all be considered allocated for
84    /// historical and future external use
85    ExternalReservedEnd = 0xcf,
86    /// 0xd0.. reserved for Fedimint internal use
87    // (see [`DbKeyPrefixInternalReserved`] for *internal* details)
88    InternalReservedStart = 0xd0,
89    /// Per-module instance data
90    ModuleGlobalPrefix = 0xff,
91}

100pub(crate) async fn verify_client_db_integrity_dbtx(dbtx: &mut DatabaseTransaction<'_>) {
101    let prefixes: BTreeSet<u8> = DbKeyPrefix::iter().map(|prefix| prefix as u8).collect();
102
103    let mut records = dbtx.raw_find_by_prefix(&[]).await.expect("DB fail");
104    while let Some((k, v)) = records.next().await {
105        // from here and above, we don't want to spend time verifying it
106        if DbKeyPrefix::UserData as u8 <= k[0] {
107            break;
108        }
109
110        assert!(
111            prefixes.contains(&k[0]),
112            "Unexpected client db record found: {}: {}",
113            k.as_hex(),
114            v.as_hex()
115        );
116    }
117}

313/// Client initialization mode
314#[derive(Debug, Encodable, Decodable)]
315pub enum InitMode {
316    /// Should only be used with freshly generated root secret
317    Fresh,
318    /// Should be used with root secrets provided by the user to recover a
319    /// (even if just possibly) already used secret.
320    Recover { snapshot: Option<ClientBackup> },
321}
322
323/// Like `InitMode`, but without no longer required data.
324///
325/// This is distinct from `InitMode` to prevent holding on to `snapshot`
326/// forever both for user's privacy and space use. In case user get hacked
327/// or phone gets stolen.
328#[derive(Debug, Encodable, Decodable)]
329pub enum InitModeComplete {
330    Fresh,
331    Recover,
332}
333
334/// The state of the client initialization
335#[derive(Debug, Encodable, Decodable)]
336pub enum InitState {
337    /// Client data initialization might still require some work (e.g. client
338    /// recovery)
339    Pending(InitMode),
340    /// Client initialization was complete
341    Complete(InitModeComplete),
342}
343
344impl InitState {
345    pub fn into_complete(self) -> Self {
346        match self {
347            InitState::Pending(p) => InitState::Complete(match p {
348                InitMode::Fresh => InitModeComplete::Fresh,
349                InitMode::Recover { .. } => InitModeComplete::Recover,
350            }),
351            InitState::Complete(t) => InitState::Complete(t),
352        }
353    }
354
355    pub fn does_require_recovery(&self) -> Option<Option<ClientBackup>> {
356        match self {
357            InitState::Pending(p) => match p {
358                InitMode::Fresh => None,
359                InitMode::Recover { snapshot } => Some(snapshot.clone()),
360            },
361            InitState::Complete(_) => None,
362        }
363    }
364
365    pub fn is_pending(&self) -> bool {
366        match self {
367            InitState::Pending(_) => true,
368            InitState::Complete(_) => false,
369        }
370    }
371}

471pub fn get_core_client_database_migrations()
472-> BTreeMap<DatabaseVersion, fedimint_core::db::ClientCoreDbMigrationFn> {
473    let mut migrations: BTreeMap<DatabaseVersion, fedimint_core::db::ClientCoreDbMigrationFn> =
474        BTreeMap::new();
475    migrations.insert(
476        DatabaseVersion(0),
477        Box::new(|mut ctx| {
478            Box::pin(async move {
479                let mut dbtx = ctx.dbtx();
480
481                let config_v0 = dbtx
482                    .find_by_prefix(&ClientConfigKeyPrefixV0)
483                    .await
484                    .collect::<Vec<_>>()
485                    .await;
486
487                assert!(config_v0.len() <= 1);
488                let Some((id, config_v0)) = config_v0.into_iter().next() else {
489                    return Ok(());
490                };
491
492                let global = GlobalClientConfig {
493                    api_endpoints: config_v0.global.api_endpoints,
494                    broadcast_public_keys: None,
495                    consensus_version: config_v0.global.consensus_version,
496                    meta: config_v0.global.meta,
497                };
498
499                let config = ClientConfig {
500                    global,
501                    modules: config_v0.modules,
502                };
503
504                dbtx.remove_entry(&id).await;
505                dbtx.insert_new_entry(&ClientConfigKey, &config).await;
506                Ok(())
507            })
508        }),
509    );
510
511    // Migration to add outcome_time to OperationLogEntry
512    migrations.insert(
513        DatabaseVersion(1),
514        Box::new(|mut ctx| {
515            Box::pin(async move {
516                let mut dbtx = ctx.dbtx();
517
518                // Read all OperationLogEntries using V0 format
519                let operation_logs = dbtx
520                    .find_by_prefix(&OperationLogKeyPrefixV0)
521                    .await
522                    .collect::<Vec<_>>()
523                    .await;
524
525                // Build a map from operation_id -> max_time of inactive state
526                let mut op_id_max_time = BTreeMap::new();
527
528                // Process inactive states
529                {
530                    let mut inactive_states_stream =
531                        dbtx.find_by_prefix(&InactiveStateKeyPrefixBytes).await;
532
533                    while let Some((state, meta)) = inactive_states_stream.next().await {
534                        let entry = op_id_max_time
535                            .entry(state.operation_id)
536                            .or_insert(meta.exited_at);
537                        *entry = (*entry).max(meta.exited_at);
538                    }
539                }
540                // Migrate each V0 operation log entry to the new format
541                for (op_key_v0, log_entry_v0) in operation_logs {
542                    let new_entry = OperationLogEntry::new(
543                        log_entry_v0.operation_module_kind,
544                        log_entry_v0.meta,
545                        log_entry_v0.outcome.map(|outcome| {
546                            OperationOutcome {
547                                outcome,
548                                // If we found state times, use the max, otherwise use
549                                // current time
550                                time: op_id_max_time
551                                    .get(&op_key_v0.operation_id)
552                                    .copied()
553                                    .unwrap_or_else(fedimint_core::time::now),
554                            }
555                        }),
556                    );
557
558                    dbtx.remove_entry(&op_key_v0).await;
559                    dbtx.insert_entry(
560                        &OperationLogKey {
561                            operation_id: op_key_v0.operation_id,
562                        },
563                        &new_entry,
564                    )
565                    .await;
566                }
567
568                Ok(())
569            })
570        }),
571    );
572
573    // Fix #6948
574    migrations.insert(
575        DatabaseVersion(2),
576        Box::new(|mut ctx: fedimint_core::db::DbMigrationFnContext<'_, _>| {
577            Box::pin(async move {
578                let mut dbtx = ctx.dbtx();
579
580                // Migrate unordered keys that got written to ordered table
581                {
582                    let mut ordered_log_entries = dbtx
583                        .raw_find_by_prefix(&[DB_KEY_PREFIX_EVENT_LOG])
584                        .await
585                        .expect("DB operation failed");
586                    let mut keys_to_migrate = vec![];
587                    while let Some((k, _v)) = ordered_log_entries.next().await {
588                        trace!(target: LOG_CLIENT_DB,
589                            k=%k.as_hex(),
590                            "Checking ordered log key"
591                        );
592                        if EventLogId::consensus_decode_whole(&k[1..], &Default::default()).is_err()
593                        {
594                            assert!(
595                                UnordedEventLogId::consensus_decode_whole(
596                                    &k[1..],
597                                    &Default::default()
598                                )
599                                .is_ok()
600                            );
601                            keys_to_migrate.push(k);
602                        }
603                    }
604                    drop(ordered_log_entries);
605                    for mut key_to_migrate in keys_to_migrate {
606                        warn!(target: LOG_CLIENT_DB,
607                            k=%key_to_migrate.as_hex(),
608                            "Migrating unordered event log entry written to an ordered log"
609                        );
610                        let v = dbtx
611                            .raw_remove_entry(&key_to_migrate)
612                            .await
613                            .expect("DB operation failed")
614                            .expect("Was there a moment ago");
615                        assert_eq!(key_to_migrate[0], 0x39);
616                        key_to_migrate[0] = DB_KEY_PREFIX_UNORDERED_EVENT_LOG;
617                        assert_eq!(key_to_migrate[0], 0x3a);
618                        dbtx.raw_insert_bytes(&key_to_migrate, &v)
619                            .await
620                            .expect("DB operation failed");
621                    }
622                }
623
624                // Migrate ordered keys that got written to unordered table
625                {
626                    let mut unordered_log_entries = dbtx
627                        .raw_find_by_prefix(&[DB_KEY_PREFIX_UNORDERED_EVENT_LOG])
628                        .await
629                        .expect("DB operation failed");
630                    let mut keys_to_migrate = vec![];
631                    while let Some((k, _v)) = unordered_log_entries.next().await {
632                        trace!(target: LOG_CLIENT_DB,
633                            k=%k.as_hex(),
634                            "Checking unordered log key"
635                        );
636                        if UnordedEventLogId::consensus_decode_whole(&k[1..], &Default::default())
637                            .is_err()
638                        {
639                            assert!(
640                                EventLogId::consensus_decode_whole(&k[1..], &Default::default())
641                                    .is_ok()
642                            );
643                            keys_to_migrate.push(k);
644                        }
645                    }
646                    drop(unordered_log_entries);
647                    for mut key_to_migrate in keys_to_migrate {
648                        warn!(target: LOG_CLIENT_DB,
649                            k=%key_to_migrate.as_hex(),
650                            "Migrating ordered event log entry written to an unordered log"
651                        );
652                        let v = dbtx
653                            .raw_remove_entry(&key_to_migrate)
654                            .await
655                            .expect("DB operation failed")
656                            .expect("Was there a moment ago");
657                        assert_eq!(key_to_migrate[0], 0x3a);
658                        key_to_migrate[0] = DB_KEY_PREFIX_EVENT_LOG;
659                        assert_eq!(key_to_migrate[0], 0x39);
660                        dbtx.raw_insert_bytes(&key_to_migrate, &v)
661                            .await
662                            .expect("DB operation failed");
663                    }
664                }
665                Ok(())
666            })
667        }),
668    );
669
670    // Fix #7367
671    migrations.insert(
672        DatabaseVersion(3),
673        Box::new(|mut ctx: fedimint_core::db::DbMigrationFnContext<'_, _>| {
674            Box::pin(async move {
675                let mut dbtx = ctx.dbtx();
676
677                for module_id in 0..u16::MAX {
678                    let old_key = ClientModuleRecoveryIncorrectDoNotUse {
679                        module_instance_id: module_id,
680                    };
681                    let new_key = ClientModuleRecovery {
682                        module_instance_id: module_id,
683                    };
684                    let Some(value) = dbtx.get_value(&old_key).await else {
685                        debug!(target: LOG_CLIENT_DB, %module_id, "No more ClientModuleRecovery keys found for migartion");
686                        break;
687                    };
688
689                    debug!(target: LOG_CLIENT_DB, %module_id, "Migrating old ClientModuleRecovery key");
690                    dbtx.remove_entry(&old_key).await.expect("Is there.");
691                    assert!(dbtx.insert_entry(&new_key, &value).await.is_none());
692                }
693
694                Ok(())
695            })
696        }),
697    );
698    migrations
699}

719/// `apply_migrations_client` iterates from the on disk database version for the
720/// client module up to `target_db_version` and executes all of the migrations
721/// that exist in the migrations map, including state machine migrations.
722/// Each migration in the migrations map updates the database to have the
723/// correct on-disk data structures that the code is expecting. The entire
724/// process is atomic, (i.e migration from 0->1 and 1->2 happen atomically).
725/// This function is called before the module is initialized and as long as the
726/// correct migrations are supplied in the migrations map, the module
727/// will be able to read and write from the database successfully.
728pub async fn apply_migrations_client_module(
729    db: &Database,
730    kind: String,
731    migrations: BTreeMap<DatabaseVersion, ClientModuleMigrationFn>,
732    module_instance_id: ModuleInstanceId,
733) -> Result<(), anyhow::Error> {
734    let mut dbtx = db.begin_transaction().await;
735    apply_migrations_client_module_dbtx(
736        &mut dbtx.to_ref_nc(),
737        kind,
738        migrations,
739        module_instance_id,
740    )
741    .await?;
742    dbtx.commit_tx_result()
743        .await
744        .map_err(|e| anyhow::Error::msg(e.to_string()))
745}

747pub async fn apply_migrations_client_module_dbtx(
748    dbtx: &mut DatabaseTransaction<'_>,
749    kind: String,
750    migrations: BTreeMap<DatabaseVersion, ClientModuleMigrationFn>,
751    module_instance_id: ModuleInstanceId,
752) -> Result<(), anyhow::Error> {
753    // Newly created databases will not have any data underneath the
754    // `MODULE_GLOBAL_PREFIX` since they have just been instantiated.
755    let is_new_db = dbtx
756        .raw_find_by_prefix(&[MODULE_GLOBAL_PREFIX])
757        .await?
758        .next()
759        .await
760        .is_none();
761
762    let target_version = get_current_database_version(&migrations);
763
764    // First write the database version to disk if it does not exist.
765    create_database_version_dbtx(
766        dbtx,
767        target_version,
768        Some(module_instance_id),
769        kind.clone(),
770        is_new_db,
771    )
772    .await?;
773
774    let current_version = dbtx
775        .get_value(&DatabaseVersionKey(module_instance_id))
776        .await;
777
778    let db_version = if let Some(mut current_version) = current_version {
779        if current_version == target_version {
780            trace!(
781                target: LOG_CLIENT_DB,
782                %current_version,
783                %target_version,
784                module_instance_id,
785                kind,
786                "Database version up to date"
787            );
788            return Ok(());
789        }
790
791        if target_version < current_version {
792            return Err(anyhow!(format!(
793                "On disk database version for module {kind} was higher ({}) than the target database version ({}).",
794                current_version, target_version,
795            )));
796        }
797
798        info!(
799            target: LOG_CLIENT_DB,
800            %current_version,
801            %target_version,
802            module_instance_id,
803            kind,
804            "Migrating client module database"
805        );
806        let mut active_states = get_active_states(&mut dbtx.to_ref_nc(), module_instance_id).await;
807        let mut inactive_states =
808            get_inactive_states(&mut dbtx.to_ref_nc(), module_instance_id).await;
809
810        while current_version < target_version {
811            let new_states = if let Some(migration) = migrations.get(&current_version) {
812                debug!(
813                     target: LOG_CLIENT_DB,
814                     module_instance_id,
815                     %kind,
816                     %current_version,
817                     %target_version,
818                     "Running module db migration");
819
820                migration(
821                    &mut dbtx
822                        .to_ref_with_prefix_module_id(module_instance_id)
823                        .0
824                        .into_nc(),
825                    active_states.clone(),
826                    inactive_states.clone(),
827                )
828                .await?
829            } else {
830                warn!(
831                    target: LOG_CLIENT_DB,
832                    ?current_version, "Missing client db migration");
833                None
834            };
835
836            // If the client migration returned new states, a state machine migration has
837            // occurred, and the new states need to be persisted to the database.
838            if let Some((new_active_states, new_inactive_states)) = new_states {
839                remove_old_and_persist_new_active_states(
840                    &mut dbtx.to_ref_nc(),
841                    new_active_states.clone(),
842                    active_states.clone(),
843                    module_instance_id,
844                )
845                .await;
846                remove_old_and_persist_new_inactive_states(
847                    &mut dbtx.to_ref_nc(),
848                    new_inactive_states.clone(),
849                    inactive_states.clone(),
850                    module_instance_id,
851                )
852                .await;
853
854                // the new states become the old states for the next migration
855                active_states = new_active_states;
856                inactive_states = new_inactive_states;
857            }
858
859            current_version = current_version.increment();
860            dbtx.insert_entry(&DatabaseVersionKey(module_instance_id), &current_version)
861                .await;
862        }
863
864        current_version
865    } else {
866        target_version
867    };
868
869    debug!(
870        target: LOG_CLIENT_DB,
871        ?kind, ?db_version, "Client DB Version");
872    Ok(())
873}

875/// Reads all active states from the database and returns `Vec<DynState>`.
876/// TODO: It is unfortunate that we can't read states by the module's instance
877/// id so we are forced to return all active states. Once we do a db migration
878/// to add `module_instance_id` to `ActiveStateKey`, this can be improved to
879/// only read the module's relevant states.
880pub async fn get_active_states(
881    dbtx: &mut DatabaseTransaction<'_>,
882    module_instance_id: ModuleInstanceId,
883) -> Vec<(Vec<u8>, OperationId)> {
884    dbtx.find_by_prefix(&ActiveStateKeyPrefixBytes)
885        .await
886        .filter_map(|(state, _)| async move {
887            if module_instance_id == state.module_instance_id {
888                Some((state.state, state.operation_id))
889            } else {
890                None
891            }
892        })
893        .collect::<Vec<_>>()
894        .await
895}
896
897/// Reads all inactive states from the database and returns `Vec<DynState>`.
898/// TODO: It is unfortunate that we can't read states by the module's instance
899/// id so we are forced to return all inactive states. Once we do a db migration
900/// to add `module_instance_id` to `InactiveStateKey`, this can be improved to
901/// only read the module's relevant states.
902pub async fn get_inactive_states(
903    dbtx: &mut DatabaseTransaction<'_>,
904    module_instance_id: ModuleInstanceId,
905) -> Vec<(Vec<u8>, OperationId)> {
906    dbtx.find_by_prefix(&InactiveStateKeyPrefixBytes)
907        .await
908        .filter_map(|(state, _)| async move {
909            if module_instance_id == state.module_instance_id {
910                Some((state.state, state.operation_id))
911            } else {
912                None
913            }
914        })
915        .collect::<Vec<_>>()
916        .await
917}

919/// Persists new active states by first removing all current active states, and
920/// re-writing with the new set of active states. `new_active_states` is
921/// expected to contain all active states, not just the newly created states.
922pub async fn remove_old_and_persist_new_active_states(
923    dbtx: &mut DatabaseTransaction<'_>,
924    new_active_states: Vec<(Vec<u8>, OperationId)>,
925    states_to_remove: Vec<(Vec<u8>, OperationId)>,
926    module_instance_id: ModuleInstanceId,
927) {
928    // Remove all existing active states
929    for (bytes, operation_id) in states_to_remove {
930        dbtx.remove_entry(&ActiveStateKeyBytes {
931            operation_id,
932            module_instance_id,
933            state: bytes,
934        })
935        .await
936        .expect("Did not delete anything");
937    }
938
939    // Insert new "migrated" active states
940    for (bytes, operation_id) in new_active_states {
941        dbtx.insert_new_entry(
942            &ActiveStateKeyBytes {
943                operation_id,
944                module_instance_id,
945                state: bytes,
946            },
947            &ActiveStateMeta::default(),
948        )
949        .await;
950    }
951}

953/// Persists new inactive states by first removing all current inactive states,
954/// and re-writing with the new set of inactive states. `new_inactive_states` is
955/// expected to contain all inactive states, not just the newly created states.
956pub async fn remove_old_and_persist_new_inactive_states(
957    dbtx: &mut DatabaseTransaction<'_>,
958    new_inactive_states: Vec<(Vec<u8>, OperationId)>,
959    states_to_remove: Vec<(Vec<u8>, OperationId)>,
960    module_instance_id: ModuleInstanceId,
961) {
962    // Remove all existing active states
963    for (bytes, operation_id) in states_to_remove {
964        dbtx.remove_entry(&InactiveStateKeyBytes {
965            operation_id,
966            module_instance_id,
967            state: bytes,
968        })
969        .await
970        .expect("Did not delete anything");
971    }
972
973    // Insert new "migrated" inactive states
974    for (bytes, operation_id) in new_inactive_states {
975        dbtx.insert_new_entry(
976            &InactiveStateKeyBytes {
977                operation_id,
978                module_instance_id,
979                state: bytes,
980            },
981            &InactiveStateMeta {
982                created_at: fedimint_core::time::now(),
983                exited_at: fedimint_core::time::now(),
984            },
985        )
986        .await;
987    }

990/// Fetches the encoded client secret from the database and decodes it.
991/// If an encoded client secret is not present in the database, or if
992/// decoding fails, an error is returned.
993pub async fn get_decoded_client_secret<T: Decodable>(db: &Database) -> anyhow::Result<T> {
994    let mut tx = db.begin_transaction_nc().await;
995    let client_secret = tx.get_value(&EncodedClientSecretKey).await;
996
997    match client_secret {
998        Some(client_secret) => {
999            T::consensus_decode_whole(&client_secret, &ModuleRegistry::default())
1000                .map_err(|e| anyhow!("Decoding failed: {e}"))
1001        }
1002        None => bail!("Encoded client secret not present in DB"),
1003    }
1004}
