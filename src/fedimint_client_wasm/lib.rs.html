95    /// Parse a bolt11 invoice and extract its components
96    /// without joining the federation
97    pub fn parse_bolt11_invoice(invoice_str: &str) -> Result<String, JsError> {
98        let invoice = lightning_invoice::Bolt11Invoice::from_str(invoice_str)
99            .map_err(|e| JsError::new(&format!("Failed to parse Lightning invoice: {}", e)))?;
100
101        let amount_msat = invoice.amount_milli_satoshis().unwrap_or(0);
102        let amount_sat = amount_msat as f64 / 1000.0;
103
104        let expiry_seconds = invoice.expiry_time().as_secs();
105
106        // memo
107        let description = match invoice.description() {
108            Bolt11InvoiceDescriptionRef::Direct(desc) => desc.to_string(),
109            Bolt11InvoiceDescriptionRef::Hash(_) => "Description hash only".to_string(),
110        };
111
112        let response = json!({
113            "amount": amount_sat,
114            "expiry": expiry_seconds,
115            "memo": description,
116        });
117        Ok(serde_json::to_string(&response).map_err(|e| JsError::new(&e.to_string()))?)
118    }

141    async fn open_inner(client_name: String) -> anyhow::Result<Option<WasmClient>> {
142        let db = Database::from(MemAndIndexedDb::new(&client_name).await?);
143        if !fedimint_client::Client::is_initialized(&db).await {
144            return Ok(None);
145        }
146        let client_secret = fedimint_client::Client::load_or_generate_client_secret(&db).await?;
147        let root_secret = PlainRootSecretStrategy::to_root_secret(&client_secret);
148        let builder = Self::client_builder(db).await?;
149        Ok(Some(Self {
150            client: Arc::new(builder.open(root_secret).await?),
151        }))
152    }

154    async fn join_federation_inner(
155        client_name: String,
156        invite_code: String,
157    ) -> anyhow::Result<WasmClient> {
158        let db = Database::from(MemAndIndexedDb::new(&client_name).await?);
159        let client_secret = fedimint_client::Client::load_or_generate_client_secret(&db).await?;
160        let root_secret = PlainRootSecretStrategy::to_root_secret(&client_secret);
161        let builder = Self::client_builder(db).await?;
162        let invite_code = InviteCode::from_str(&invite_code)?;
163        let config = fedimint_api_client::api::net::Connector::default()
164            .download_from_invite_code(&invite_code)
165            .await?;
166        let client = Arc::new(builder.join(root_secret, config, None).await?);
167        Ok(Self { client })
168    }

170    #[wasm_bindgen]
171    /// Call a fedimint client rpc the responses are returned using `cb`
172    /// callback. Each rpc call *can* return multiple responses by calling
173    /// `cb` multiple times. The returned RpcHandle can be used to cancel the
174    /// operation.
175    pub fn rpc(
176        &self,
177        module: &str,
178        method: &str,
179        payload: String,
180        cb: &js_sys::Function,
181    ) -> RpcHandle {
182        let (abort_handle, abort_registration) = AbortHandle::new_pair();
183        let rpc_handle = RpcHandle { abort_handle };
184
185        let client = self.client.clone();
186        let module = module.to_string();
187        let method = method.to_string();
188        let cb = cb.clone();
189
190        wasm_bindgen_futures::spawn_local(async move {
191            let future = async {
192                let mut stream = pin!(Self::rpc_inner(&client, &module, &method, payload));
193
194                while let Some(item) = stream.next().await {
195                    let this = JsValue::null();
196                    let _ = match item {
197                        Ok(item) => cb.call1(
198                            &this,
199                            &JsValue::from_str(
200                                &serde_json::to_string(&json!({"data": item})).unwrap(),
201                            ),
202                        ),
203                        Err(err) => cb.call1(
204                            &this,
205                            &JsValue::from_str(
206                                &serde_json::to_string(&json!({"error": err.to_string()})).unwrap(),
207                            ),
208                        ),
209                    };
210                }
211
212                // Send the end message
213                let _ = cb.call1(
214                    &JsValue::null(),
215                    &JsValue::from_str(&serde_json::to_string(&json!({"end": null})).unwrap()),
216                );
217            };
218
219            let abortable_future = Abortable::new(future, abort_registration);
220            let _ = abortable_future.await;
221        });
222
223        rpc_handle
224    }
225    fn rpc_inner<'a>(
226        client: &'a ClientHandleArc,
227        module: &'a str,
228        method: &'a str,
229        payload: String,
230    ) -> impl futures::Stream<Item = anyhow::Result<serde_json::Value>> + 'a {
231        try_stream! {
232            let payload: serde_json::Value = serde_json::from_str(&payload)?;
233            match module {
234                "" => {
235                    let mut stream = client.handle_global_rpc(method.to_owned(), payload);
236                    while let Some(item) = stream.next().await {
237                        yield item?;
238                    }
239                }
240                "ln" => {
241                    let ln = client
242                        .get_first_module::<LightningClientModule>()?
243                        .inner();
244                    let mut stream = ln.handle_rpc(method.to_owned(), payload).await;
245                    while let Some(item) = stream.next().await {
246                        yield item?;
247                    }
248                }
249                "mint" => {
250                    let mint = client
251                        .get_first_module::<fedimint_mint_client::MintClientModule>()?
252                        .inner();
253                    let mut stream = mint.handle_rpc(method.to_owned(), payload).await;
254                    while let Some(item) = stream.next().await {
255                        yield item?;
256                    }
257                }
258                "wallet" => {
259                    let wallet = client
260                        .get_first_module::<WalletClientModule>()?
261                        .inner();
262                    let mut stream = wallet.handle_rpc(method.to_owned(), payload).await;
263                    while let Some(item) = stream.next().await {
264                        yield item?;
265                    }
266                }
267                _ => {
268                    Err(anyhow::format_err!("module not found: {module}"))?;
269                    unreachable!()
270                },
271            }
272        }
273    }
