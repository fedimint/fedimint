5//! Constants for categorizing the logging type
6//!
7//! To help stabilize logging targets, avoid typos and improve consistency,
8//! it's preferable for logging statements use static target constants,
9//! that we define in this module.
10//!
11//! Core + server side components should use global namespace,
12//! while client should generally be prefixed with `client::`.
13//! This makes it easier to filter interesting calls when
14//! running e.g. `devimint`, that will run both server and client
15//! side.

63/// Consolidates the setup of server tracing into a helper
64#[derive(Default)]
65pub struct TracingSetup {
66    base_level: Option<String>,
67    extra_directives: Option<String>,
68    #[cfg(feature = "telemetry")]
69    tokio_console_bind: Option<std::net::SocketAddr>,
70    #[cfg(feature = "telemetry")]
71    with_jaeger: bool,
72    with_file: Option<File>,
73}
74
75impl TracingSetup {
76    /// Setup a console server for tokio logging <https://docs.rs/console-subscriber>
77    #[cfg(feature = "telemetry")]
78    pub fn tokio_console_bind(&mut self, address: Option<std::net::SocketAddr>) -> &mut Self {
79        self.tokio_console_bind = address;
80        self
81    }
82
83    /// Setup telemetry through Jaeger <https://docs.rs/tracing-jaeger>
84    #[cfg(feature = "telemetry")]
85    pub fn with_jaeger(&mut self, enabled: bool) -> &mut Self {
86        self.with_jaeger = enabled;
87        self
88    }
89
90    pub fn with_file(&mut self, file: Option<File>) -> &mut Self {
91        self.with_file = file;
92        self
93    }
94
95    /// Sets the log level applied to most modules. Some overly chatty modules
96    /// are muted even if this is set to a lower log level, use the `RUST_LOG`
97    /// environment variable to override.
98    pub fn with_base_level(&mut self, level: impl Into<String>) -> &mut Self {
99        self.base_level = Some(level.into());
100        self
101    }
102
103    /// Add a filter directive.
104    pub fn with_directive(&mut self, directive: &str) -> &mut Self {
105        if let Some(old) = self.extra_directives.as_mut() {
106            *old = format!("{old},{directive}");
107        } else {
108            self.extra_directives = Some(directive.to_owned());
109        }
110        self
111    }
112
113    /// Initialize the logging, must be called for tracing to begin
114    pub fn init(&mut self) -> anyhow::Result<()> {
115        use tracing_subscriber::fmt::writer::{BoxMakeWriter, Tee};
116
117        let var = env::var(tracing_subscriber::EnvFilter::DEFAULT_ENV).unwrap_or_default();
118        let filter_layer = EnvFilter::builder().parse(format!(
119            // We prefix everything with a default general log level and
120            // good per-module specific default. User provided RUST_LOG
121            // can override one or both
122            "{},{},{},{},{},{},{},{}",
123            self.base_level.as_deref().unwrap_or("info"),
124            "jsonrpsee_core::client::async_client=off",
125            "hyper=off",
126            "h2=off",
127            "jsonrpsee_server=warn,jsonrpsee_server::transport=off",
128            "AlephBFT-=error",
129            var,
130            self.extra_directives.as_deref().unwrap_or(""),
131        ))?;
132
133        let fmt_writer = match self.with_file.take() {
134            Some(file) => BoxMakeWriter::new(Tee::new(io::stderr, file)),
135            _ => BoxMakeWriter::new(io::stderr),
136        };
137
138        let fmt_layer = tracing_subscriber::fmt::layer()
139            .with_thread_names(false) // can be enabled for debugging
140            .with_writer(fmt_writer)
141            .with_filter(filter_layer);
142
143        let console_opt = || -> Option<Box<dyn Layer<_> + Send + Sync + 'static>> {
144            #[cfg(feature = "telemetry")]
145            if let Some(l) = self.tokio_console_bind {
146                let tracer = console_subscriber::ConsoleLayer::builder()
147                    .retention(std::time::Duration::from_secs(60))
148                    .server_addr(l)
149                    .spawn()
150                    // tokio-console cares only about these layers, so we filter separately for it
151                    .with_filter(EnvFilter::new("tokio=trace,runtime=trace"));
152                return Some(tracer.boxed());
153            }
154            None
155        };
156
157        let telemetry_layer_opt = || -> Option<Box<dyn Layer<_> + Send + Sync + 'static>> {
158            #[cfg(feature = "telemetry")]
159            if self.with_jaeger {
160                // TODO: https://github.com/fedimint/fedimint/issues/4591
161                #[allow(deprecated)]
162                let tracer = opentelemetry_jaeger::new_agent_pipeline()
163                    .with_service_name("fedimint")
164                    .install_simple()
165                    .unwrap();
166
167                return Some(tracing_opentelemetry::layer().with_tracer(tracer).boxed());
168            }
169            None
170        };
171
172        tracing_subscriber::registry()
173            .with(fmt_layer)
174            .with(console_opt())
175            .with(telemetry_layer_opt())
176            .try_init()?;
177        Ok(())
178    }
179}
180
181pub fn shutdown() {
182    #[cfg(feature = "telemetry")]
183    opentelemetry::global::shutdown_tracer_provider();
