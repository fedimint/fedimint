55    /// Workaround for a serde `flatten` quirk
56    ///
57    /// We serialize config with no fields as: eg. `{ kind: "ln" }`.
58    ///
59    /// When `kind` gets removed and `value` is parsed, it will
60    /// parse as `Value::Object` that is empty.
61    ///
62    /// However empty module structs, like `struct FooConfigLocal;` (unit
63    /// struct), will fail to deserialize with this value, as they expect
64    /// `Value::Null`.
65    ///
66    /// We can turn manually empty object into null, and that's what
67    /// we do in this function. This fixes the deserialization into
68    /// unit type, but in turn breaks deserialization into `struct Foo{}`,
69    /// which is arguably much less common, but valid.
70    ///
71    /// TODO: In the future, we should have a typed and erased versions of
72    /// module construction traits, and then we can try with and
73    /// without the workaround to have both cases working.
74    /// See <https://github.com/fedimint/fedimint/issues/1303>
75    pub fn with_fixed_empty_value(self) -> Self {
76        if let serde_json::Value::Object(ref o) = self.value
77            && o.is_empty()
78        {
79            return Self {
80                kind: self.kind,
81                value: serde_json::Value::Null,
82            };
83        }
84
85        self
86    }
87
88    pub fn value(&self) -> &serde_json::Value {
89        &self.value
90    }
91
92    pub fn kind(&self) -> &ModuleKind {
93        &self.kind
94    }
95
96    pub fn is_kind(&self, kind: &ModuleKind) -> bool {
97        &self.kind == kind
98    }
99}

132fn de_int_key<'de, D, K, V>(deserializer: D) -> Result<BTreeMap<K, V>, D::Error>
133where
134    D: Deserializer<'de>,
135    K: Eq + Ord + FromStr,
136    K::Err: Display,
137    V: Deserialize<'de>,
138{
139    let string_map = <BTreeMap<String, V>>::deserialize(deserializer)?;
140    let map = string_map
141        .into_iter()
142        .map(|(key_str, value)| {
143            let key = K::from_str(&key_str).map_err(serde::de::Error::custom)?;
144            Ok((key, value))
145        })
146        .collect::<Result<BTreeMap<_, _>, _>>()?;
147    Ok(map)
148}
149
150fn optional_de_int_key<'de, D, K, V>(deserializer: D) -> Result<Option<BTreeMap<K, V>>, D::Error>
151where
152    D: Deserializer<'de>,
153    K: Eq + Ord + FromStr,
154    K::Err: Display,
155    V: Deserialize<'de>,
156{
157    let Some(string_map) = <Option<BTreeMap<String, V>>>::deserialize(deserializer)? else {
158        return Ok(None);
159    };
160
161    let map = string_map
162        .into_iter()
163        .map(|(key_str, value)| {
164            let key = K::from_str(&key_str).map_err(serde::de::Error::custom)?;
165            Ok((key, value))
166        })
167        .collect::<Result<BTreeMap<_, _>, _>>()?;
168
169    Ok(Some(map))
170}

251    /// Get the value of a given meta field
252    pub fn meta<V: serde::de::DeserializeOwned + 'static>(
253        &self,
254        key: &str,
255    ) -> Result<Option<V>, anyhow::Error> {
256        let Some(str_value) = self.global.meta.get(key) else {
257            return Ok(None);
258        };
259        let res = serde_json::from_str(str_value)
260            .map(Some)
261            .context(format!("Decoding meta field '{key}' failed"));
262
263        // In the past we encoded some string fields as "just a string" without quotes,
264        // this code ensures that old meta values still parse since config is hard to
265        // change
266        if res.is_err() && std::any::TypeId::of::<V>() == std::any::TypeId::of::<String>() {
267            let string_ret = Box::new(str_value.clone());
268            let ret = unsafe {
269                // We can transmute a String to V because we know that V==String
270                std::mem::transmute::<Box<String>, Box<V>>(string_ret)
271            };
272            Ok(Some(*ret))
273        } else {
274            res
275        }
276    }

279    /// struct that when encoded as JSON shows the fields of module configs
280    /// instead of a consensus-encoded hex string.
281    ///
282    /// In case of unknown module the config value is a hex string.
283    pub fn to_json(&self) -> JsonClientConfig {
284        JsonClientConfig {
285            global: self.global.clone(),
286            modules: self

288                .iter()
289                .map(|(&module_instance_id, module_config)| {
290                    let module_config_json = JsonWithKind {

298                        })),
299                    };
300                    (module_instance_id, module_config_json)
301                })
302                .collect(),
303        }

582impl<M> ModuleInitRegistry<M>
583where
584    M: AsRef<dyn IDynCommonModuleInit + Send + Sync + 'static>,

586    #[deprecated(
587        note = "You probably want `available_decoders` to support missing module kinds. If you really want a strict behavior, use `decoders_strict`"
588    )]
589    pub fn decoders<'a>(
590        &self,
591        modules: impl Iterator<Item = (ModuleInstanceId, &'a ModuleKind)>,
592    ) -> anyhow::Result<ModuleDecoderRegistry> {
593        self.decoders_strict(modules)
594    }
595
596    /// Get decoders for `modules` and fail if any is unsupported
597    pub fn decoders_strict<'a>(
598        &self,
599        modules: impl Iterator<Item = (ModuleInstanceId, &'a ModuleKind)>,
600    ) -> anyhow::Result<ModuleDecoderRegistry> {
601        let mut decoders = BTreeMap::new();
602        for (id, kind) in modules {
603            let Some(init) = self.0.get(kind) else {
604                anyhow::bail!(
605                    "Detected configuration for unsupported module id: {id}, kind: {kind}"
606                )
607            };
608
609            decoders.insert(id, (kind.clone(), init.as_ref().decoder()));
610        }
611        Ok(ModuleDecoderRegistry::from(decoders))
612    }
613
614    /// Get decoders for `modules` and skip unsupported ones
615    pub fn available_decoders<'a>(
616        &self,
617        modules: impl Iterator<Item = (ModuleInstanceId, &'a ModuleKind)>,
618    ) -> anyhow::Result<ModuleDecoderRegistry> {
619        let mut decoders = BTreeMap::new();
620        for (id, kind) in modules {
621            let Some(init) = self.0.get(kind) else {
622                warn!(target: LOG_CORE, "Unsupported module id: {id}, kind: {kind}");
623                continue;
624            };
625
626            decoders.insert(id, (kind.clone(), init.as_ref().decoder()));
627        }
628        Ok(ModuleDecoderRegistry::from(decoders))
629    }
630}

678    pub fn is_kind(&self, kind: &ModuleKind) -> bool {
679        &self.kind == kind
680    }

682    pub fn kind(&self) -> &ModuleKind {
683        &self.kind
684    }

700/// Config for the server-side of a particular Federation module
701///
702/// See [`ClientModuleConfig`].
703#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
704pub struct ServerModuleConfig {
705    pub private: JsonWithKind,
706    pub consensus: ServerModuleConsensusConfig,
707}
708
709impl ServerModuleConfig {
710    pub fn from(private: JsonWithKind, consensus: ServerModuleConsensusConfig) -> Self {
711        Self { private, consensus }
712    }
713
714    pub fn to_typed<T: TypedServerModuleConfig>(&self) -> anyhow::Result<T> {
715        let private = serde_json::from_value(self.private.value().clone())?;
716        let consensus = <T::Consensus>::consensus_decode_whole(
717            &self.consensus.config[..],
718            &ModuleRegistry::default(),
719        )?;
720
721        Ok(TypedServerModuleConfig::from_parts(private, consensus))
722    }
723}
724
725/// Consensus-critical part of a server side module config
726pub trait TypedServerModuleConsensusConfig:
727    DeserializeOwned + Serialize + Encodable + Decodable
728{
729    fn kind(&self) -> ModuleKind;
730
731    fn version(&self) -> ModuleConsensusVersion;
732
733    fn from_erased(erased: &ServerModuleConsensusConfig) -> anyhow::Result<Self> {
734        Ok(Self::consensus_decode_whole(
735            &erased.config[..],
736            &ModuleRegistry::default(),
737        )?)
738    }
739}
740
741/// Module (server side) config, typed
742pub trait TypedServerModuleConfig: DeserializeOwned + Serialize {
743    /// Private for this federation member data that are security sensitive and
744    /// will be encrypted at rest
745    type Private: DeserializeOwned + Serialize;
746    /// Shared consensus-critical config
747    type Consensus: TypedServerModuleConsensusConfig;
748
749    /// Assemble from the three functionally distinct parts
750    fn from_parts(private: Self::Private, consensus: Self::Consensus) -> Self;
751
752    /// Split the config into its two functionally distinct parts
753    fn to_parts(self) -> (ModuleKind, Self::Private, Self::Consensus);
754
755    /// Turn the typed config into type-erased version
756    fn to_erased(self) -> ServerModuleConfig {
757        let (kind, private, consensus) = self.to_parts();
758
759        ServerModuleConfig {
760            private: JsonWithKind::new(
761                kind,
762                serde_json::to_value(private).expect("serialization can't fail"),
763            ),
764            consensus: ServerModuleConsensusConfig {
765                kind: consensus.kind(),
766                version: consensus.version(),
767                config: consensus.consensus_encode_to_vec(),
768            },
769        }
770    }
771}
