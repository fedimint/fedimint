46    pub async fn new(db: impl IRawDatabase + 'static, base_url: SafeUrl) -> anyhow::Result<Self> {
47        let db = Database::new(db, Default::default());
48
49        let mut clients = HashMap::<_, ClientHandleArc>::new();
50
51        for (federation_id, db) in load_federation_client_databases(&db).await {
52            let mut client_builder = Client::builder(db).await?;
53            client_builder.with_module(LightningClientInit::default());
54            client_builder.with_module(MintClientInit);
55            client_builder.with_primary_module_kind(ModuleKind::from_static_str("mint"));
56            let client = client_builder
57                .open(fedimint_client::RootSecret::LegacyDoubleDerive(
58                    Self::default_secret(),
59                ))
60                .await?;
61            clients.insert(federation_id, Arc::new(client));
62        }
63
64        Ok(Self {
65            db,
66            clients: Arc::new(RwLock::new(clients)),
67            invoice_generated: Arc::new(Default::default()),
68            base_url,
69        })
70    }

73    /// the client with externally supplied key material and to track

75    fn default_secret() -> DerivableSecret {
76        DerivableSecret::new_root(&[], &[])

79    pub async fn register_federation(
80        &self,
81        invite_code: &InviteCode,
82    ) -> Result<FederationId, RecurringPaymentError> {
83        let federation_id = invite_code.federation_id();
84        info!("Registering federation {}", federation_id);
85
86        // We lock to prevent parallel join attempts
87        // TODO: lock per federation
88        let mut clients = self.clients.write().await;
89        if clients.contains_key(&federation_id) {
90            return Err(RecurringPaymentError::FederationAlreadyRegistered(
91                federation_id,
92            ));
93        }
94
95        // We don't know if joining will succeed or be interrupted. We use a random DB
96        // prefix to initialize the client and only write the prefix to the DB if that
97        // succeeds. If it fails we end up with some orphaned data in the DB, if it ever
98        // becomes a problem we can clean it up later.
99        let client_db_prefix = FederationDbPrefix::random();
100        let client_db = open_client_db(&self.db, client_db_prefix);
101
102        match Self::join_federation_static(client_db, invite_code).await {
103            Ok(client) => {
104                try_add_federation_database(&self.db, federation_id, client_db_prefix)
105                    .await
106                    .expect("We hold a global lock, no parallel joining can happen");
107                clients.insert(federation_id, client);
108                Ok(federation_id)
109            }
110            Err(e) => {
111                // TODO: clean up DB?
112                Err(e)
113            }
114        }
115    }

117    async fn join_federation_static(
118        client_db: Database,
119        invite_code: &InviteCode,
120    ) -> Result<ClientHandleArc, RecurringPaymentError> {
121        let mut client_builder = Client::builder(client_db)
122            .await
123            .map_err(RecurringPaymentError::JoiningFederationFailed)?;
124
125        client_builder.with_connector(Connector::default());
126        client_builder.with_module(LightningClientInit::default());
127        client_builder.with_module(MintClientInit);
128        client_builder.with_primary_module_kind(ModuleKind::from_static_str("mint"));
129
130        let client = client_builder
131            .preview(invite_code)
132            .await?
133            .join(fedimint_client::RootSecret::LegacyDoubleDerive(
134                Self::default_secret(),
135            ))
136            .await
137            .map_err(RecurringPaymentError::JoiningFederationFailed)?;
138        Ok(Arc::new(client))
139    }

141    pub async fn register_recurring_payment_code(
142        &self,
143        federation_id: FederationId,
144        payment_code_root_key: PaymentCodeRootKey,
145        protocol: RecurringPaymentProtocol,
146        meta: &str,
147    ) -> Result<String, RecurringPaymentError> {
148        // TODO: support BOLT12
149        if protocol != RecurringPaymentProtocol::LNURL {
150            return Err(RecurringPaymentError::UnsupportedProtocol(protocol));
151        }
152
153        // Ensure the federation is supported
154        self.get_federation_client(federation_id).await?;
155
156        let payment_code = self.create_lnurl(payment_code_root_key.to_payment_code_id());
157        let payment_code_entry = PaymentCodeEntry {
158            root_key: payment_code_root_key,
159            federation_id,
160            protocol,
161            payment_code: payment_code.clone(),
162            variant: PaymentCodeVariant::Lnurl {
163                meta: meta.to_owned(),
164            },
165        };
166
167        let mut dbtx = self.db.begin_transaction().await;
168        if let Some(existing_code) = dbtx
169            .insert_entry(
170                &PaymentCodeKey {
171                    payment_code_id: payment_code_root_key.to_payment_code_id(),
172                },
173                &payment_code_entry,
174            )
175            .await
176        {
177            if existing_code != payment_code_entry {
178                return Err(RecurringPaymentError::PaymentCodeAlreadyExists(
179                    payment_code_root_key,
180                ));
181            }
182
183            dbtx.ignore_uncommitted();
184            return Ok(payment_code);
185        }
186
187        dbtx.insert_new_entry(
188            &PaymentCodeNextInvoiceIndexKey {
189                payment_code_id: payment_code_root_key.to_payment_code_id(),
190            },
191            &0,
192        )
193        .await;
194        dbtx.commit_tx_result().await?;
195
196        Ok(payment_code)
197    }

238    async fn create_bolt11_invoice(
239        &self,
240        payment_code_id: PaymentCodeId,
241        amount: Amount,
242    ) -> Result<Bolt11Invoice, RecurringPaymentError> {
243        // Invoices are valid for one day by default, might become dynamic with BOLT12
244        // support
245        const DEFAULT_EXPIRY_TIME: u64 = 60 * 60 * 24;
246
247        let payment_code = self.get_payment_code(payment_code_id).await?;
248        let invoice_index = self.get_next_invoice_index(payment_code_id).await;
249
250        let federation_client = self
251            .get_federation_client(payment_code.federation_id)
252            .await?;
253        let federation_client_ln_module = federation_client
254            .get_first_module::<LightningClientModule>()
255            .map_err(|e| {
256                warn!("No compatible lightning module found {e}");
257                RecurringPaymentError::NoLightningModuleFound
258            })?;
259
260        let gateway = federation_client_ln_module
261            .get_gateway(None, false)
262            .await?
263            .ok_or(RecurringPaymentError::NoGatewayFound)?;
264
265        let lnurl_meta = match payment_code.variant {
266            PaymentCodeVariant::Lnurl { meta } => meta,
267        };
268        let meta_hash = Sha256(sha256::Hash::hash(lnurl_meta.as_bytes()));
269        let description = Bolt11InvoiceDescription::Hash(meta_hash);
270
271        // TODO: ideally creating the invoice would take a dbtx as argument so we don't
272        // get holes in our used indexes in case this function fails/is cancelled
273        let (operation_id, invoice, _preimage) = federation_client_ln_module
274            .create_bolt11_invoice_for_user_tweaked(
275                amount,
276                description,
277                Some(DEFAULT_EXPIRY_TIME),
278                payment_code.root_key.0,
279                invoice_index,
280                serde_json::Value::Null,
281                Some(gateway),
282            )
283            .await?;
284
285        let mut dbtx = self.db.begin_transaction().await;
286        dbtx.insert_new_entry(
287            &PaymentCodeInvoiceKey {
288                payment_code_id,
289                index: invoice_index,
290            },
291            &PaymentCodeInvoiceEntry {
292                operation_id,
293                invoice: PaymentCodeInvoice::Bolt11(invoice.clone()),
294            },
295        )
296        .await;
297
298        let invoice_generated_notifier = self.invoice_generated.clone();
299        dbtx.on_commit(move || {
300            invoice_generated_notifier.notify_waiters();
301        });
302        dbtx.commit_tx().await;
303
304        await_invoice_confirmed(&federation_client_ln_module, operation_id).await?;
305
306        Ok(invoice)
307    }

321    pub async fn await_invoice_index_generated(
322        &self,
323        payment_code_id: PaymentCodeId,
324        invoice_index: u64,
325    ) -> Result<PaymentCodeInvoiceEntry, RecurringPaymentError> {
326        self.get_payment_code(payment_code_id).await?;
327
328        let mut notified = self.invoice_generated.notified();
329        loop {
330            let mut dbtx = self.db.begin_transaction_nc().await;
331            if let Some(invoice_entry) = dbtx
332                .get_value(&PaymentCodeInvoiceKey {
333                    payment_code_id,
334                    index: invoice_index,
335                })
336                .await
337            {
338                break Ok(invoice_entry);
339            };
340
341            notified.await;
342            notified = self.invoice_generated.notified();
343        }
344    }

346    async fn get_next_invoice_index(&self, payment_code_id: PaymentCodeId) -> u64 {
347        self.db
348            .autocommit(
349                |dbtx, _| {
350                    Box::pin(async move {
351                        let next_index = dbtx
352                            .get_value(&PaymentCodeNextInvoiceIndexKey { payment_code_id })
353                            .await
354                            .map(|index| index + 1)
355                            .unwrap_or(0);
356                        dbtx.insert_entry(
357                            &PaymentCodeNextInvoiceIndexKey { payment_code_id },
358                            &next_index,
359                        )
360                        .await;
361                        Result::<_, ()>::Ok(next_index)
362                    })
363                },
364                None,
365            )
366            .await
367            .expect("Loops forever and never returns errors internally")
368    }

387async fn await_invoice_confirmed(
388    ln_module: &ClientModuleInstance<'_, LightningClientModule>,
389    operation_id: OperationId,
390) -> Result<(), RecurringPaymentError> {
391    let mut operation_updated = ln_module
392        .subscribe_ln_receive(operation_id)
393        .await?
394        .into_stream();
395
396    while let Some(update) = operation_updated.next().await {
397        if matches!(update, LnReceiveState::WaitingForPayment { .. }) {
398            return Ok(());
399        }
400    }
401
402    Err(RecurringPaymentError::Other(anyhow!(
403        "BOLT11 invoice not confirmed"
404    )))
405}
