20pub async fn render<E>(api: &DynGatewayApi<E>, federations: &[FederationInfo]) -> Markup
21where
22    E: std::fmt::Display,
23{
24    let now = now();
25    let now_millis = now
26        .duration_since(UNIX_EPOCH)
27        .expect("Before unix epoch")
28        .as_millis() as u64;
29
30    let one_day_ago = now
31        .checked_sub(Duration::from_secs(60 * 60 * 24))
32        .expect("Before unix epoch");
33    let one_day_ago_millis = one_day_ago
34        .duration_since(UNIX_EPOCH)
35        .expect("Before unix epoch")
36        .as_millis() as u64;
37
38    // Fetch payment summary safely
39    let payment_summary = api
40        .handle_payment_summary_msg(PaymentSummaryPayload {
41            start_millis: one_day_ago_millis,
42            end_millis: now_millis,
43        })
44        .await;
45
46    render_tabs(payment_summary, federations)
47}

225pub async fn payment_log_fragment_handler<E>(
226    State(state): State<UiState<DynGatewayApi<E>>>,
227    _auth: UserAuth,
228    Query(params): Query<HashMap<String, String>>,
229) -> Html<String>
230where
231    E: std::fmt::Display + std::fmt::Debug,
232{
233    let federation_id = match params.get("federation_id") {
234        Some(v) => match v.parse::<FederationId>() {
235            Ok(id) => id,
236            Err(_) => {
237                return Html(
238                    html! {
239                        div class="alert alert-danger mb-0" { "Invalid federation ID." }
240                    }
241                    .into_string(),
242                );
243            }
244        },
245        None => {
246            return Html(
247                html! {
248                    div class="alert alert-warning mb-0" { "No federation selected." }
249                }
250                .into_string(),
251            );
252        }
253    };
254
255    let pagination_size = 10;
256
257    // Use end_position query param if present
258    let end_position = params
259        .get("end_position")
260        .and_then(|v| v.parse::<EventLogId>().ok());
261
262    let result = state
263        .api
264        .handle_payment_log_msg(PaymentLogPayload {
265            end_position,
266            pagination_size,
267            federation_id,
268            event_kinds: vec![],
269        })
270        .await;
271
272    Html(render_payment_log_result(&result, federation_id).into_string())
273}

275fn render_payment_log_result<E>(
276    result: &Result<PaymentLogResponse, E>,
277    federation_id: FederationId,
278) -> Markup
279where
280    E: std::fmt::Display,
281{
282    match result {
283        Ok(PaymentLogResponse(entries)) if !entries.is_empty() => {
284            // Compute next end_position as last entry position - 1
285            let next_end_position = entries.last().expect("Cannot be empty").id().checked_sub(1);
286
287            html! {
288                div {
289                    table class="table table-sm table-hover mb-2" {
290                        thead {
291                            tr {
292                                th { "Event Kind" }
293                                th { "Timestamp" }
294                                th { "Details" }
295                            }
296                        }
297                        tbody {
298                            @for (idx, entry) in entries.iter().enumerate() {
299                                tr {
300                                    td { code { (entry.as_raw().kind) } }
301                                    td { (format_timestamp(entry.as_raw().ts_usecs)) }
302                                    td {
303                                        button
304                                            class="btn btn-sm btn-outline-secondary"
305                                            type="button"
306                                            onclick=(format!(
307                                                "document.getElementById('payment-details-{}').classList.toggle('d-none');",
308                                                idx
309                                            ))
310                                        {
311                                            "Details"
312                                        }
313                                    }
314                                }
315
316                                tr id=(format!("payment-details-{}", idx)) class="d-none" {
317                                    td colspan="3" {
318                                        pre class="bg-dark text-light p-3 rounded small mb-0" {
319                                            (serde_json::to_string_pretty(entry).unwrap_or_else(|_| "<invalid json>".to_string()))
320                                        }
321                                    }
322                                }
323                            }
324                        }
325                    }
326
327                    @if let Some(next_pos) = next_end_position {
328                        div class="d-flex justify-content-end" {
329                            button
330                                class="btn btn-sm btn-outline-primary"
331                                type="button"
332                                hx-get=(PAYMENT_LOG_ROUTE)
333                                hx-target="#payment-log-content"
334                                hx-include="closest form"
335                                hx-vals=(serde_json::json!({
336                                    "federation_id": federation_id.to_string(),
337                                    "end_position": next_pos
338                                }))
339                            {
340                                "Next"
341                            }
342                        }
343                    }
344                }
345            }
346        }
347        Ok(_) => html! {
348            div class="text-muted" { "No payment events found for this federation." }
349        },
350        Err(e) => html! {
351            div class="alert alert-danger mb-0" {
352                strong { "Failed to load payment log: " }
353                (e.to_string())
354            }
355        },
356    }
357}

359fn format_timestamp(ts_usecs: u64) -> String {
360    let secs = ts_usecs / 1_000_000;
361    let nanos = (ts_usecs % 1_000_000) * 1_000;
362
363    let ts = UNIX_EPOCH + Duration::new(secs, nanos as u32);
364    let dt: chrono::DateTime<chrono::Utc> = ts.into();
365
366    dt.format("%Y-%m-%d %H:%M:%S UTC").to_string()
367}
368
369fn format_duration(d: Duration) -> String {
370    if d.as_secs() > 0 {
371        format!("{:.2}s", d.as_secs_f64())
372    } else {
373        format!("{} ms", d.as_millis())
374    }
