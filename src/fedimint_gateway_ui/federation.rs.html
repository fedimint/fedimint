160pub async fn leave_federation_handler<E: Display>(
161    State(state): State<UiState<DynGatewayApi<E>>>,
162    Path(id): Path<String>,
163    _auth: UserAuth,
164) -> impl IntoResponse {
165    let federation_id = FederationId::from_str(&id);
166    if let Ok(federation_id) = federation_id {
167        match state
168            .api
169            .handle_leave_federation(LeaveFedPayload { federation_id })
170            .await
171        {
172            Ok(info) => {
173                // Redirect back to dashboard after success
174                redirect_success(format!(
175                    "Successfully left {}",
176                    info.federation_name
177                        .unwrap_or("Unnamed Federation".to_string())
178                ))
179                .into_response()
180            }
181            Err(err) => {
182                redirect_error(format!("Failed to leave federation: {err}")).into_response()
183            }
184        }
185    } else {
186        redirect_error("Failed to leave federation: Invalid federation id".to_string())
187            .into_response()
188    }
189}

191pub async fn set_fees_handler<E: Display>(
192    State(state): State<UiState<DynGatewayApi<E>>>,
193    _auth: UserAuth,
194    Form(payload): Form<SetFeesPayload>,
195) -> impl IntoResponse {
196    tracing::info!("Received fees payload: {:?}", payload);
197
198    match state.api.handle_set_fees_msg(payload).await {
199        Ok(_) => redirect_success("Successfully set fees".to_string()).into_response(),
200        Err(err) => redirect_error(format!("Failed to update fees: {err}")).into_response(),
201    }
