240fn time_ago(t: SystemTime) -> String {
241    let now = fedimint_core::time::now();
242    let diff = match now.duration_since(t) {
243        Ok(d) => d,
244        Err(_) => Duration::from_secs(0),
245    };
246
247    let secs = diff.as_secs();
248
249    match secs {
250        0..=59 => format!("{} seconds ago", secs),
251        60..=3599 => format!("{} minutes ago", secs / 60),
252        _ => format!("{} hours ago", secs / 3600),
253    }
254}

256pub async fn leave_federation_handler<E: Display>(
257    State(state): State<UiState<DynGatewayApi<E>>>,
258    Path(id): Path<String>,
259    _auth: UserAuth,
260) -> impl IntoResponse {
261    let federation_id = FederationId::from_str(&id);
262    if let Ok(federation_id) = federation_id {
263        match state
264            .api
265            .handle_leave_federation(LeaveFedPayload { federation_id })
266            .await
267        {
268            Ok(info) => {
269                // Redirect back to dashboard after success
270                redirect_success(format!(
271                    "Successfully left {}",
272                    info.federation_name
273                        .unwrap_or("Unnamed Federation".to_string())
274                ))
275                .into_response()
276            }
277            Err(err) => {
278                redirect_error(format!("Failed to leave federation: {err}")).into_response()
279            }
280        }
281    } else {
282        redirect_error("Failed to leave federation: Invalid federation id".to_string())
283            .into_response()
284    }
285}

287pub async fn set_fees_handler<E: Display>(
288    State(state): State<UiState<DynGatewayApi<E>>>,
289    _auth: UserAuth,
290    Form(payload): Form<SetFeesPayload>,
291) -> impl IntoResponse {
292    tracing::info!("Received fees payload: {:?}", payload);
293
294    match state.api.handle_set_fees_msg(payload).await {
295        Ok(_) => redirect_success("Successfully set fees".to_string()).into_response(),
296        Err(err) => redirect_error(format!("Failed to update fees: {err}")).into_response(),
297    }
298}

300pub async fn deposit_address_handler<E: Display>(
301    State(state): State<UiState<DynGatewayApi<E>>>,
302    _auth: UserAuth,
303    Form(payload): Form<DepositAddressPayload>,
304) -> impl IntoResponse {
305    let markup = match state.api.handle_deposit_address_msg(payload).await {
306        Ok(address) => {
307            let code =
308                QrCode::new(address.to_qr_uri().as_bytes()).expect("Failed to generate QR code");
309            let qr_svg = code.render::<svg::Color>().build();
310
311            html! {
312                div class="card card-body bg-light d-flex flex-column align-items-center mt-2" {
313                    span class="fw-bold mb-3" { "Deposit Address:" }
314
315                    div class="d-flex flex-row align-items-center gap-3 flex-wrap" style="width: 100%;" {
316
317                        // Copyable input + text
318                        div class="d-flex flex-column flex-grow-1" style="min-width: 300px;" {
319                            input type="text"
320                                readonly
321                                class="form-control mb-2"
322                                style="text-align:left; font-family: monospace; font-size:1rem;"
323                                value=(address)
324                                onclick="copyToClipboard(this)"
325                            {}
326                            small class="text-muted" { "Click to copy" }
327                        }
328
329                        // QR code
330                        div class="border rounded p-2 bg-white d-flex justify-content-center align-items-center"
331                            style="width: 300px; height: 300px; min-width: 200px; min-height: 200px;"
332                        {
333                            (PreEscaped(format!(
334                                r#"<svg style="width: 100%; height: 100%; display: block;">{}</svg>"#,
335                                qr_svg.replace("width=", "data-width=").replace("height=", "data-height=")
336                            )))
337                        }
338                    }
339                }
340            }
341        }
342        Err(err) => {
343            html! {
344                div class="alert alert-danger mt-2" {
345                    "Failed to generate deposit address: " (err)
346                }
347            }
348        }
349    };
350    Html(markup.into_string())
