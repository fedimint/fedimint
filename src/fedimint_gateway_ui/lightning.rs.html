1279pub async fn transactions_fragment_handler<E>(
1280    State(state): State<UiState<DynGatewayApi<E>>>,
1281    _auth: UserAuth,
1282    Query(params): Query<HashMap<String, String>>,
1283) -> Html<String>
1284where
1285    E: std::fmt::Display + std::fmt::Debug,
1286{
1287    let now = fedimint_core::time::now();
1288    let end_secs = now
1289        .duration_since(std::time::UNIX_EPOCH)
1290        .expect("Time went backwards")
1291        .as_secs();
1292
1293    let start_secs = now
1294        .checked_sub(std::time::Duration::from_secs(60 * 60 * 24))
1295        .unwrap_or(now)
1296        .duration_since(std::time::UNIX_EPOCH)
1297        .expect("Time went backwards")
1298        .as_secs();
1299
1300    let parse = |key: &str| -> Option<u64> {
1301        params.get(key).and_then(|s| {
1302            chrono::NaiveDateTime::parse_from_str(s, "%Y-%m-%dT%H:%M:%S")
1303                .ok()
1304                .map(|dt| {
1305                    let dt_utc: chrono::DateTime<Utc> = Utc.from_utc_datetime(&dt);
1306                    dt_utc.timestamp() as u64
1307                })
1308        })
1309    };
1310
1311    let start_secs = parse("start_secs").unwrap_or(start_secs);
1312    let end_secs = parse("end_secs").unwrap_or(end_secs);
1313
1314    let transactions_result = state
1315        .api
1316        .handle_list_transactions_msg(ListTransactionsPayload {
1317            start_secs,
1318            end_secs,
1319        })
1320        .await;
1321
1322    Html(transactions_fragment_markup(&transactions_result, start_secs, end_secs).into_string())
1323}
